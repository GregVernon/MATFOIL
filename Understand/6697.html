21           SUBROUTINE QDES
22     C------------------------------------------------------
23     C     Mixed-Inverse design routine. Based on the 
24     C     same panel formulation as basic analysis method.
25     C------------------------------------------------------
26           INCLUDE 'XFOIL.INC'
27           CHARACTER*4 COMAND, COMOLD
28           LOGICAL LRECALC
29     C
30           CHARACTER*128 COMARG, ARGOLD
31           CHARACTER*1 CHKEY
32     C
33           REAL XBOX(2), YBOX(2)
34           REAL XSP(IBX), YSP(IBX,IPX), YSPD(IBX,IPX)
35     C
36           DIMENSION IINPUT(20)
37           DIMENSION RINPUT(20)
38           LOGICAL ERROR, LPLNEW
39     C
40           EXTERNAL NEWPLOTQ
41     C
42           SAVE COMOLD, ARGOLD
43     C
44     C---- statement function for compressible Karman-Tsien velocity
45           QCOMP(G) = G*(1.0-TKLAM) / (1.0 - TKLAM*(G/QINF)**2)
46     C
47     C
48           COMAND = '****'
49           COMARG = ' '
50           LRECALC = .FALSE.
51     C
52           IF(N.EQ.0) THEN
53            WRITE(*,*)
54            WRITE(*,*) '***  No airfoil available  ***'
55            RETURN
56           ENDIF
57     C
58           LSYM = .TRUE.
59     C
60     C---- number of sub-intervals for Qspec(s) plotting
61           NTQSPL = 1
62           IF(LQSLOP) NTQSPL = 8
63     C
64     C---- make sure a current solution exists
65           CALL SPECAL
66     C
67     C---- see if current Qspec, if any, didn't come from Full-Inverse
68           IF(NSP.NE.N) THEN
69             LQSPEC = .FALSE.
70             LIQSET = .FALSE.
71           ENDIF
72     C
73     C---- set alpha, etc corresponding to Q
74           ALGAM = ALFA
75           CLGAM = CL
76           CMGAM = CM
77     C
78     C---- set "old" speed distribution Q, arc length, and x/c,y/c arrays
79           CHX = XTE - XLE
80           CHY = YTE - YLE
81           CHSQ = CHX**2 + CHY**2
82           NSP = N
83           DO I=1, NSP
84             QGAMM(I) = GAM(I)
85             SSPEC(I) = S(I)/S(N)
86             XSPOC(I) = ((X(I)-XLE)*CHX + (Y(I)-YLE)*CHY)/CHSQ
87             YSPOC(I) = ((Y(I)-YLE)*CHX - (X(I)-XLE)*CHY)/CHSQ
88           ENDDO
89           SSPLE = SLE/S(N)
90     C
91           WRITE(*,1150) ALGAM/DTOR, CLGAM
92      1150 FORMAT(/' Current Q operating condition:'
93          &       /' alpha = ', F8.3, ' deg.      CL = ', F8.4 / )
94     C
95           IF(.NOT.LQSPEC) THEN
96     C----- initialize Qspec to "old" solution and notify user
97            NQSP = 1
98            KQTARG = 1
99            CALL GAMQSP(1)
100            WRITE(*,1155)
101            LQSPEC = .TRUE.
102           ENDIF
103     C
104     C---- initialize blowup parameters and plot Qspec(s)
105           CALL QPLINI(.TRUE.)
106           CALL QSPLOT
107     C
108     C
109     C====================================================
110     C---- start of menu loop
111      500  CONTINUE
112           COMOLD = COMAND
113           ARGOLD = COMARG
114     C
115      501  CALL ASKC('.QDES^',COMAND,COMARG)
116     C
117     C--------------------------------------------------------
118     C---- process previous command ?
119           IF(COMAND(1:1).EQ.'!') THEN
120             IF(COMOLD.EQ.'****') THEN
121               WRITE(*,*) 'Previous .QDES command not valid'
122               GO TO 501
123             ELSE
124               COMAND = COMOLD
125               COMARG = ARGOLD
126               LRECALC = .TRUE.
127             ENDIF
128           ELSE
129             LRECALC = .FALSE.
130           ENDIF
131     C
132           IF(COMAND.EQ.'    ') THEN
133     C----- just <return> was typed... clean up plotting and exit OPER
134            IF(LPLOT) CALL PLEND
135            LPLOT = .FALSE.
136            LQSYM = .FALSE.
137            LQSPPL = .FALSE.
138            CALL CLRZOOM
139            RETURN
140           ENDIF
141     C
142     C---- extract command line numeric arguments
143           DO I=1, 20
144             IINPUT(I) = 0
145             RINPUT(I) = 0.0
146           ENDDO
147           NINPUT = 0
148           CALL GETINT(COMARG,IINPUT,NINPUT,ERROR)
149           NINPUT = 0
150           CALL GETFLT(COMARG,RINPUT,NINPUT,ERROR)
151     C
152     C--------------------------------------------------------
153           IF(COMAND.EQ.'?   ') THEN
154            WRITE(*,1050)
155      1050  FORMAT(
156          &  /'   <cr>   Return to Top Level'
157          & //'   QSET   Reset Qspec <== Q'
158          & //'   Modi   Modify Qspec'
159          &  /'   MARK   Mark off target segment'
160          &  /'   SMOO   Smooth Qspec inside target segment'
161          &  /'   SLOP   Toggle modified-Qspec slope matching flag'
162          & //'   eXec i Execute mixed-inverse calculation'
163          &  /'   REST   Restore geometry from buffer airfoil'
164          &  /'   CPXX   CPxx endpoint constraint toggle'
165          & //'   Visc   Qvis overlay toggle'
166          &  /'   REFL   Reflected Qspec overlay toggle'
167          & //'   Plot   Plot Qspec (line) and Q (symbols)'
168          &  /'   Blow   Blowup plot region'
169          &  /'   Rese   Reset plot scale and origin'
170          &  /'   Wind     Plot window adjust via cursor and keys'
171          & //'   SIZE r Change absolute plot-object size'
172          &  /'  .ANNO   Annotate plot'
173          &  /'   HARD   Hardcopy current plot')
174     C
175     C--------------------------------------------------------
176           ELSEIF(COMAND.EQ.'Z   ') THEN
177            CALL USETZOOM(.TRUE.,.TRUE.)
178            CALL REPLOT(IDEV)
179     C
180     C--------------------------------------------------------
181           ELSEIF(COMAND.EQ.'U   ') THEN
182            CALL CLRZOOM
183            CALL REPLOT(IDEV)
184     C
185     C--------------------------------------------------------
186     C---- re-initialize Qspec to Q
187           ELSEIF(COMAND.EQ.'QSET') THEN
188            CALL GAMQSP(1)
189            CALL QPLINI(.FALSE.)
190            CALL QSPLOT
191           GO TO 500
192     C
193     C--------------------------------------------------------
194     C---- toggle Qvis plotting flag
195           ELSEIF(COMAND.EQ.'VISC' .OR.
196          &       COMAND.EQ.'V   '      ) THEN
197            LQVDES = .NOT.LQVDES
198            IF(LQVDES) THEN
199             WRITE(*,*) 'Qspec & Qvis will be plotted'
200            ELSE
201             WRITE(*,*) 'Only Qspec will be plotted'
202             CALL QPLINI(.FALSE.)
203            ENDIF
204            CALL QSPLOT
205           GO TO 500
206     C
207     C--------------------------------------------------------
208     C---- toggle reflected Qspec plotting flag
209           ELSEIF(COMAND.EQ.'REFL') THEN
210            LQREFL = .NOT.LQREFL
211            IF(LQREFL) THEN
212             WRITE(*,*) 'Reflected Qspec will be plotted'
213            ELSE
214             WRITE(*,*) 'Reflected Qspec will not be plotted'
215             CALL QPLINI(.FALSE.)
216            ENDIF
217            CALL QSPLOT
218           GO TO 500
219     C
220     C--------------------------------------------------------
221     C---- get target segment endpoints
222           ELSEIF(COMAND.EQ.'MARK') THEN
223            CALL IQSGET
224           GO TO 500
225     C
226     C--------------------------------------------------------
227     C---- modify Qspec
228           ELSEIF(COMAND.EQ.'MODI' .OR.
229          &       COMAND.EQ.'M   '      ) THEN
230     C----- make sure there is a Qspec(s) plot on the screen
231            IF(.NOT.LQSPPL) THEN
232             CALL QPLINI(.FALSE.)
233             CALL QSPLOT
234            ENDIF
235            CALL GETCOLOR(ICOL0)
236     C
237     C----- set up arrays for calling MODIFY
238            IFRST = 1
239            ILAST = NSP
240            NSIDE = 1
241            NLINE = NQSP
242            DO I = 1, NSP
243              ISP = NSP - I + 1
244              XSP(ISP) = 1.0 - SSPEC(I)
245              DO KQSP = 1, NQSP
246                GCOMP = QCOMP(QSPEC(I,KQSP))/QINF
247                YSP(ISP,KQSP) = QFAC*GCOMP
248              ENDDO
249            ENDDO
250            DO KQSP = 1, NQSP
251              CALL SEGSPL(YSP(1,KQSP),YSPD(1,KQSP),XSP,NSP)
252            ENDDO
253     C
254     C----- get the user's modifying input
255            XBOX(1) = XMARG
256            XBOX(2) = XPAGE-XMARG
257            YBOX(1) = YMARG
258            YBOX(2) = YPAGE-YMARG
259            CALL MODIFY(IBX,IFRST,ILAST,NSIDE,NLINE,
260          &             XSP,YSP,YSPD, LQSLOP,
261          &             ISP1,ISP2,ISMOD,KQSP,
262          &             XBOX,YBOX, XBOX,YBOX,SIZE,
263          &             XOFF,YOFF,XSF,YSF, 'RED',' ',
264          &             NEWPLOTQ )
265     C
266     C----- put modified info back into global arrays
267            IQMOD2 = NSP - ISP1 + 1
268            IQMOD1 = NSP - ISP2 + 1
269            DO I=1, NSP
270              ISP = NSP - I + 1
271              QSCOM =  QINF*YSP(ISP,KQSP)/QFAC
272              QSPEC(I,KQSP) = QINCOM(QSCOM,QINF,TKLAM)
273            ENDDO
274     C
275     C----- display new splined Qspec(s)
276            CALL SPLQSP(KQSP)
277            CALL NEWCOLORNAME('MAGENTA')
278            CALL QSPPLT(IQMOD1,IQMOD2,KQSP,NTQSPL)
279            CALL NEWCOLOR(ICOL0)
280     C
281     C----- print forces associated with modified Qspec(s)
282            CALL PLFLUSH
283            CALL CLCALC(N,X,Y,QSPEC(1,KQSP),W1,ALFA,MINF,QINF, XCMREF,YCMREF, 
284          &             CLQSP(KQSP),CMQSP(KQSP),CDPQ, CLQ_ALF,CLQ_MSQ)
285            WRITE(*,1200) CL,CM,CLQSP(KQSP),CMQSP(KQSP)
286           GO TO 500
287     C
288     C--------------------------------------------------------
289     C---- smooth Qspec within target segment, or entire Qspec if not marked off
290           ELSEIF(COMAND.EQ.'SMOO') THEN
291            CALL GETCOLOR(ICOL0)
292     C
293            KQSP = 1
294            CALL SMOOQ(IQ1,IQ2,KQSP)
295            CALL SPLQSP(KQSP)
296     C
297            CALL NEWCOLORNAME('magenta')
298            CALL QSPPLT(IQ1,IQ2,KQSP,NTQSPL)
299            CALL NEWCOLOR(ICOL0)
300            CALL PLFLUSH
301            LQSPPL = .FALSE.
302     C
303            CALL CLCALC(N,X,Y,QSPEC(1,KQSP),W1,ALFA,MINF,QINF, XCMREF,YCMREF, 
304          &             CLQSP(KQSP),CMQSP(KQSP),CDPQ, CLQ_ALF,CLQ_MSQ)
305             WRITE(*,1200) CL,CM,CLQSP(KQSP),CMQSP(KQSP)
306           GO TO 500
307     C
308     C--------------------------------------------------------
309     C---- toggle Qspec endpoint slope matching
310           ELSEIF(COMAND.EQ.'SLOP') THEN
311            LQSLOP = .NOT.LQSLOP
312            IF(LQSLOP) THEN
313              WRITE(*,*)
314          &    'Modified Qspec piece will be made tangent at endpoints'
315            ELSE
316              WRITE(*,*)
317          &   'Modified Qspec piece will not be made tangent at endpoints'
318            ENDIF
319           GO TO 500
320     C
321     C--------------------------------------------------------
322     C---- hardcopy replot
323           ELSEIF(COMAND.EQ.'HARD') THEN
324            IF(LPLOT) CALL PLEND
325            LPLOT = .FALSE.
326            CALL REPLOT(IDEVRP)
327           GO TO 500
328     C
329     C--------------------------------------------------------
330     C---- plot Qspec and Q distributions
331           ELSEIF(COMAND.EQ.'PLOT' .OR.
332          &       COMAND.EQ.'P   '      ) THEN
333            CALL QPLINI(.FALSE.)
334            CALL QSPLOT
335           GO TO 500
336     C
337     C--------------------------------------------------------
338     C---- get blowup parameters
339           ELSEIF(COMAND.EQ.'BLOW' .OR.
340          &       COMAND.EQ.'B   '      ) THEN
341            XWS = XWIND/SIZE
342            YWS = YWIND/SIZE
343            CALL OFFGET(XOFF,YOFF,XSF,YSF,XWS,YWS, .FALSE. , .TRUE. )
344            CALL QPLINI(.FALSE.)
345            CALL QSPLOT
346           GO TO 500
347     C
348     C--------------------------------------------------------
349     C---- reset blowup parameters and replot
350           ELSEIF(COMAND.EQ.'RESE' .OR.
351          &       COMAND.EQ.'R   '      ) THEN
352            CALL QPLINI(.TRUE.)
353            CALL QSPLOT
354           GO TO 500
355     C
356     C--------------------------------------------------------
357           ELSEIF(COMAND.EQ.'WIND' .OR.
358          &       COMAND.EQ.'W   '      ) THEN
359            XWS = XWIND/SIZE
360            YWS = YWIND/SIZE
361     C
362            WRITE(*,*) ' '
363            WRITE(*,*) 'Type I,O,P to In,Out,Pan with cursor...'
364     C
365      80    CALL QPLINI(.FALSE.)
366            CALL QSPLOT
367     C
368            CALL GETCURSORXY(XCRS,YCRS,CHKEY)
369     C
370     C----- do possible pan,zoom operations based on CHKEY
371            CALL KEYOFF(XCRS,YCRS,CHKEY, XWS,YWS, XOFF,YOFF,XSF,YSF, LPLNEW)
372     C
373            IF(LPLNEW) THEN
374             GO TO 80
375            ENDIF
376     C
377     C--------------------------------------------------------
378     C---- annotate plot
379           ELSEIF(COMAND.EQ.'ANNO') THEN
380            IF(LPLOT) THEN
381             CALL ANNOT(CH)
382            ELSE
383             WRITE(*,*) 'No active plot to annotate'
384            ENDIF
385           GO TO 500
386     C
387     C--------------------------------------------------------
388     C---- change plot size
389           ELSEIF(COMAND.EQ.'SIZE') THEN
390            IF(NINPUT.GE.1) THEN
391             SIZE = RINPUT(1)
392            ELSE
393             WRITE(*,*) 'Current plot size =', SIZE
394             CALL ASKR('Enter new plot size^',SIZE)
395            ENDIF
396     C
397            CALL QPLINI(.FALSE.)
398            CALL QSPLOT
399           GO TO 500
400     C
401     C--------------------------------------------------------
402     C---- toggle CPxx preservation constraints
403           ELSEIF(COMAND.EQ.'CPXX') THEN
404            LCPXX = .NOT.LCPXX
405            IF(LCPXX) THEN
406             WRITE(*,*) 'CPxx will be constrained'
407            ELSE
408             WRITE(*,*) 'CPxx will not be constrained'
409            ENDIF
410           GO TO 500
411     C
412     C--------------------------------------------------------
413     C---- set up for mixed-inverse calculation
414           ELSEIF(COMAND.EQ.'EXEC' .OR.
415          &       COMAND.EQ.'X   '      ) THEN
416            IF(.NOT.LIQSET) THEN
417             WRITE(*,*) '***  Must mark off target segment first  ***'
418             GO TO 500
419            ENDIF
420     C
421     C---- check if target segment includes stagnation point
422            IST = 0
423            DO I=IQ1, IQ2-1
424              IF(QGAMM(I).GE.0.0 .AND. QGAMM(I+1).LT.0.0) IST = I
425            ENDDO
426     C
427            IF(IST.NE.0) THEN 
428             WRITE(*,*)
429             WRITE(*,*) 'Target segment cannot include ',
430          &             'stagnation point in mixed-inverse.'
431             GO TO 500
432            ENDIF
433     C
434            KQSP = 1
435            CLSPEC = CLQSP(KQSP)
436     CCC      CALL ASKR('Enter specified CL^',CLSPEC)
437     C
438     C----- save current coordinates for restoration if requested
439            DO I=1, N
440              XB(I) = X(I)
441              YB(I) = Y(I)
442              SB(I) = S(I)
443              XBP(I) = XP(I)
444              YBP(I) = YP(I)
445            ENDDO
446            NB = N
447            LGSAME = .TRUE.
448     C
449            WRITE(*,*)
450            WRITE(*,*) 'Current airfoil saved in buffer airfoil'
451     C
452     C----- execute mixed-inverse calculation
453            IF(NINPUT.GE.1) THEN
454             NITERQ = IINPUT(1)
455            ELSE
456             CALL ASKI('Enter max number of iterations^',NITERQ)
457            ENDIF
458     C
459            CALL MIXED(KQSP,NITERQ)
460            ADEG = ALFA/DTOR
461     C
462     C----- spline new airfoil shape
463            CALL SCALC(X,Y,S,N)
464            CALL SPLIND(X,XP,S,N,-999.0,-999.0)
465            CALL SPLIND(Y,YP,S,N,-999.0,-999.0)
466            CALL NCALC(X,Y,S,N,NX,NY)
467            CALL LEFIND(SLE,X,XP,Y,YP,S,N)
468            XLE = SEVAL(SLE,X,XP,S,N)
469            YLE = SEVAL(SLE,Y,YP,S,N)
470            CHORD  = SQRT( (0.5*(X(1)+X(N)) - XLE)**2
471          &              + (0.5*(Y(1)+Y(N)) - YLE)**2 )
472            CALL TECALC
473            CALL APCALC
474     C
475            ALGAM = ALFA
476     C
477            NSP = N
478            DO I=1, N
479              QGAMM(I) = GAM(I)
480              SSPEC(I) = S(I)/S(N)
481            ENDDO
482            SSPLE = SLE/S(N)
483     C
484     C----- set inviscid surface speeds and calculate compressible Cp
485            DO I=1, N
486              QINV(I) = GAM(I)
487            ENDDO
488            CALL CPCALC(N,QINV,QINF,MINF,CPI)
489     C
490     C----- influence coefficients & other stuff is no longer valid for new airfoil
491            LGAMU = .FALSE.
492            LQINU = .FALSE.
493            LWAKE = .FALSE.
494            LQAIJ = .FALSE.
495            LADIJ = .FALSE.
496            LWDIJ = .FALSE.
497            LIPAN = .FALSE.
498            LVCONV = .FALSE.
499            LSCINI = .FALSE.
500     CCC      LBLINI = .FALSE.
501            LGSAME = .FALSE.
502     C
503     cc       CALL NAMMOD(NAME,1,1)
504     cc       CALL STRIP(NAME,NNAME)
505     C
506     C--------------------------------------------------------
507     C---- restore and spline old airfoil
508           ELSEIF(COMAND.EQ.'REST') THEN
509            DO I=1, N
510              X(I) = XB(I)
511              Y(I) = YB(I)
512            ENDDO
513            CALL SCALC(X,Y,S,N)
514            CALL SPLIND(X,XP,S,N,-999.0,-999.0)
515            CALL SPLIND(Y,YP,S,N,-999.0,-999.0)
516            CALL NCALC(X,Y,S,N,NX,NY)
517            CALL LEFIND(SLE,X,XP,Y,YP,S,N)
518            XLE = SEVAL(SLE,X,XP,S,N)
519            YLE = SEVAL(SLE,Y,YP,S,N)
520            CHORD  = SQRT( (0.5*(X(1)+X(N)) - XLE)**2
521          &              + (0.5*(Y(1)+Y(N)) - YLE)**2 )
522            CALL TECALC
523            CALL APCALC
524            LGAMU = .FALSE.
525            LQINU = .FALSE.
526            LGSAME = .TRUE.
527     C
528     cc       CALL NAMMOD(NAME,-1,1)
529     cc       CALL STRIP(NAME,NNAME)
530     C
531     C--------------------------------------------------------
532           ELSE
533            WRITE(*,1100) COMAND
534      1100  FORMAT(' Command ',A4,' not recognized.  Type a " ? " for list.')
535     C
536            COMAND = '****'
537           ENDIF
538     C
539           GO TO 500
540     C
541     C....................................................
542     C
543      1155 FORMAT(/' Qspec initialized to current Q.'/ )
544      1200 FORMAT(/' Q    :   CL =',F11.6, '    CM =',F11.6
545          &       /' Qspec:   CL =',F11.6, '    CM =',F11.6 )
546           END

556           SUBROUTINE QPLINI(LDEF)
557     C----------------------------------------------
558     C     Sets up Qspec(s) plot.
559     C     If LDEF=t, sets default offsets.
560     C----------------------------------------------
561           INCLUDE 'XFOIL.INC'
562           LOGICAL LDEF
563           LOGICAL LAIR
564     C
565     C---- number of x/c grid lines
566           PARAMETER (NG=10,NQ=20)
567           DIMENSION SSPG(-NG:NG), SLPG(-NG:NG), QSPG(-NQ:NQ)
568           DATA LMASK1, LMASK2, LMASK3 / -32640, -30584, -21846 /
569     C
570           INCLUDE 'XDES.INC'
571     C
572     C---- statement function for compressible Karman-Tsien velocity
573           QCOMP(G) = G*(1.0-TKLAM) / (1.0 - TKLAM*(G/QINF)**2)
574     C
575     C
576     C---- make room for airfoil plot if complex-mapping routine is being used
577           LAIR = NSP .EQ. NC1
578     C
579     C---- speed annotation increment
580           DQANN = 0.5
581     C
582     C---- find max and min speeds for current Qgamm and Qspec
583           QMIN = QGAMM(1)
584           QMAX = QGAMM(1)
585           DO 5 I=2, NSP
586             QMIN = MIN(QMIN,QGAMM(I))
587             QMAX = MAX(QMAX,QGAMM(I))
588         5 CONTINUE
589     C
590           DO 7 KQSP=1, NQSP
591             DO 72 I=2, NSP
592               QMIN = MIN(QMIN,QSPEC(I,KQSP))
593               QMAX = MAX(QMAX,QSPEC(I,KQSP))
594      72     CONTINUE
595      7    CONTINUE
596     C
597           QMIN = QCOMP(QMIN)/QINF
598           QMAX = QCOMP(QMAX)/QINF
599     C
600     C---- round up to bounding annotations
601           NMIN = INT(QMIN/DQANN) - 1
602           NMAX = INT(QMAX/DQANN) + 1
603     C
604           IF(LQREFL) THEN
605     C----- set limits so reflectes Qspec(s) also fits on plot
606            NMAX = MAX( ABS(NMIN) , ABS(NMAX) )
607            NMIN = -NMAX
608           ENDIF
609     C
610           QMIN = DQANN*FLOAT(NMIN)
611           QMAX = DQANN*FLOAT(NMAX)
612     C
613     C
614     C---- start new plot
615           CALL PLTINI
616     C
617     C---- speed plotting scale factor
618           QFAC = 1.0/(QMAX-QMIN)
619     C
620     C---- default offsets
621           IF(LDEF) THEN
622             XADD = 0.050
623             YADD = 0.075
624     C
625             XWMIN = MIN( XWIND - XMARG , XPAGE - 2.0*XMARG )
626             YWMIN = MIN( YWIND - YMARG , YPAGE - 2.0*YMARG )
627     C
628             XSF = (XWMIN/SIZE) / (1.0 + 2.0*XADD)
629             YSF = (YWMIN/SIZE) / (1.0 + 2.0*YADD)
630             CHQ = 0.7*CH * XSF
631             XOFF = -XADD - 2.0*CHQ/XSF
632             YOFF = -YADD + QMIN*QFAC
633           ENDIF
634     C
635           CALL SPLIND(XSPOC,W7,SSPEC,NSP,-999.0,-999.0)
636           CALL SPLIND(YSPOC,W8,SSPEC,NSP,-999.0,-999.0)
637     C
638           DO 11 IG=1, NG
639             XOC = FLOAT(IG)/FLOAT(NG)
640             SSP = SSPLE + (SSPEC(1)-SSPLE)*XOC
641             CALL SINVRT(SSP,XOC,XSPOC,W7,SSPEC,NSP)
642             SSPG(IG) = XMOD(1.0-SSP)
643     C
644             XOC = 0.1*FLOAT(IG)/FLOAT(NG)
645             SSP = SSPLE + (SSPEC(1)-SSPLE)*XOC
646             CALL SINVRT(SSP,XOC,XSPOC,W7,SSPEC,NSP)
647             SLPG(IG) = XMOD(1.0-SSP)
648        11 CONTINUE
649     C
650           SSPG(0) = XMOD(1.0-SSPLE)
651           SLPG(0) = XMOD(1.0-SSPLE)
652     C
653           DO 12 IG=-NG,-1
654             XOC = FLOAT(-IG)/FLOAT(NG)

1052     c        I = 1

1072             DO 15 I=1, NSP

1075     C
1076     ccc      ENDIF
1077     C
1078     C

1081            RETURN
1082           ENDIF

1085           SMOOL = 0.002*(W8(NSP) - W8(1))
1086     CCC   CALL ASKR('Enter Qspec smoothing length^',SMOOL)

1092             DSP = W8(I+1) - W8(I  )

1127            QSPP2 = W1(I)*QSPEC(I-1,KQSP)
1128          &       + W2(I)*QSPEC(I  ,KQSP)
1129          &       + W3(I)*QSPEC(I+1,KQSP)

1131            QSPEC(KQ1+1,KQSP) = QSPP1
1132            QSPEC(KQ2-1,KQSP) = QSPP2

1143     cc 40     CONTINUE
1144     cc      ENDIF
1145     C
1146           RETURN
1147           END

1150           FUNCTION QINCOM(QC,QINF,TKLAM)
1151     C-------------------------------------
1152     C     Sets incompressible speed from
1153     C     Karman-Tsien compressible speed
1154     C-------------------------------------
1155     C
1156           IF(TKLAM.LT.1.0E-4 .OR. ABS(QC).LT.1.0E-4) THEN
1157     C----- for nearly incompressible case or very small speed, use asymptotic
1158     C      expansion of singular quadratic formula to avoid numerical problems
1159            QINCOM = QC/(1.0 - TKLAM)
1160           ELSE
1161     C----- use quadratic formula for typical case
1162            TMP = 0.5*(1.0 - TKLAM)*QINF/(QC*TKLAM)
1163            QINCOM = QINF*TMP*(SQRT(1.0 + 1.0/(TKLAM*TMP**2)) - 1.0)
1164           ENDIF
1165           RETURN
1166           END

1241           SUBROUTINE MIXED(KQSP,NITERQ)
1242     C-------------------------------------------------
1243     C     Performs a mixed-inverse calculation using 
1244     C     the specified surface speed array QSPEC.
1245     C-------------------------------------------------
1246           INCLUDE 'XFOIL.INC'
1247     C
1248     C---- distance of internal control point ahead of sharp TE
1249     C-    (fraction of smaller panel length adjacent to TE)
1250           BWT = 0.1
1251     C
1252           COSA = COS(ALFA)
1253           SINA = SIN(ALFA)
1254           CALL SCALC(X,Y,S,N)
1255     C
1256     C---- zero-out and set DOF shape functions
1257           DO 1 I=1, N
1258             QF0(I) = 0.0
1259             QF1(I) = 0.0
1260             QF2(I) = 0.0
1261             QF3(I) = 0.0
1262         1 CONTINUE
1263     C
1264     C---- set DOF shape functions and specified speed
1265           DO 2 I=IQ1, IQ2
1266             FS = (S(I)-S(IQ1)) / (S(IQ2)-S(IQ1))
1267     CCC        QF0(I) = (1.0-FS)**2
1268     CCC        QF1(I) = FS**2
1269             QF0(I) = 1.0 - FS
1270             QF1(I) = FS
1271             IF(LCPXX) THEN
1272              QF2(I) = EXP(-5.0*     FS )
1273              QF3(I) = EXP(-5.0*(1.0-FS))
1274             ELSE
1275              QF2(I) = 0.0
1276              QF3(I) = 0.0
1277             ENDIF
1278             GAM(I) = QSPEC(I,KQSP) + QDOF0*QF0(I) + QDOF1*QF1(I)
1279          &                         + QDOF2*QF2(I) + QDOF3*QF3(I)
1280         2 CONTINUE
1281     C
1282        99 CONTINUE
1283     C
1284     C---- perform Newton iterations on the new geometry
1285           DO 1000 ITER=1, NITERQ
1286     C
1287           DO 3 I=1, N+5
1288             DO 31 J=1, N+5
1289               Q(I,J) = 0.
1290        31   CONTINUE
1291         3 CONTINUE
1292     C
1293     C---- calculate normal direction vectors along which the nodes move
1294           CALL NCALC(X,Y,S,N,NX,NY)
1295     C
1296     C---- go over all nodes, setting up  Psi = Psi0  equations
1297           DO 20 I=1, N
1298             CALL PSILIN(I,X(I),Y(I),NX(I),NY(I),PSI,PSI_N,.TRUE.,.FALSE.)
1299     C
1300             DZDN(I) = DZDN(I) + PSI_N
1301     C
1302     C------ fill columns for specified geometry location
1303             DO 201 J=1, IQ1-1
1304               Q(I,J) = Q(I,J) + DZDG(J)
1305       201   CONTINUE
1306     C
1307     C------ fill columns for specified surface speed location
1308             DO 202 J=IQ1, IQ2
1309               Q(I,J) = Q(I,J) + DZDN(J)
1310       202   CONTINUE
1311     C
1312     C------ fill columns for specified geometry location
1313             DO 203 J=IQ2+1, N
1314               Q(I,J) = Q(I,J) + DZDG(J)
1315       203   CONTINUE
1316     C
1317     C------ set residual
1318             DQ(I) = PSIO - PSI
1319     C
1320     C------ fill global unknown columns
1321             Q(I,N+1) = Q(I,N+1) - 1.0
1322             Q(I,N+2) = Q(I,N+2) + Z_QDOF0
1323             Q(I,N+3) = Q(I,N+3) + Z_QDOF1
1324             Q(I,N+4) = Q(I,N+4) + Z_QDOF2
1325             Q(I,N+5) = Q(I,N+5) + Z_QDOF3
1326        20 CONTINUE
1327     C
1328     C---- set up Kutta condition
1329           DQ(N+1) = -( GAM(1) + GAM(N) )
1330           CALL GAMLIN(N+1,1,1.0)
1331           CALL GAMLIN(N+1,N,1.0)
1332     C
1333           IF(SHARP) THEN
1334     C----- set zero internal velocity in TE corner 
1335     C
1336     C----- set TE bisector angle
1337            AG1 = ATAN2(-YP(1),-XP(1)    )
1338            AG2 = ATANC( YP(N), XP(N),AG1)
1339            ABIS = 0.5*(AG1+AG2)
1340            CBIS = COS(ABIS)
1341            SBIS = SIN(ABIS)
1342     C
1343     C----- minimum panel length adjacent to TE
1344            DS1 = SQRT( (X(1)-X(2)  )**2 + (Y(1)-Y(2)  )**2 )
1345            DS2 = SQRT( (X(N)-X(N-1))**2 + (Y(N)-Y(N-1))**2 )
1346            DSMIN = MIN( DS1 , DS2 )
1347     C
1348     C----- control point on bisector just ahead of TE point
1349            XBIS = XTE - BWT*DSMIN*CBIS
1350            YBIS = YTE - BWT*DSMIN*SBIS
1351     ccc       write(*,*) xbis, ybis
1352     C
1353     C----- set velocity component along bisector line
1354            CALL PSILIN(0,XBIS,YBIS,-SBIS,CBIS,PSI,QBIS,.FALSE.,.TRUE.)
1355     C
1356     CCC--- RES = DQDGj*Gamj + DQDMj*Massj + QINF*(COSA*CBIS + SINA*SBIS)
1357            RES = QBIS
1358     C
1359            DO J=1, N+5
1360              Q(N,J) = 0.
1361            ENDDO
1362     C
1363     C----- dRes/dgamj
1364            DO J=1, N
1365              CALL GAMLIN(N,J, DQDG(J) )
1366              Q(N,J) = DQDG(J)
1367            ENDDO
1368     C
1369     C----- dRes/dPsio
1370            Q(N,N+1) = 0.
1371     C
1372     C----- -dRes/dUinf
1373            DQ(N) = -RES
1374           ENDIF
1375     C
1376     C---- pinned IQ1 point condition
1377           Q(N+2,IQ1) = 1.0
1378           DQ(N+2) = 0.0
1379     C
1380     C---- pinned IQ2 point condition
1381           Q(N+3,IQ2) = 1.0
1382           DQ(N+3) = 0.0
1383     C
1384           IF(IQ1.GT.1 .AND. LCPXX) THEN
1385     C----- speed regularity IQ1 condition
1386            RES = GAM(IQ1-1)      - 2.0*  GAM(IQ1)      +   GAM(IQ1+1)
1387          &  - (QSPEC(IQ1-1,KQSP) - 2.0*QSPEC(IQ1,KQSP) + QSPEC(IQ1+1,KQSP) )
1388            CALL GAMLIN(N+4,IQ1-1, 1.0)
1389            CALL GAMLIN(N+4,IQ1  ,-2.0)
1390            CALL GAMLIN(N+4,IQ1+1, 1.0)
1391            DQ(N+4) = -RES
1392           ELSE
1393     C----- zero DOF condition
1394            Q(N+4,N+4) = 1.0
1395            DQ(N+4) = -QDOF2
1396           ENDIF
1397     C
1398           IF(IQ2.LT.N .AND. LCPXX) THEN
1399     C----- speed regularity IQ2 condition
1400            RES = GAM(IQ2-1)      - 2.0*  GAM(IQ2)      +   GAM(IQ2+1)
1401          &  - (QSPEC(IQ2-1,KQSP) - 2.0*QSPEC(IQ2,KQSP) + QSPEC(IQ2+1,KQSP) )
1402            CALL GAMLIN(N+5,IQ2-1, 1.0)
1403            CALL GAMLIN(N+5,IQ2  ,-2.0)
1404            CALL GAMLIN(N+5,IQ2+1, 1.0)
1405            DQ(N+5) = -RES
1406           ELSE
1407     C----- zero DOF condition
1408            Q(N+5,N+5) = 1.0
1409            DQ(N+5) = -QDOF3
1410           ENDIF
1411     C
1412           CALL GAUSS(IQX,N+5,Q,DQ,1)
1413     C
1414           INMAX = 0
1415           IGMAX = 0
1416           DNMAX = 0.0
1417           DGMAX = 0.0
1418     C
1419     C---- update surface speed GAM before target segment
1420           DO 100 I=1, IQ1-1
1421             GAM(I) = GAM(I) + DQ(I)
1422             IF(ABS(DQ(I)) .GT. ABS(DGMAX)) THEN
1423              DGMAX = DQ(I)
1424              IGMAX = I
1425             ENDIF
1426       100 CONTINUE
1427     C
1428     C---- update panel nodes inside target segment
1429           DO 110 I=IQ1, IQ2
1430             X(I) = X(I) + NX(I)*DQ(I)
1431             Y(I) = Y(I) + NY(I)*DQ(I)
1432             IF(ABS(DQ(I)) .GT. ABS(DNMAX)) THEN
1433              DNMAX = DQ(I)
1434              INMAX = I
1435             ENDIF
1436       110 CONTINUE
1437     C
1438     C---- update surface speed GAM after target segment
1439           DO 120 I=IQ2+1, N
1440             GAM(I) = GAM(I) + DQ(I)
1441             IF(ABS(DQ(I)) .GT. ABS(DGMAX)) THEN
1442              DGMAX = DQ(I)
1443              IGMAX = I
1444             ENDIF
1445       120 CONTINUE
1446     C
1447     C---- update gloabal variables
1448           PSIO  = PSIO  + DQ(N+1)
1449           QDOF0 = QDOF0 + DQ(N+2)
1450           QDOF1 = QDOF1 + DQ(N+3)
1451           QDOF2 = QDOF2 + DQ(N+4)
1452           QDOF3 = QDOF3 + DQ(N+5)
1453     C
1454           COSA = COS(ALFA)
1455           SINA = SIN(ALFA)
1456           CALL SCALC(X,Y,S,N)
1457     C
1458     C---- set correct surface speed over target segment including DOF contributions
1459           DO 140 I=IQ1, IQ2
1460             GAM(I) = QSPEC(I,KQSP) + QDOF0*QF0(I) + QDOF1*QF1(I)
1461          &                         + QDOF2*QF2(I) + QDOF3*QF3(I)
1462       140 CONTINUE
1463     C
1464     C---- update everything else
1465           CALL TECALC
1466           CALL CLCALC(N,X,Y,GAM,GAM_A,ALFA,MINF,QINF, XCMREF,YCMREF,
1467          &            CL,CM,CDP, CL_ALF,CL_MSQ)
1468           WRITE(*,2000) DNMAX,INMAX,DGMAX,IGMAX,CL
1469          &             ,DQ(N+2),DQ(N+3)
1470          &             ,DQ(N+4),DQ(N+5)
1471      2000 FORMAT(/' dNmax =',E10.3,I4,'   dQmax =',E10.3,I4,'    CL =',F7.4
1472          &       /' dQf1  =',E10.3,4X,'   dQf2  =',E10.3
1473          &       /' dQf3  =',E10.3,4X,'   dQf4  =',E10.3)
1474     C
1475           IF(ABS(DNMAX).LT.5.0E-5 .AND. ABS(DGMAX).LT.5.0E-4) THEN
1476            WRITE(*,*)
1477            WRITE(*,*) 'New current airfoil generated'
1478            WRITE(*,*) 'Old buffer  airfoil unchanged'
1479            RETURN
1480           ENDIF
1481     C
1482      1000 CONTINUE
1483           WRITE(*,*) 'Not quite converged.  Can EXEC again if necessary.'
1484           RETURN
1485     C
1486           END

1489           SUBROUTINE GAMLIN(I,J,COEF)
1490     C-------------------------------------------------------------------
1491     C     Adds on Jacobian entry for point I due to node speed GAM at J.
1492     C     GAM is either a local unknown if outside target segment,
1493     C     or dependent on global Qspec DOF's if inside target segment.
1494     C-------------------------------------------------------------------
1495           INCLUDE 'XFOIL.INC'
1496     C
1497           IF(J.GE.IQ1 .AND. J.LE.IQ2) THEN
1498     C----- inside target segment
1499            Q(I,N+2) = Q(I,N+2) + COEF*QF0(J)
1500            Q(I,N+3) = Q(I,N+3) + COEF*QF1(J)
1501            Q(I,N+4) = Q(I,N+4) + COEF*QF2(J)
1502            Q(I,N+5) = Q(I,N+5) + COEF*QF3(J)
1503           ELSE
1504     C----- outside target segment
1505            Q(I,J) = Q(I,J) + COEF
1506           ENDIF
1507           RETURN
