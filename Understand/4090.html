35           SUBROUTINE AXSET( HK1,    T1,    RT1,    A1,
36          &                  HK2,    T2,    RT2,    A2,  ACRIT, IDAMPV,
37          &           AX, AX_HK1, AX_T1, AX_RT1, AX_A1,
38          &               AX_HK2, AX_T2, AX_RT2, AX_A2 )
39     C----------------------------------------------------------
40     C     Returns average amplification AX over interval 1..2
41     C----------------------------------------------------------
42     C
43     cC==========================
44     cC---- 1st-order -- based on "1" quantities only
45     c      CALL DAMPL( HK1, T1, RT1, AX1, AX1_HK1, AX1_T1, AX1_RT1 )
46     c      AX2_HK2 = 0.0
47     c      AX2_T2  = 0.0
48     c      AX2_RT2 = 0.0
49     cC
50     c      AX1_A1 = 0.0
51     c      AX2_A2 = 0.0
52     cC
53     c      AX     = AX1
54     c      AX_AX1 = 1.0
55     c      AX_AX2 = 0.0
56     cC
57     c      ARG = MIN( 20.0*(ACRIT-A1) , 20.0 )
58     c      EXN    = EXP(-ARG)
59     c      EXN_A1 = 20.0*EXN
60     c      EXN_A2 = 0.
61     cC
62     c      DAX    = EXN   * 0.0004/T1
63     c      DAX_A1 = EXN_A1* 0.0004/T1
64     c      DAX_A2 = 0.
65     c      DAX_T1 = -DAX/T1
66     c      DAX_T2 = 0.
67     C
68     C==========================
69     C---- 2nd-order
70           IF(IDAMPV.EQ.0) THEN
71            CALL DAMPL( HK1, T1, RT1, AX1, AX1_HK1, AX1_T1, AX1_RT1 )
72            CALL DAMPL( HK2, T2, RT2, AX2, AX2_HK2, AX2_T2, AX2_RT2 )
73           ELSE
74            CALL DAMPL2( HK1, T1, RT1, AX1, AX1_HK1, AX1_T1, AX1_RT1 )
75            CALL DAMPL2( HK2, T2, RT2, AX2, AX2_HK2, AX2_T2, AX2_RT2 )
76           ENDIF
77     C
78     CC---- simple-average version
79     C      AXA = 0.5*(AX1 + AX2)
80     C      IF(AXA .LE. 0.0) THEN
81     C       AXA = 0.0
82     C       AXA_AX1 = 0.0
83     C       AXA_AX2 = 0.0
84     C      ELSE
85     C       AXA_AX1 = 0.5
86     C       AXA_AX2 = 0.5
87     C      ENDIF
88     C
89     C---- rms-average version (seems a little better on coarse grids)
90           AXSQ = 0.5*(AX1**2 + AX2**2)
91           IF(AXSQ .LE. 0.0) THEN
92            AXA = 0.0
93            AXA_AX1 = 0.0
94            AXA_AX2 = 0.0
95           ELSE
96            AXA = SQRT(AXSQ)
97            AXA_AX1 = 0.5*AX1/AXA
98            AXA_AX2 = 0.5*AX2/AXA
99           ENDIF
100     C
101     C----- small additional term to ensure  dN/dx > 0  near  N = Ncrit
102            ARG = MIN( 20.0*(ACRIT-0.5*(A1+A2)) , 20.0 )
103            IF(ARG.LE.0.0) THEN
104             EXN    = 1.0
105     CC      EXN_AC = 0.
106             EXN_A1 = 0.
107             EXN_A2 = 0.
108            ELSE
109             EXN    = EXP(-ARG)
110     CC      EXN_AC = -20.0    *EXN
111             EXN_A1 =  20.0*0.5*EXN
112             EXN_A2 =  20.0*0.5*EXN
113            ENDIF
114     C
115            DAX    = EXN    * 0.002/(T1+T2)
116     CC     DAX_AC = EXN_AC * 0.002/(T1+T2)
117            DAX_A1 = EXN_A1 * 0.002/(T1+T2)
118            DAX_A2 = EXN_A2 * 0.002/(T1+T2)
119            DAX_T1 = -DAX/(T1+T2)
120            DAX_T2 = -DAX/(T1+T2)
121     C
122     c
123     c        DAX    = 0.
124     c        DAX_A1 = 0.
125     c        DAX_A2 = 0.
126     c        DAX_AC = 0.
127     c        DAX_T1 = 0.
128     c        DAX_T2 = 0.
129     C==========================
130     C
131           AX     = AXA             + DAX
132     C
133           AX_HK1 = AXA_AX1*AX1_HK1
134           AX_T1  = AXA_AX1*AX1_T1  + DAX_T1
135           AX_RT1 = AXA_AX1*AX1_RT1
136           AX_A1  =                   DAX_A1
137     C
138           AX_HK2 = AXA_AX2*AX2_HK2
139           AX_T2  = AXA_AX2*AX2_T2  + DAX_T2
140           AX_RT2 = AXA_AX2*AX2_RT2
141           AX_A2  =                   DAX_A2
142     C
143           RETURN
144           END

147     c      SUBROUTINE TRCHEK1
148     cC-------------------------------------------------
149     cC     Checks if transition occurs in the current
150     cC     interval 1..2  (IBL-1...IBL) on side IS.
151     cC
152     cC     Old first-order version. 
153     cC
154     cC     Growth rate is evaluated at the upstream 
155     cC     point "1". The discrete amplification 
156     cC     equation is
157     cC
158     cC       Ncrit - N(X1)     
159     cC       -------------  =  N'(X1)
160     cC          XT - X1        
161     cC
162     cC     which can be immediately solved for 
163     cC     the transition location XT.
164     cC-------------------------------------------------
165     c      INCLUDE 'XBL.INC'
166     cC
167     cC---- calculate AMPL2 value
168     c      CALL AXSET( HK1,    T1,    RT1, AMPL1,
169     c     &            HK2,    T2,    RT2, AMPL2,  AMCRIT, IDAMPV,
170     c     &     AX, AX_HK1, AX_T1, AX_RT1, AX_A1,
171     c     &         AX_HK2, AX_T2, AX_RT2, AX_A2 )
172     c      AMPL2 = AMPL1 + AX*(X2-X1)
173     cC
174     cC---- test for free or forced transition
175     c      TRFREE = AMPL2.GE.AMCRIT
176     c      TRFORC = XIFORC.GT.X1 .AND. XIFORC.LE.X2
177     cC
178     cC---- set transition interval flag
179     c      TRAN = TRFORC .OR. TRFREE
180     cC
181     cC---- if no transition yet, just return
182     c      IF(.NOT.TRAN) RETURN
183     cC
184     cC---- resolve if both forced and free transition
185     c      IF(TRFREE .AND. TRFORC) THEN
186     c       XT = (AMCRIT-AMPL1)/AX  +  X1
187     c       TRFORC = XIFORC .LT. XT
188     c       TRFREE = XIFORC .GE. XT
189     c      ENDIF
190     cC
191     c      IF(TRFORC) THEN
192     cC----- if forced transition, then XT is prescribed
193     c       XT = XIFORC
194     c       XT_A1 = 0.
195     c       XT_X1 = 0.
196     c       XT_T1 = 0.
197     c       XT_D1 = 0.
198     c       XT_U1 = 0.
199     c       XT_X2 = 0.
200     c       XT_T2 = 0.
201     c       XT_D2 = 0.
202     c       XT_U2 = 0.
203     c       XT_MS = 0.
204     c       XT_RE = 0.
205     c       XT_XF = 1.0
206     c      ELSE
207     cC----- if free transition, XT is related to BL variables
208     cC-     by the amplification equation
209     cC
210     c       XT    =  (AMCRIT-AMPL1)/AX     + X1
211     c       XT_AX = -(AMCRIT-AMPL1)/AX**2
212     cC
213     c       XT_A1 = -1.0/AX - (AMCRIT-AMPL1)/AX**2 * AX_A1
214     c       XT_X1 = 1.0
215     c       XT_T1 = XT_AX*(AX_HK1*HK1_T1 + AX_T1 + AX_RT1*RT1_T1)
216     c       XT_D1 = XT_AX*(AX_HK1*HK1_D1                        )
217     c       XT_U1 = XT_AX*(AX_HK1*HK1_U1         + AX_RT1*RT1_U1)
218     c       XT_X2 = 0.
219     c       XT_T2 = 0.
220     c       XT_D2 = 0.
221     c       XT_U2 = 0.
222     c       XT_MS = XT_AX*(AX_HK1*HK1_MS         + AX_RT1*RT1_MS)
223     c       XT_RE = XT_AX*(                        AX_RT1*RT1_RE)
224     c       XT_XF = 0.0
225     c      ENDIF
226     cC
227     c      RETURN
228     c      END

231           SUBROUTINE TRCHEK2
232     C----------------------------------------------------------------
233     C     New second-order version:  December 1994.
234     C
235     C     Checks if transition occurs in the current interval X1..X2.
236     C     If transition occurs, then set transition location XT, and 
237     C     its sensitivities to "1" and "2" variables.  If no transition, 
238     C     set amplification AMPL2.
239     C
240     C
241     C     Solves the implicit amplification equation for N2:
242     C
243     C       N2 - N1     N'(XT,NT) + N'(X1,N1)
244     C       -------  =  ---------------------
245     C       X2 - X1               2
246     C
247     C     In effect, a 2-point central difference is used between
248     C     X1..X2 (no transition), or X1..XT (transition).  The switch
249     C     is done by defining XT,NT in the equation above depending
250     C     on whether N2 exceeds Ncrit.
251     C
252     C  If N2<Ncrit:  NT=N2    , XT=X2                  (no transition)
253     C
254     C  If N2>Ncrit:  NT=Ncrit , XT=(Ncrit-N1)/(N2-N1)  (transition)
255     C
256     C
257     C----------------------------------------------------------------
258           INCLUDE 'XBL.INC'
259           DATA DAEPS / 5.0E-5 /
260     CCC   DATA DAEPS / 1.0D-12 /
261     C
262     C---- save variables and sensitivities at IBL ("2") for future restoration
263           DO 5 ICOM=1, NCOM
264             C2SAV(ICOM) = COM2(ICOM)
265         5 CONTINUE
266     C
267     C---- calculate average amplification rate AX over X1..X2 interval
268           CALL AXSET( HK1,    T1,    RT1, AMPL1,
269          &            HK2,    T2,    RT2, AMPL2,  AMCRIT, IDAMPV,
270          &     AX, AX_HK1, AX_T1, AX_RT1, AX_A1,
271          &         AX_HK2, AX_T2, AX_RT2, AX_A2 )
272     C
273     C---- set initial guess for iterate N2 (AMPL2) at X2
274           AMPL2 = AMPL1 + AX*(X2-X1)
275     C
276     C---- solve implicit system for amplification AMPL2
277           DO 100 ITAM=1, 30
278     C
279     C---- define weighting factors WF1,WF2 for defining "T" quantities from 1,2
280     C
281           IF(AMPL2 .LE. AMCRIT) THEN
282     C------ there is no transition yet,  "T" is the same as "2"
283             AMPLT    = AMPL2
284             AMPLT_A2 = 1.0
285             SFA    = 1.0
286             SFA_A1 = 0.
287             SFA_A2 = 0.
288           ELSE
289     C------ there is transition in X1..X2, "T" is set from N1, N2
290             AMPLT    = AMCRIT
291             AMPLT_A2 = 0.
292             SFA    = (AMPLT - AMPL1)/(AMPL2-AMPL1)
293             SFA_A1 = ( SFA  - 1.0  )/(AMPL2-AMPL1)
294             SFA_A2 = (      - SFA  )/(AMPL2-AMPL1)
295           ENDIF
296     C
297           IF(XIFORC.LT.X2) THEN
298             SFX    = (XIFORC - X1 )/(X2-X1)
299             SFX_X1 = (SFX    - 1.0)/(X2-X1)
300             SFX_X2 = (       - SFX)/(X2-X1)
301             SFX_XF =  1.0          /(X2-X1)
302           ELSE
303             SFX    = 1.0
304             SFX_X1 = 0.
305             SFX_X2 = 0.
306             SFX_XF = 0.
307           ENDIF
308     C
309     C---- set weighting factor from free or forced transition
310           IF(SFA.LT.SFX) THEN
311             WF2    = SFA
312             WF2_A1 = SFA_A1
313             WF2_A2 = SFA_A2
314             WF2_X1 = 0.
315             WF2_X2 = 0.
316             WF2_XF = 0.
317           ELSE
318             WF2    = SFX
319             WF2_A1 = 0.
320             WF2_A2 = 0.
321             WF2_X1 = SFX_X1
322             WF2_X2 = SFX_X2
323             WF2_XF = SFX_XF
324           ENDIF
325     C
326     C
327     C=====================
328     CC---- 1st-order (based on "1" quantites only, for testing)
329     C      WF2    = 0.0
330     C      WF2_A1 = 0.0
331     C      WF2_A2 = 0.0
332     C      WF2_X1 = 0.0
333     C      WF2_X2 = 0.0
334     C      WF2_XF = 0.0
335     C=====================
336     C
337           WF1    = 1.0 - WF2
338           WF1_A1 =     - WF2_A1
339           WF1_A2 =     - WF2_A2
340           WF1_X1 =     - WF2_X1
341           WF1_X2 =     - WF2_X2
342           WF1_XF =     - WF2_XF
343     C
344     C---- interpolate BL variables to XT
345           XT    = X1*WF1    + X2*WF2
346           TT    = T1*WF1    + T2*WF2
347           DT    = D1*WF1    + D2*WF2
348           UT    = U1*WF1    + U2*WF2
349     C
350           XT_A2 = X1*WF1_A2 + X2*WF2_A2
351           TT_A2 = T1*WF1_A2 + T2*WF2_A2
352           DT_A2 = D1*WF1_A2 + D2*WF2_A2
353           UT_A2 = U1*WF1_A2 + U2*WF2_A2
354     C
355     C---- temporarily set "2" variables from "T" for BLKIN
356           X2 = XT
357           T2 = TT
358           D2 = DT
359           U2 = UT
360     C
361     C---- calculate laminar secondary "T" variables HKT, RTT
362           CALL BLKIN
363     C
364           HKT    = HK2
365           HKT_TT = HK2_T2
366           HKT_DT = HK2_D2
367           HKT_UT = HK2_U2
368           HKT_MS = HK2_MS
369     C
370           RTT    = RT2
371           RTT_TT = RT2_T2
372           RTT_UT = RT2_U2
373           RTT_MS = RT2_MS
374           RTT_RE = RT2_RE
375     C
376     C---- restore clobbered "2" variables, except for AMPL2
377           AMSAVE = AMPL2
378           DO 8 ICOM=1, NCOM
379             COM2(ICOM) = C2SAV(ICOM)
380      8    CONTINUE
381           AMPL2 = AMSAVE
382     C
383     C---- calculate amplification rate AX over current X1-XT interval
384           CALL AXSET( HK1,    T1,    RT1, AMPL1,
385          &            HKT,    TT,    RTT, AMPLT,  AMCRIT, IDAMPV,
386          &     AX, AX_HK1, AX_T1, AX_RT1, AX_A1,
387          &         AX_HKT, AX_TT, AX_RTT, AX_AT )
388     C
389     C---- punch out early if there is no amplification here
390           IF(AX .LE. 0.0) GO TO 101
391     C
392     C---- set sensitivity of AX(A2)
393           AX_A2 = (AX_HKT*HKT_TT + AX_TT + AX_RTT*RTT_TT)*TT_A2
394          &      + (AX_HKT*HKT_DT                        )*DT_A2
395          &      + (AX_HKT*HKT_UT         + AX_RTT*RTT_UT)*UT_A2
396          &      +  AX_AT                                 *AMPLT_A2
397     C
398     C---- residual for implicit AMPL2 definition (amplification equation)
399           RES    = AMPL2 - AMPL1 - AX   *(X2-X1) 
400           RES_A2 = 1.0           - AX_A2*(X2-X1)
401     C
402           DA2 = -RES/RES_A2
403     C
404           RLX = 1.0
405           DXT = XT_A2*DA2
406     C
407           IF(RLX*ABS(DXT/(X2-X1)) .GT. 0.05) RLX = 0.05*ABS((X2-X1)/DXT)
408           IF(RLX*ABS(DA2)         .GT. 1.0 ) RLX = 1.0 *ABS(   1.0 /DA2)
409     C
410     C---- check if converged
411           IF(ABS(DA2) .LT. DAEPS) GO TO 101
412     C
413           IF((AMPL2.GT.AMCRIT .AND. AMPL2+RLX*DA2.LT.AMCRIT).OR.
414          &   (AMPL2.LT.AMCRIT .AND. AMPL2+RLX*DA2.GT.AMCRIT)    ) THEN
415     C------ limited Newton step so AMPL2 doesn't step across AMCRIT either way
416             AMPL2 = AMCRIT
417           ELSE
418     C------ regular Newton step
419             AMPL2 = AMPL2 + RLX*DA2
420           ENDIF
421     C
422      100  CONTINUE
423           WRITE(*,*) 'TRCHEK2: N2 convergence failed.'
424           WRITE(*,6700) X1, XT, X2, AMPL1, AMPLT, AMPL2, AX, DA2
425      6700 FORMAT(1X,'x:', 3F9.5,'  N:',3F7.3,'  Nx:',F8.3,'   dN:',E10.3)
426     C
427      101  CONTINUE
428     C
429     C
430     C---- test for free or forced transition
431           TRFREE = AMPL2 .GE. AMCRIT
432           TRFORC = XIFORC.GT.X1 .AND. XIFORC.LE.X2
433     C
434     C---- set transition interval flag
435           TRAN = TRFORC .OR. TRFREE
436     C
437           IF(.NOT.TRAN) RETURN
438     C
439     C---- resolve if both forced and free transition
440           IF(TRFREE .AND. TRFORC) THEN
441            TRFORC = XIFORC .LT. XT
442            TRFREE = XIFORC .GE. XT
443           ENDIF
444     C
445           IF(TRFORC) THEN
446     C----- if forced transition, then XT is prescribed,
447     C-     no sense calculating the sensitivities, since we know them...
448            XT = XIFORC
449            XT_A1 = 0.
450            XT_X1 = 0.
451            XT_T1 = 0.
452            XT_D1 = 0.
453            XT_U1 = 0.
454            XT_X2 = 0.
455            XT_T2 = 0.
456            XT_D2 = 0.
457            XT_U2 = 0.
458            XT_MS = 0.
459            XT_RE = 0.
460            XT_XF = 1.0
461            RETURN
462           ENDIF
463     C
464     C---- free transition ... set sensitivities of XT
465     C
466     C---- XT( X1 X2 A1 A2 XF ),  TT( T1 T2 A1 A2 X1 X2 XF),   DT( ...
467     CC    XT    = X1*WF1    + X2*WF2
468     CC    TT    = T1*WF1    + T2*WF2
469     CC    DT    = D1*WF1    + D2*WF2
470     CC    UT    = U1*WF1    + U2*WF2
471     C
472           XT_X1 =    WF1
473           TT_T1 =    WF1
474           DT_D1 =    WF1
475           UT_U1 =    WF1
476     C
477           XT_X2 =                WF2
478           TT_T2 =                WF2
479           DT_D2 =                WF2
480           UT_U2 =                WF2
481     C
482           XT_A1 = X1*WF1_A1 + X2*WF2_A1
483           TT_A1 = T1*WF1_A1 + T2*WF2_A1
484           DT_A1 = D1*WF1_A1 + D2*WF2_A1
485           UT_A1 = U1*WF1_A1 + U2*WF2_A1
486     C
487     CC    XT_A2 = X1*WF1_A2 + X2*WF2_A2
488     CC    TT_A2 = T1*WF1_A2 + T2*WF2_A2
489     CC    DT_A2 = D1*WF1_A2 + D2*WF2_A2
490     CC    UT_A2 = U1*WF1_A2 + U2*WF2_A2
491     C
492           XT_X1 = X1*WF1_X1 + X2*WF2_X1 + XT_X1
493           TT_X1 = T1*WF1_X1 + T2*WF2_X1
494           DT_X1 = D1*WF1_X1 + D2*WF2_X1
495           UT_X1 = U1*WF1_X1 + U2*WF2_X1
496     C
497           XT_X2 = X1*WF1_X2 + X2*WF2_X2 + XT_X2
498           TT_X2 = T1*WF1_X2 + T2*WF2_X2
499           DT_X2 = D1*WF1_X2 + D2*WF2_X2
500           UT_X2 = U1*WF1_X2 + U2*WF2_X2
501     C
502           XT_XF = X1*WF1_XF + X2*WF2_XF
503           TT_XF = T1*WF1_XF + T2*WF2_XF
504           DT_XF = D1*WF1_XF + D2*WF2_XF
505           UT_XF = U1*WF1_XF + U2*WF2_XF
506     C
507     C---- at this point, AX = AX( HK1, T1, RT1, A1, HKT, TT, RTT, AT )
508     C
509     C---- set sensitivities of AX( T1 D1 U1 A1 T2 D2 U2 A2 MS RE )
510           AX_T1 =  AX_HK1*HK1_T1 + AX_T1 + AX_RT1*RT1_T1
511          &      + (AX_HKT*HKT_TT + AX_TT + AX_RTT*RTT_TT)*TT_T1
512           AX_D1 =  AX_HK1*HK1_D1
513          &      + (AX_HKT*HKT_DT                        )*DT_D1
514           AX_U1 =  AX_HK1*HK1_U1         + AX_RT1*RT1_U1
515          &      + (AX_HKT*HKT_UT         + AX_RTT*RTT_UT)*UT_U1
516           AX_A1 =  AX_A1
517          &      + (AX_HKT*HKT_TT + AX_TT + AX_RTT*RTT_TT)*TT_A1
518          &      + (AX_HKT*HKT_DT                        )*DT_A1
519          &      + (AX_HKT*HKT_UT         + AX_RTT*RTT_UT)*UT_A1
520           AX_X1 = (AX_HKT*HKT_TT + AX_TT + AX_RTT*RTT_TT)*TT_X1
521          &      + (AX_HKT*HKT_DT                        )*DT_X1
522          &      + (AX_HKT*HKT_UT         + AX_RTT*RTT_UT)*UT_X1
523     C
524           AX_T2 = (AX_HKT*HKT_TT + AX_TT + AX_RTT*RTT_TT)*TT_T2
525           AX_D2 = (AX_HKT*HKT_DT                        )*DT_D2
526           AX_U2 = (AX_HKT*HKT_UT         + AX_RTT*RTT_UT)*UT_U2
527           AX_A2 =  AX_AT                                 *AMPLT_A2
528          &      + (AX_HKT*HKT_TT + AX_TT + AX_RTT*RTT_TT)*TT_A2
529          &      + (AX_HKT*HKT_DT                        )*DT_A2
530          &      + (AX_HKT*HKT_UT         + AX_RTT*RTT_UT)*UT_A2
531           AX_X2 = (AX_HKT*HKT_TT + AX_TT + AX_RTT*RTT_TT)*TT_X2
532          &      + (AX_HKT*HKT_DT                        )*DT_X2
533          &      + (AX_HKT*HKT_UT         + AX_RTT*RTT_UT)*UT_X2
534     C
535           AX_XF = (AX_HKT*HKT_TT + AX_TT + AX_RTT*RTT_TT)*TT_XF
536          &      + (AX_HKT*HKT_DT                        )*DT_XF
537          &      + (AX_HKT*HKT_UT         + AX_RTT*RTT_UT)*UT_XF
538     C
539           AX_MS =  AX_HKT*HKT_MS         + AX_RTT*RTT_MS
540          &      +  AX_HK1*HK1_MS         + AX_RT1*RT1_MS
541           AX_RE =                          AX_RTT*RTT_RE
542          &                               + AX_RT1*RT1_RE
543     C
544     C
545     C---- set sensitivities of residual RES
546     CCC   RES  = AMPL2 - AMPL1 - AX*(X2-X1)
547           Z_AX =               -    (X2-X1)
548     C
549           Z_A1 = Z_AX*AX_A1 - 1.0
550           Z_T1 = Z_AX*AX_T1
551           Z_D1 = Z_AX*AX_D1
552           Z_U1 = Z_AX*AX_U1
553           Z_X1 = Z_AX*AX_X1 + AX
554     C
555           Z_A2 = Z_AX*AX_A2 + 1.0
556           Z_T2 = Z_AX*AX_T2
557           Z_D2 = Z_AX*AX_D2
558           Z_U2 = Z_AX*AX_U2
559           Z_X2 = Z_AX*AX_X2 - AX
560     C
561           Z_XF = Z_AX*AX_XF
562           Z_MS = Z_AX*AX_MS
563           Z_RE = Z_AX*AX_RE
564     C
565     C---- set sensitivities of XT, with RES being stationary for A2 constraint
566           XT_A1 = XT_A1 - (XT_A2/Z_A2)*Z_A1
567           XT_T1 =       - (XT_A2/Z_A2)*Z_T1
568           XT_D1 =       - (XT_A2/Z_A2)*Z_D1
569           XT_U1 =       - (XT_A2/Z_A2)*Z_U1
570           XT_X1 = XT_X1 - (XT_A2/Z_A2)*Z_X1
571           XT_T2 =       - (XT_A2/Z_A2)*Z_T2
572           XT_D2 =       - (XT_A2/Z_A2)*Z_D2
573           XT_U2 =       - (XT_A2/Z_A2)*Z_U2
574           XT_X2 = XT_X2 - (XT_A2/Z_A2)*Z_X2
575           XT_MS =       - (XT_A2/Z_A2)*Z_MS
576           XT_RE =       - (XT_A2/Z_A2)*Z_RE
577           XT_XF = 0.0
578     C
579           RETURN
580           END

583           SUBROUTINE BLSYS
584     C------------------------------------------------------------------
585     C
586     C     Sets up the BL Newton system governing the current interval:
587     C
588     C     |       ||dA1|     |       ||dA2|       |     |
589     C     |  VS1  ||dT1|  +  |  VS2  ||dT2|   =   |VSREZ|
590     C     |       ||dD1|     |       ||dD2|       |     |
591     C              |dU1|              |dU2|
592     C              |dX1|              |dX2|
593     C
594     C        3x5    5x1         3x5    5x1          3x1
595     C
596     C     The system as shown corresponds to a laminar station
597     C     If TRAN, then  dS2  replaces  dA2
598     C     If TURB, then  dS1, dS2  replace  dA1, dA2
599     C
600     C------------------------------------------------------------------
601           IMPLICIT REAL(M)
602           INCLUDE 'XBL.INC'
603     C
604     C---- calculate secondary BL variables and their sensitivities
605           IF(WAKE) THEN
606            CALL BLVAR(3)
607            CALL BLMID(3)
608           ELSE IF(TURB.OR.TRAN) THEN
609            CALL BLVAR(2)
610            CALL BLMID(2)
611           ELSE
612            CALL BLVAR(1)
613            CALL BLMID(1)
614           ENDIF
615     C
616     C---- for the similarity station, "1" and "2" variables are the same
617           IF(SIMI) THEN
618            DO 3 ICOM=1, NCOM
619              COM1(ICOM) = COM2(ICOM)
620         3  CONTINUE
621           ENDIF
622     C
623     C---- set up appropriate finite difference system for current interval
624           IF(TRAN) THEN
625            CALL TRDIF
626           ELSE IF(SIMI) THEN
627            CALL BLDIF(0)
628           ELSE IF(.NOT.TURB) THEN
629            CALL BLDIF(1)
630           ELSE IF(WAKE) THEN
631            CALL BLDIF(3)
632           ELSE IF(TURB) THEN
633            CALL BLDIF(2)
634           ENDIF
635     C
636           IF(SIMI) THEN
637     C----- at similarity station, "1" variables are really "2" variables
638            DO 10 K=1, 4
639              DO 101 L=1, 5
640                VS2(K,L) = VS1(K,L) + VS2(K,L)
641                VS1(K,L) = 0.
642       101    CONTINUE
643        10  CONTINUE
644           ENDIF
645     C
646     C---- change system over into incompressible Uei and Mach
647           DO 20 K=1, 4
648     C
649     C------ residual derivatives wrt compressible Uec
650             RES_U1 = VS1(K,4)
651             RES_U2 = VS2(K,4)
652             RES_MS = VSM(K)
653     C
654     C------ combine with derivatives of compressible  U1,U2 = Uec(Uei M)
655             VS1(K,4) = RES_U1*U1_UEI
656             VS2(K,4) =                RES_U2*U2_UEI
657             VSM(K)   = RES_U1*U1_MS + RES_U2*U2_MS  + RES_MS
658        20 CONTINUE
659     C
660           RETURN
661           END

667     C     and first wake point infinitesimally behind TE.

725           SUBROUTINE BLKIN
726     C----------------------------------------------------------
727     C     Calculates turbulence-independent secondary "2" 
728     C     variables from the primary "2" variables.
729     C----------------------------------------------------------
730           IMPLICIT REAL(M)
731           INCLUDE 'XBL.INC'
732     C
733     C---- set edge Mach number ** 2
734           M2    = U2*U2*HSTINV / (GM1BL*(1.0 - 0.5*U2*U2*HSTINV))
735           TR2   = 1.0 + 0.5*GM1BL*M2
736           M2_U2 = 2.0*M2*TR2/U2
737           M2_MS = U2*U2*TR2    / (GM1BL*(1.0 - 0.5*U2*U2*HSTINV))
738          &      * HSTINV_MS
739     C
740     C---- set edge static density (isentropic relation)
741           R2    = RSTBL   *TR2**(-1.0/GM1BL)
742           R2_U2 = -R2/TR2 * 0.5*M2_U2
743           R2_MS = -R2/TR2 * 0.5*M2_MS
744          &      + RSTBL_MS*TR2**(-1.0/GM1BL)
745     C
746     C---- set shape parameter
747           H2    =  D2/T2
748           H2_D2 = 1.0/T2
749           H2_T2 = -H2/T2
750     C
751     C---- set edge static/stagnation enthalpy
752           HERAT = 1.0 - 0.5*U2*U2*HSTINV
753           HE_U2 =     -        U2*HSTINV
754           HE_MS =     - 0.5*U2*U2*HSTINV_MS
755     C
756     C---- set molecular viscosity
757           V2 = SQRT((HERAT)**3) * (1.0+HVRAT)/(HERAT+HVRAT)/REYBL
758           V2_HE = V2*(1.5/HERAT - 1.0/(HERAT+HVRAT))
759     C
760           V2_U2 =                        V2_HE*HE_U2
761           V2_MS = -V2/REYBL * REYBL_MS + V2_HE*HE_MS
762           V2_RE = -V2/REYBL * REYBL_RE
763     C
764     C---- set kinematic shape parameter
765           CALL HKIN( H2, M2, HK2, HK2_H2, HK2_M2 )
766     C
767           HK2_U2 =                HK2_M2*M2_U2
768           HK2_T2 = HK2_H2*H2_T2
769           HK2_D2 = HK2_H2*H2_D2
770           HK2_MS =                HK2_M2*M2_MS
771     C
772     C---- set momentum thickness Reynolds number
773           RT2    = R2*U2*T2/V2
774           RT2_U2 = RT2*(1.0/U2 + R2_U2/R2 - V2_U2/V2)
775           RT2_T2 = RT2/T2
776           RT2_MS = RT2*(         R2_MS/R2 - V2_MS/V2)
777           RT2_RE = RT2*(                  - V2_RE/V2)
778     C
779           RETURN
780           END ! BLKIN

784           SUBROUTINE BLVAR(ITYP)
785     C----------------------------------------------------
786     C     Calculates all secondary "2" variables from
787     C     the primary "2" variables X2, U2, T2, D2, S2.
788     C     Also calculates the sensitivities of the
789     C     secondary variables wrt the primary variables.
790     C
791     C      ITYP = 1 :  laminar
792     C      ITYP = 2 :  turbulent
793     C      ITYP = 3 :  turbulent wake
794     C----------------------------------------------------
795           IMPLICIT REAL(M)
796           INCLUDE 'XBL.INC'
797     C
798           IF(ITYP.EQ.3) HK2 = MAX(HK2,1.00005)
799           IF(ITYP.NE.3) HK2 = MAX(HK2,1.05000)
800     C
801     C---- density thickness shape parameter     ( H** )
802           CALL HCT( HK2, M2, HC2, HC2_HK2, HC2_M2 )
803           HC2_U2 = HC2_HK2*HK2_U2 + HC2_M2*M2_U2
804           HC2_T2 = HC2_HK2*HK2_T2
805           HC2_D2 = HC2_HK2*HK2_D2
806           HC2_MS = HC2_HK2*HK2_MS + HC2_M2*M2_MS
807     C
808     C---- set KE thickness shape parameter from  H - H*  correlations
809           IF(ITYP.EQ.1) THEN
810            CALL HSL( HK2, RT2, M2, HS2, HS2_HK2, HS2_RT2, HS2_M2 )
811           ELSE
812            CALL HST( HK2, RT2, M2, HS2, HS2_HK2, HS2_RT2, HS2_M2 )
813           ENDIF
814     C
815           HS2_U2 = HS2_HK2*HK2_U2 + HS2_RT2*RT2_U2 + HS2_M2*M2_U2
816           HS2_T2 = HS2_HK2*HK2_T2 + HS2_RT2*RT2_T2
817           HS2_D2 = HS2_HK2*HK2_D2
818           HS2_MS = HS2_HK2*HK2_MS + HS2_RT2*RT2_MS + HS2_M2*M2_MS
819           HS2_RE =                  HS2_RT2*RT2_RE
820     C
821     C---- normalized slip velocity  Us
822           US2     = 0.5*HS2*( 1.0 - (HK2-1.0)/(GBCON*H2) )
823           US2_HS2 = 0.5  *  ( 1.0 - (HK2-1.0)/(GBCON*H2) )
824           US2_HK2 = 0.5*HS2*(     -  1.0     /(GBCON*H2) )
825           US2_H2  = 0.5*HS2*        (HK2-1.0)/(GBCON*H2**2)
826     C
827           US2_U2 = US2_HS2*HS2_U2 + US2_HK2*HK2_U2
828           US2_T2 = US2_HS2*HS2_T2 + US2_HK2*HK2_T2 + US2_H2*H2_T2
829           US2_D2 = US2_HS2*HS2_D2 + US2_HK2*HK2_D2 + US2_H2*H2_D2
830           US2_MS = US2_HS2*HS2_MS + US2_HK2*HK2_MS
831           US2_RE = US2_HS2*HS2_RE
832     C
833           IF(ITYP.LE.2 .AND. US2.GT.0.95) THEN
834     CCC       WRITE(*,*) 'BLVAR: Us clamped:', US2
835            US2 = 0.98
836            US2_U2 = 0.
837            US2_T2 = 0.
838            US2_D2 = 0.
839            US2_MS = 0.
840            US2_RE = 0.
841           ENDIF
842     C
843           IF(ITYP.EQ.3 .AND. US2.GT.0.99995) THEN
844     CCC       WRITE(*,*) 'BLVAR: Wake Us clamped:', US2
845            US2 = 0.99995
846            US2_U2 = 0.
847            US2_T2 = 0.
848            US2_D2 = 0.
849            US2_MS = 0.
850            US2_RE = 0.
851           ENDIF
852     C
853     C---- equilibrium wake layer shear coefficient (Ctau)EQ ** 1/2
854     C   ...  NEW  12 Oct 94
855           GCC = 0.0
856           HKC = HK2 - 1.0
857           HKC_HK2 = 1.0
858           HKC_RT2 = 0.0
859           IF(ITYP.EQ.2) THEN
860            GCC = GCCON
861            HKC     = HK2 - 1.0 - GCC/RT2
862            HKC_HK2 = 1.0
863            HKC_RT2 =             GCC/RT2**2
864            IF(HKC .LT. 0.01) THEN
865             HKC = 0.01
866             HKC_HK2 = 0.0
867             HKC_RT2 = 0.0
868            ENDIF
869           ENDIF
870     C
871           HKB = HK2 - 1.0
872           USB = 1.0 - US2
873           CQ2     =
874          &    SQRT( CTCON*HS2*HKB*HKC**2 / (USB*H2*HK2**2) )
875           CQ2_HS2 = CTCON    *HKB*HKC**2 / (USB*H2*HK2**2)       * 0.5/CQ2
876           CQ2_US2 = CTCON*HS2*HKB*HKC**2 / (USB*H2*HK2**2) / USB * 0.5/CQ2
877           CQ2_HK2 = CTCON*HS2    *HKC**2 / (USB*H2*HK2**2)       * 0.5/CQ2
878          &        - CTCON*HS2*HKB*HKC**2 / (USB*H2*HK2**3) * 2.0 * 0.5/CQ2
879          &        + CTCON*HS2*HKB*HKC    / (USB*H2*HK2**2) * 2.0 * 0.5/CQ2
880          &         *HKC_HK2
881           CQ2_RT2 = CTCON*HS2*HKB*HKC    / (USB*H2*HK2**2) * 2.0 * 0.5/CQ2
882          &         *HKC_RT2
883           CQ2_H2  =-CTCON*HS2*HKB*HKC**2 / (USB*H2*HK2**2) / H2  * 0.5/CQ2
884     C
885           CQ2_U2 = CQ2_HS2*HS2_U2 + CQ2_US2*US2_U2 + CQ2_HK2*HK2_U2
886           CQ2_T2 = CQ2_HS2*HS2_T2 + CQ2_US2*US2_T2 + CQ2_HK2*HK2_T2
887           CQ2_D2 = CQ2_HS2*HS2_D2 + CQ2_US2*US2_D2 + CQ2_HK2*HK2_D2
888           CQ2_MS = CQ2_HS2*HS2_MS + CQ2_US2*US2_MS + CQ2_HK2*HK2_MS
889           CQ2_RE = CQ2_HS2*HS2_RE + CQ2_US2*US2_RE
890     C
891           CQ2_U2 = CQ2_U2                + CQ2_RT2*RT2_U2
892           CQ2_T2 = CQ2_T2 + CQ2_H2*H2_T2 + CQ2_RT2*RT2_T2
893           CQ2_D2 = CQ2_D2 + CQ2_H2*H2_D2
894           CQ2_MS = CQ2_MS                + CQ2_RT2*RT2_MS
895           CQ2_RE = CQ2_RE                + CQ2_RT2*RT2_RE
896     C
897     C
898     C---- set skin friction coefficient 
899           IF(ITYP.EQ.3) THEN
900     C----- wake
901            CF2     = 0.
902            CF2_HK2 = 0.
903            CF2_RT2 = 0.
904            CF2_M2  = 0.
905           ELSE IF(ITYP.EQ.1) THEN
906     C----- laminar
907            CALL CFL( HK2, RT2, M2, CF2, CF2_HK2, CF2_RT2, CF2_M2 )
908           ELSE
909     C----- turbulent
910            CALL CFT( HK2, RT2, M2, CF2, CF2_HK2, CF2_RT2, CF2_M2 )
911            CALL CFL( HK2, RT2, M2, CF2L,CF2L_HK2,CF2L_RT2,CF2L_M2)
912            IF(CF2L.GT.CF2) THEN
913     C------- laminar Cf is greater than turbulent Cf -- use laminar
914     C-       (this will only occur for unreasonably small Rtheta)
915     ccc      write(*,*) 'Cft Cfl Rt Hk:', CF2, CF2L, RT2, HK2, X2
916              CF2     = CF2L
917              CF2_HK2 = CF2L_HK2
918              CF2_RT2 = CF2L_RT2
919              CF2_M2  = CF2L_M2
920            ENDIF
921           ENDIF
922     C
923           CF2_U2 = CF2_HK2*HK2_U2 + CF2_RT2*RT2_U2 + CF2_M2*M2_U2
924           CF2_T2 = CF2_HK2*HK2_T2 + CF2_RT2*RT2_T2
925           CF2_D2 = CF2_HK2*HK2_D2
926           CF2_MS = CF2_HK2*HK2_MS + CF2_RT2*RT2_MS + CF2_M2*M2_MS
927           CF2_RE =                  CF2_RT2*RT2_RE
928     C
929     C---- dissipation function    2 CD / H*
930           IF(ITYP.EQ.1) THEN
931     C
932     C----- laminar
933            CALL DIL( HK2, RT2, DI2, DI2_HK2, DI2_RT2 )
934     C
935            DI2_U2 = DI2_HK2*HK2_U2 + DI2_RT2*RT2_U2
936            DI2_T2 = DI2_HK2*HK2_T2 + DI2_RT2*RT2_T2
937            DI2_D2 = DI2_HK2*HK2_D2
938            DI2_S2 = 0.
939            DI2_MS = DI2_HK2*HK2_MS + DI2_RT2*RT2_MS
940            DI2_RE =                  DI2_RT2*RT2_RE
941     C
942           ELSE IF(ITYP.EQ.2) THEN
943     C
944     CCC       CALL DIT(     HS2,     US2,     CF2,     S2, DI2,
945     CCC     &           DI2_HS2, DI2_US2, DI2_CF2, DI2_S2      )
946     C
947     C----- turbulent wall contribution
948            CALL CFT(HK2, RT2, M2, CF2T, CF2T_HK2, CF2T_RT2, CF2T_M2)
949            CF2T_U2 = CF2T_HK2*HK2_U2 + CF2T_RT2*RT2_U2 + CF2T_M2*M2_U2
950            CF2T_T2 = CF2T_HK2*HK2_T2 + CF2T_RT2*RT2_T2
951            CF2T_D2 = CF2T_HK2*HK2_D2
952            CF2T_MS = CF2T_HK2*HK2_MS + CF2T_RT2*RT2_MS + CF2T_M2*M2_MS
953            CF2T_RE =                   CF2T_RT2*RT2_RE
954     C
955            DI2      =  ( 0.5*CF2T*US2 ) * 2.0/HS2
956            DI2_HS2  = -( 0.5*CF2T*US2 ) * 2.0/HS2**2
957            DI2_US2  =  ( 0.5*CF2T     ) * 2.0/HS2
958            DI2_CF2T =  ( 0.5     *US2 ) * 2.0/HS2
959     C
960            DI2_S2 = 0.0
961            DI2_U2 = DI2_HS2*HS2_U2 + DI2_US2*US2_U2 + DI2_CF2T*CF2T_U2
962            DI2_T2 = DI2_HS2*HS2_T2 + DI2_US2*US2_T2 + DI2_CF2T*CF2T_T2
963            DI2_D2 = DI2_HS2*HS2_D2 + DI2_US2*US2_D2 + DI2_CF2T*CF2T_D2
964            DI2_MS = DI2_HS2*HS2_MS + DI2_US2*US2_MS + DI2_CF2T*CF2T_MS
965            DI2_RE = DI2_HS2*HS2_RE + DI2_US2*US2_RE + DI2_CF2T*CF2T_RE
966     C
967     C
968     C----- set minimum Hk for wake layer to still exist
969            GRT = LOG(RT2)
970            HMIN = 1.0 + 2.1/GRT
971            HM_RT2 = -(2.1/GRT**2) / RT2
972     C
973     C----- set factor DFAC for correcting wall dissipation for very low Hk
974            FL = (HK2-1.0)/(HMIN-1.0)
975            FL_HK2 =   1.0/(HMIN-1.0)
976            FL_RT2 = ( -FL/(HMIN-1.0) ) * HM_RT2
977     C
978            TFL = TANH(FL)
979            DFAC  = 0.5 + 0.5* TFL
980            DF_FL =       0.5*(1.0 - TFL**2)
981     C
982            DF_HK2 = DF_FL*FL_HK2
983            DF_RT2 = DF_FL*FL_RT2
984     C
985            DI2_S2 = DI2_S2*DFAC
986            DI2_U2 = DI2_U2*DFAC + DI2*(DF_HK2*HK2_U2 + DF_RT2*RT2_U2)
987            DI2_T2 = DI2_T2*DFAC + DI2*(DF_HK2*HK2_T2 + DF_RT2*RT2_T2)
988            DI2_D2 = DI2_D2*DFAC + DI2*(DF_HK2*HK2_D2                )
989            DI2_MS = DI2_MS*DFAC + DI2*(DF_HK2*HK2_MS + DF_RT2*RT2_MS)
990            DI2_RE = DI2_RE*DFAC + DI2*(                DF_RT2*RT2_RE)
991            DI2    = DI2   *DFAC
992     C
993           ELSE
994     C
995     C----- zero wall contribution for wake
996            DI2    = 0.0
997            DI2_S2 = 0.0
998            DI2_U2 = 0.0
999            DI2_T2 = 0.0
1000            DI2_D2 = 0.0
1001            DI2_MS = 0.0
1002            DI2_RE = 0.0
1003     C
1004           ENDIF
1005     C
1006     C
1007     C---- Add on turbulent outer layer contribution
1008           IF(ITYP.NE.1) THEN
1009     C
1010            DD     =  S2**2 * (0.995-US2) * 2.0/HS2
1011            DD_HS2 = -S2**2 * (0.995-US2) * 2.0/HS2**2
1012            DD_US2 = -S2**2               * 2.0/HS2
1013            DD_S2  =  S2*2.0* (0.995-US2) * 2.0/HS2
1014     C

1016            DI2_S2 =          DD_S2
1017            DI2_U2 = DI2_U2 + DD_HS2*HS2_U2 + DD_US2*US2_U2
1018            DI2_T2 = DI2_T2 + DD_HS2*HS2_T2 + DD_US2*US2_T2
1019            DI2_D2 = DI2_D2 + DD_HS2*HS2_D2 + DD_US2*US2_D2
1020            DI2_MS = DI2_MS + DD_HS2*HS2_MS + DD_US2*US2_MS
1021            DI2_RE = DI2_RE + DD_HS2*HS2_RE + DD_US2*US2_RE

1023     C----- add laminar stress contribution to outer layer CD
1024     c###
1025            DD     =  0.15*(0.995-US2)**2 / RT2  * 2.0/HS2
1026            DD_US2 = -0.15*(0.995-US2)*2. / RT2  * 2.0/HS2
1027            DD_HS2 = -DD/HS2
1028            DD_RT2 = -DD/RT2
1029     C
1030            DI2    = DI2    + DD
1031            DI2_U2 = DI2_U2 + DD_HS2*HS2_U2 + DD_US2*US2_U2 + DD_RT2*RT2_U2
1032            DI2_T2 = DI2_T2 + DD_HS2*HS2_T2 + DD_US2*US2_T2 + DD_RT2*RT2_T2
1033            DI2_D2 = DI2_D2 + DD_HS2*HS2_D2 + DD_US2*US2_D2
1034            DI2_MS = DI2_MS + DD_HS2*HS2_MS + DD_US2*US2_MS + DD_RT2*RT2_MS
1035            DI2_RE = DI2_RE + DD_HS2*HS2_RE + DD_US2*US2_RE + DD_RT2*RT2_RE
1036     C
1037           ENDIF
1038     C
1039     C
1040           IF(ITYP.EQ.2) THEN
1041             CALL DIL( HK2, RT2, DI2L, DI2L_HK2, DI2L_RT2 )
1042     C
1043             IF(DI2L.GT.DI2) THEN
1044     C------- laminar CD is greater than turbulent CD -- use laminar
1045     C-       (this will only occur for unreasonably small Rtheta)
1046     ccc       write(*,*) 'CDt CDl Rt Hk:', DI2, DI2L, RT2, HK2
1047               DI2    = DI2L
1048               DI2_S2 = 0.
1049               DI2_U2 = DI2L_HK2*HK2_U2 + DI2L_RT2*RT2_U2
1050               DI2_T2 = DI2L_HK2*HK2_T2 + DI2L_RT2*RT2_T2
1051               DI2_D2 = DI2L_HK2*HK2_D2
1052               DI2_MS = DI2L_HK2*HK2_MS + DI2L_RT2*RT2_MS
1053               DI2_RE =                   DI2L_RT2*RT2_RE
1054             ENDIF
1055           ENDIF
1056     C
1057     cC----- add on CD contribution of inner shear layer
1058     c       IF(ITYP.EQ.3 .AND. DW2.GT.0.0) THEN
1059     c        DKON = 0.03*0.75**3
1060     c        DDI = DKON*US2**3
1061     c        DDI_US2 = 3.0*DKON*US2**2
1062     c        DI2 = DI2 + DDI * DW2/DWTE
1063     c        DI2_U2 = DI2_U2 + DDI_US2*US2_U2 * DW2/DWTE
1064     c        DI2_T2 = DI2_T2 + DDI_US2*US2_T2 * DW2/DWTE
1065     c        DI2_D2 = DI2_D2 + DDI_US2*US2_D2 * DW2/DWTE
1066     c        DI2_MS = DI2_MS + DDI_US2*US2_MS * DW2/DWTE
1067     c        DI2_RE = DI2_RE + DDI_US2*US2_RE * DW2/DWTE
1068     c       ENDIF
1069     C
1070           IF(ITYP.EQ.3) THEN
1071     C------ laminar wake CD
1072             CALL DILW( HK2, RT2, DI2L, DI2L_HK2, DI2L_RT2 )
1073             IF(DI2L .GT. DI2) THEN
1074     C------- laminar wake CD is greater than turbulent CD -- use laminar
1075     C-       (this will only occur for unreasonably small Rtheta)
1076     ccc         write(*,*) 'CDt CDl Rt Hk:', DI2, DI2L, RT2, HK2
1077              DI2    = DI2L
1078              DI2_S2 = 0.
1079              DI2_U2 = DI2L_HK2*HK2_U2 + DI2L_RT2*RT2_U2
1080              DI2_T2 = DI2L_HK2*HK2_T2 + DI2L_RT2*RT2_T2
1081              DI2_D2 = DI2L_HK2*HK2_D2
1082              DI2_MS = DI2L_HK2*HK2_MS + DI2L_RT2*RT2_MS
1083              DI2_RE =                   DI2L_RT2*RT2_RE
1084             ENDIF
1085           ENDIF
1086     C
1087     C
1088           IF(ITYP.EQ.3) THEN
1089     C----- double dissipation for the wake (two wake halves)
1090            DI2    = DI2   *2.0
1091            DI2_S2 = DI2_S2*2.0
1092            DI2_U2 = DI2_U2*2.0
1093            DI2_T2 = DI2_T2*2.0
1094            DI2_D2 = DI2_D2*2.0
1095            DI2_MS = DI2_MS*2.0
1096            DI2_RE = DI2_RE*2.0
1097           ENDIF
1098     C
1099     C---- BL thickness (Delta) from simplified Green's correlation
1100           DE2     = (3.15 + 1.72/(HK2-1.0)   )*T2  +  D2
1101           DE2_HK2 = (     - 1.72/(HK2-1.0)**2)*T2
1102     C
1103           DE2_U2 = DE2_HK2*HK2_U2
1104           DE2_T2 = DE2_HK2*HK2_T2 + (3.15 + 1.72/(HK2-1.0))
1105           DE2_D2 = DE2_HK2*HK2_D2 + 1.0
1106           DE2_MS = DE2_HK2*HK2_MS
1107     C
1108     ccc      HDMAX = 15.0
1109           HDMAX = 12.0
1110           IF(DE2 .GT. HDMAX*T2) THEN
1111     cccc      IF(DE2 .GT. HDMAX*T2 .AND. (HK2 .GT. 4.0 .OR. ITYP.EQ.3)) THEN
1112            DE2    = HDMAX*T2
1113            DE2_U2 =  0.0
1114            DE2_T2 = HDMAX
1115            DE2_D2 =  0.0
1116            DE2_MS =  0.0
1117           ENDIF
1118     C
1119           RETURN
1120           END

1123           SUBROUTINE BLMID(ITYP)
1124     C----------------------------------------------------
1125     C     Calculates midpoint skin friction CFM
1126     C
1127     C      ITYP = 1 :  laminar
1128     C      ITYP = 2 :  turbulent
1129     C      ITYP = 3 :  turbulent wake
1130     C----------------------------------------------------
1131           IMPLICIT REAL(M)
1132           INCLUDE 'XBL.INC'
1133     C
1134     C---- set similarity variables if not defined
1135           IF(SIMI) THEN
1136            HK1    = HK2
1137            HK1_T1 = HK2_T2
1138            HK1_D1 = HK2_D2
1139            HK1_U1 = HK2_U2
1140            HK1_MS = HK2_MS
1141            RT1    = RT2
1142            RT1_T1 = RT2_T2
1143            RT1_U1 = RT2_U2
1144            RT1_MS = RT2_MS
1145            RT1_RE = RT2_RE
1146            M1    = M2
1147            M1_U1 = M2_U2
1148            M1_MS = M2_MS
1149           ENDIF
1150     C
1151     C---- define stuff for midpoint CF
1152           HKA = 0.5*(HK1 + HK2)
1153           RTA = 0.5*(RT1 + RT2)
1154           MA  = 0.5*(M1  + M2 )
1155     C
1156     C---- midpoint skin friction coefficient  (zero in wake)
1157           IF(ITYP.EQ.3) THEN
1158            CFM     = 0.
1159            CFM_HKA = 0.
1160            CFM_RTA = 0.
1161            CFM_MA  = 0.
1162            CFM_MS  = 0.
1163           ELSE IF(ITYP.EQ.1) THEN
1164            CALL CFL( HKA, RTA, MA, CFM, CFM_HKA, CFM_RTA, CFM_MA )
1165           ELSE
1166            CALL CFT( HKA, RTA, MA, CFM, CFM_HKA, CFM_RTA, CFM_MA )
1167            CALL CFL( HKA, RTA, MA, CFML,CFML_HKA,CFML_RTA,CFML_MA)
1168            IF(CFML.GT.CFM) THEN
1169     ccc      write(*,*) 'Cft Cfl Rt Hk:', CFM, CFML, RTA, HKA, 0.5*(X1+X2)
1170              CFM     = CFML
1171              CFM_HKA = CFML_HKA
1172              CFM_RTA = CFML_RTA
1173              CFM_MA  = CFML_MA
1174            ENDIF
1175           ENDIF
1176     C
1177           CFM_U1 = 0.5*(CFM_HKA*HK1_U1 + CFM_MA*M1_U1 + CFM_RTA*RT1_U1)
1178           CFM_T1 = 0.5*(CFM_HKA*HK1_T1 +                CFM_RTA*RT1_T1)
1179           CFM_D1 = 0.5*(CFM_HKA*HK1_D1                                )
1180     C
1181           CFM_U2 = 0.5*(CFM_HKA*HK2_U2 + CFM_MA*M2_U2 + CFM_RTA*RT2_U2)
1182           CFM_T2 = 0.5*(CFM_HKA*HK2_T2 +                CFM_RTA*RT2_T2)
1183           CFM_D2 = 0.5*(CFM_HKA*HK2_D2                                )
1184     C
1185           CFM_MS = 0.5*(CFM_HKA*HK1_MS + CFM_MA*M1_MS + CFM_RTA*RT1_MS
1186          &            + CFM_HKA*HK2_MS + CFM_MA*M2_MS + CFM_RTA*RT2_MS)
1187           CFM_RE = 0.5*(                                CFM_RTA*RT1_RE
1188          &                                            + CFM_RTA*RT2_RE)
1189     C
1190           RETURN
1191           END ! BLMID

1194           SUBROUTINE TRDIF
1195     C-----------------------------------------------
1196     C     Sets up the Newton system governing the
1197     C     transition interval.  Equations governing
1198     C     the  laminar  part  X1 < xi < XT  and
1199     C     the turbulent part  XT < xi < X2
1200     C     are simply summed.
1201     C-----------------------------------------------
1202           IMPLICIT REAL(M)
1203           INCLUDE 'XBL.INC'
1204           REAL  BL1(4,5), BL2(4,5), BLREZ(4), BLM(4), BLR(4), BLX(4)
1205          &    , BT1(4,5), BT2(4,5), BTREZ(4), BTM(4), BTR(4), BTX(4)
1206     C
1207     C---- save variables and sensitivities for future restoration
1208           DO 5 ICOM=1, NCOM
1209             C1SAV(ICOM) = COM1(ICOM)
1210             C2SAV(ICOM) = COM2(ICOM)
1211         5 CONTINUE
1212     C
1213     C---- weighting factors for linear interpolation to transition point
1214           WF2    = (XT-X1)/(X2-X1)
1215           WF2_XT = 1.0/(X2-X1)
1216     C
1217           WF2_A1 = WF2_XT*XT_A1
1218           WF2_X1 = WF2_XT*XT_X1 + (WF2-1.0)/(X2-X1)
1219           WF2_X2 = WF2_XT*XT_X2 -  WF2     /(X2-X1)
1220           WF2_T1 = WF2_XT*XT_T1
1221           WF2_T2 = WF2_XT*XT_T2
1222           WF2_D1 = WF2_XT*XT_D1
1223           WF2_D2 = WF2_XT*XT_D2
1224           WF2_U1 = WF2_XT*XT_U1
1225           WF2_U2 = WF2_XT*XT_U2
1226           WF2_MS = WF2_XT*XT_MS
1227           WF2_RE = WF2_XT*XT_RE
1228           WF2_XF = WF2_XT*XT_XF
1229     C
1230           WF1    = 1.0 - WF2
1231           WF1_A1 = -WF2_A1
1232           WF1_X1 = -WF2_X1
1233           WF1_X2 = -WF2_X2
1234           WF1_T1 = -WF2_T1
1235           WF1_T2 = -WF2_T2
1236           WF1_D1 = -WF2_D1
1237           WF1_D2 = -WF2_D2
1238           WF1_U1 = -WF2_U1
1239           WF1_U2 = -WF2_U2
1240           WF1_MS = -WF2_MS
1241           WF1_RE = -WF2_RE
1242           WF1_XF = -WF2_XF
1243     C
1244     C
1245     C**** FIRST,  do laminar part between X1 and XT
1246     C
1247     C-----interpolate primary variables to transition point
1248           TT    = T1*WF1    + T2*WF2
1249           TT_A1 = T1*WF1_A1 + T2*WF2_A1
1250           TT_X1 = T1*WF1_X1 + T2*WF2_X1
1251           TT_X2 = T1*WF1_X2 + T2*WF2_X2
1252           TT_T1 = T1*WF1_T1 + T2*WF2_T1 + WF1
1253           TT_T2 = T1*WF1_T2 + T2*WF2_T2 + WF2
1254           TT_D1 = T1*WF1_D1 + T2*WF2_D1
1255           TT_D2 = T1*WF1_D2 + T2*WF2_D2
1256           TT_U1 = T1*WF1_U1 + T2*WF2_U1
1257           TT_U2 = T1*WF1_U2 + T2*WF2_U2
1258           TT_MS = T1*WF1_MS + T2*WF2_MS
1259           TT_RE = T1*WF1_RE + T2*WF2_RE
1260           TT_XF = T1*WF1_XF + T2*WF2_XF
1261     C
1262           DT    = D1*WF1    + D2*WF2
1263           DT_A1 = D1*WF1_A1 + D2*WF2_A1
1264           DT_X1 = D1*WF1_X1 + D2*WF2_X1
1265           DT_X2 = D1*WF1_X2 + D2*WF2_X2
1266           DT_T1 = D1*WF1_T1 + D2*WF2_T1
1267           DT_T2 = D1*WF1_T2 + D2*WF2_T2
1268           DT_D1 = D1*WF1_D1 + D2*WF2_D1 + WF1
1269           DT_D2 = D1*WF1_D2 + D2*WF2_D2 + WF2
1270           DT_U1 = D1*WF1_U1 + D2*WF2_U1
1271           DT_U2 = D1*WF1_U2 + D2*WF2_U2
1272           DT_MS = D1*WF1_MS + D2*WF2_MS
1273           DT_RE = D1*WF1_RE + D2*WF2_RE
1274           DT_XF = D1*WF1_XF + D2*WF2_XF
1275     C
1276           UT    = U1*WF1    + U2*WF2
1277           UT_A1 = U1*WF1_A1 + U2*WF2_A1
1278           UT_X1 = U1*WF1_X1 + U2*WF2_X1
1279           UT_X2 = U1*WF1_X2 + U2*WF2_X2
1280           UT_T1 = U1*WF1_T1 + U2*WF2_T1
1281           UT_T2 = U1*WF1_T2 + U2*WF2_T2
1282           UT_D1 = U1*WF1_D1 + U2*WF2_D1
1283           UT_D2 = U1*WF1_D2 + U2*WF2_D2
1284           UT_U1 = U1*WF1_U1 + U2*WF2_U1 + WF1
1285           UT_U2 = U1*WF1_U2 + U2*WF2_U2 + WF2
1286           UT_MS = U1*WF1_MS + U2*WF2_MS
1287           UT_RE = U1*WF1_RE + U2*WF2_RE
1288           UT_XF = U1*WF1_XF + U2*WF2_XF
1289     C
1290     C---- set primary "T" variables at XT  (really placed into "2" variables)
1291           X2 = XT
1292           T2 = TT
1293           D2 = DT
1294           U2 = UT
1295     C
1296           AMPL2 = AMCRIT
1297           S2 = 0.
1298     C
1299     C---- calculate laminar secondary "T" variables
1300           CALL BLKIN
1301           CALL BLVAR(1)
1302     C
1303     C---- calculate X1-XT midpoint CFM value
1304           CALL BLMID(1)
1305     C=
1306     C=    at this point, all "2" variables are really "T" variables at XT
1307     C=
1308     C
1309     C---- set up Newton system for dAm, dTh, dDs, dUe, dXi  at  X1 and XT
1310           CALL BLDIF(1)
1311     C
1312     C---- The current Newton system is in terms of "1" and "T" variables,
1313     C-    so calculate its equivalent in terms of "1" and "2" variables.
1314     C-    In other words, convert residual sensitivities wrt "T" variables
1315     C-    into sensitivities wrt "1" and "2" variables.  The amplification
1316     C-    equation is unnecessary here, so the K=1 row is left empty.
1317           DO 10 K=2, 3
1318             BLREZ(K) = VSREZ(K)
1319             BLM(K)   = VSM(K)
1320          &           + VS2(K,2)*TT_MS
1321          &           + VS2(K,3)*DT_MS
1322          &           + VS2(K,4)*UT_MS
1323          &           + VS2(K,5)*XT_MS
1324             BLR(K)   = VSR(K)
1325          &           + VS2(K,2)*TT_RE
1326          &           + VS2(K,3)*DT_RE
1327          &           + VS2(K,4)*UT_RE
1328          &           + VS2(K,5)*XT_RE
1329             BLX(K)   = VSX(K)
1330          &           + VS2(K,2)*TT_XF
1331          &           + VS2(K,3)*DT_XF
1332          &           + VS2(K,4)*UT_XF
1333          &           + VS2(K,5)*XT_XF
1334     C
1335             BL1(K,1) = VS1(K,1)
1336          &           + VS2(K,2)*TT_A1
1337          &           + VS2(K,3)*DT_A1
1338          &           + VS2(K,4)*UT_A1
1339          &           + VS2(K,5)*XT_A1
1340             BL1(K,2) = VS1(K,2)
1341          &           + VS2(K,2)*TT_T1
1342          &           + VS2(K,3)*DT_T1
1343          &           + VS2(K,4)*UT_T1
1344          &           + VS2(K,5)*XT_T1
1345             BL1(K,3) = VS1(K,3)
1346          &           + VS2(K,2)*TT_D1
1347          &           + VS2(K,3)*DT_D1
1348          &           + VS2(K,4)*UT_D1
1349          &           + VS2(K,5)*XT_D1
1350             BL1(K,4) = VS1(K,4)
1351          &           + VS2(K,2)*TT_U1
1352          &           + VS2(K,3)*DT_U1
1353          &           + VS2(K,4)*UT_U1
1354          &           + VS2(K,5)*XT_U1
1355             BL1(K,5) = VS1(K,5)
1356          &           + VS2(K,2)*TT_X1
1357          &           + VS2(K,3)*DT_X1
1358          &           + VS2(K,4)*UT_X1
1359          &           + VS2(K,5)*XT_X1
1360     C
1361             BL2(K,1) = 0.
1362             BL2(K,2) = VS2(K,2)*TT_T2
1363          &           + VS2(K,3)*DT_T2
1364          &           + VS2(K,4)*UT_T2
1365          &           + VS2(K,5)*XT_T2
1366             BL2(K,3) = VS2(K,2)*TT_D2
1367          &           + VS2(K,3)*DT_D2
1368          &           + VS2(K,4)*UT_D2
1369          &           + VS2(K,5)*XT_D2
1370             BL2(K,4) = VS2(K,2)*TT_U2
1371          &           + VS2(K,3)*DT_U2
1372          &           + VS2(K,4)*UT_U2
1373          &           + VS2(K,5)*XT_U2
1374             BL2(K,5) = VS2(K,2)*TT_X2
1375          &           + VS2(K,3)*DT_X2
1376          &           + VS2(K,4)*UT_X2
1377          &           + VS2(K,5)*XT_X2
1378     C
1379        10 CONTINUE
1380     C
1381     C
1382     C**** SECOND, set up turbulent part between XT and X2  ****
1383     C
1384     C---- calculate equilibrium shear coefficient CQT at transition point
1385           CALL BLVAR(2)
1386     C
1387     C---- set initial shear coefficient value ST at transition point
1388     C-    ( note that CQ2, CQ2_T2, etc. are really "CQT", "CQT_TT", etc.)
1389     C
1390           CTR     = CTRCON*EXP(-CTRCEX/(HK2-1.0))
1391           CTR_HK2 = CTR * CTRCEX/(HK2-1.0)**2
1392     C
1393     c      CTR     = 1.1*EXP(-10.0/HK2**2)
1394     c      CTR_HK2 = CTR * 10.0 * 2.0/HK2**3
1395     C
1396     CCC      CTR = 1.2
1397     CCC      CTR = 0.7
1398     CCC      CTR_HK2 = 0.0
1399     C
1400           ST    = CTR*CQ2
1401           ST_TT = CTR*CQ2_T2 + CQ2*CTR_HK2*HK2_T2
1402           ST_DT = CTR*CQ2_D2 + CQ2*CTR_HK2*HK2_D2
1403           ST_UT = CTR*CQ2_U2 + CQ2*CTR_HK2*HK2_U2
1404           ST_MS = CTR*CQ2_MS + CQ2*CTR_HK2*HK2_MS
1405           ST_RE = CTR*CQ2_RE
1406     C
1407     C---- calculate ST sensitivities wrt the actual "1" and "2" variables
1408           ST_A1 = ST_TT*TT_A1 + ST_DT*DT_A1 + ST_UT*UT_A1
1409           ST_X1 = ST_TT*TT_X1 + ST_DT*DT_X1 + ST_UT*UT_X1
1410           ST_X2 = ST_TT*TT_X2 + ST_DT*DT_X2 + ST_UT*UT_X2
1411           ST_T1 = ST_TT*TT_T1 + ST_DT*DT_T1 + ST_UT*UT_T1
1412           ST_T2 = ST_TT*TT_T2 + ST_DT*DT_T2 + ST_UT*UT_T2
1413           ST_D1 = ST_TT*TT_D1 + ST_DT*DT_D1 + ST_UT*UT_D1
1414           ST_D2 = ST_TT*TT_D2 + ST_DT*DT_D2 + ST_UT*UT_D2
1415           ST_U1 = ST_TT*TT_U1 + ST_DT*DT_U1 + ST_UT*UT_U1
1416           ST_U2 = ST_TT*TT_U2 + ST_DT*DT_U2 + ST_UT*UT_U2
1417           ST_MS = ST_TT*TT_MS + ST_DT*DT_MS + ST_UT*UT_MS + ST_MS
1418           ST_RE = ST_TT*TT_RE + ST_DT*DT_RE + ST_UT*UT_RE + ST_RE
1419           ST_XF = ST_TT*TT_XF + ST_DT*DT_XF + ST_UT*UT_XF
1420     C
1421           AMPL2 = 0.
1422           S2 = ST
1423     C
1424     C---- recalculate turbulent secondary "T" variables using proper CTI
1425           CALL BLVAR(2)
1426     C
1427     C---- set "1" variables to "T" variables and reset "2" variables
1428     C-    to their saved turbulent values
1429           DO 30 ICOM=1, NCOM
1430             COM1(ICOM) = COM2(ICOM)
1431             COM2(ICOM) = C2SAV(ICOM)
1432        30 CONTINUE
1433     C
1434     C---- calculate XT-X2 midpoint CFM value
1435           CALL BLMID(2)
1436     C
1437     C---- set up Newton system for dCt, dTh, dDs, dUe, dXi  at  XT and X2
1438           CALL BLDIF(2)

1441     C-    wrt "1" and "2" variables as done before for the laminar part
1442           DO 40 K=1, 3

1445          &           + VS1(K,1)*ST_MS
1446          &           + VS1(K,2)*TT_MS

1448          &           + VS1(K,4)*UT_MS

1454          &           + VS1(K,4)*UT_RE
1455          &           + VS1(K,5)*XT_RE

1457          &           + VS1(K,1)*ST_XF
1458          &           + VS1(K,2)*TT_XF
1459          &           + VS1(K,3)*DT_XF
1460          &           + VS1(K,4)*UT_XF
1461          &           + VS1(K,5)*XT_XF
1462     C
1463             BT1(K,1) = VS1(K,1)*ST_A1
1464          &           + VS1(K,2)*TT_A1
1465          &           + VS1(K,3)*DT_A1
1466          &           + VS1(K,4)*UT_A1
1467          &           + VS1(K,5)*XT_A1
1468             BT1(K,2) = VS1(K,1)*ST_T1
1469          &           + VS1(K,2)*TT_T1
1470          &           + VS1(K,3)*DT_T1
1471          &           + VS1(K,4)*UT_T1
1472          &           + VS1(K,5)*XT_T1
1473             BT1(K,3) = VS1(K,1)*ST_D1
1474          &           + VS1(K,2)*TT_D1
1475          &           + VS1(K,3)*DT_D1
1476          &           + VS1(K,4)*UT_D1
1477          &           + VS1(K,5)*XT_D1
1478             BT1(K,4) = VS1(K,1)*ST_U1
1479          &           + VS1(K,2)*TT_U1
1480          &           + VS1(K,3)*DT_U1
1481          &           + VS1(K,4)*UT_U1
1482          &           + VS1(K,5)*XT_U1
1483             BT1(K,5) = VS1(K,1)*ST_X1
1484          &           + VS1(K,2)*TT_X1
1485          &           + VS1(K,3)*DT_X1
1486          &           + VS1(K,4)*UT_X1
1487          &           + VS1(K,5)*XT_X1
1488     C
1489             BT2(K,1) = VS2(K,1)
1490             BT2(K,2) = VS2(K,2)
1491          &           + VS1(K,1)*ST_T2
1492          &           + VS1(K,2)*TT_T2
1493          &           + VS1(K,3)*DT_T2
1494          &           + VS1(K,4)*UT_T2
1495          &           + VS1(K,5)*XT_T2

1497          &           + VS1(K,1)*ST_D2
1498          &           + VS1(K,2)*TT_D2
1499          &           + VS1(K,3)*DT_D2
1500          &           + VS1(K,4)*UT_D2
1501          &           + VS1(K,5)*XT_D2

1515        40 CONTINUE
1516     C

1518     C-    in terms of honest-to-God "1" and "2" variables.

1520           VSREZ(2) = BLREZ(2) + BTREZ(2)
1521           VSREZ(3) = BLREZ(3) + BTREZ(3)

1523           VSM(2)   = BLM(2)   + BTM(2)
1524           VSM(3)   = BLM(3)   + BTM(3)
1525           VSR(1)   =            BTR(1)
1526           VSR(2)   = BLR(2)   + BTR(2)
1527           VSR(3)   = BLR(3)   + BTR(3)
1528           VSX(1)   =            BTX(1)
1529           VSX(2)   = BLX(2)   + BTX(2)
1530           VSX(3)   = BLX(3)   + BTX(3)
1531           DO 60 L=1, 5
1532             VS1(1,L) =            BT1(1,L)
1533             VS2(1,L) =            BT2(1,L)
1534             VS1(2,L) = BL1(2,L) + BT1(2,L)

1541     C-    in all of the numerical gymnastics above.  The "2" variables

1543           DO 70 ICOM=1, NCOM

1550      
1551           SUBROUTINE BLDIF(ITYP)
1552     C-----------------------------------------------------------
1553     C     Sets up the Newton system coefficients and residuals
1554     C
1555     C        ITYP = 0 :  similarity station
1556     C        ITYP = 1 :  laminar interval
1557     C        ITYP = 2 :  turbulent interval
1558     C        ITYP = 3 :  wake interval
1559     C
1560     C     This routine knows nothing about a transition interval,
1561     C     which is taken care of by TRDIF.
1562     C-----------------------------------------------------------
1563           IMPLICIT REAL(M)
1564           INCLUDE 'XBL.INC'
1565     C
1566           IF(ITYP.EQ.0) THEN
1567     C----- similarity logarithmic differences  (prescribed)
1568            XLOG = 1.0
1569            ULOG = BULE
1570            TLOG = 0.5*(1.0 - BULE)
1571            HLOG = 0.
1572            DDLOG = 0.
1573           ELSE
1574     C----- usual logarithmic differences
1575            XLOG = LOG(X2/X1)
1576            ULOG = LOG(U2/U1)
1577            TLOG = LOG(T2/T1)
1578            HLOG = LOG(HS2/HS1)
1579     C       XLOG = 2.0*(X2-X1)/(X2+X1)
1580     C       ULOG = 2.0*(U2-U1)/(U2+U1)
1581     C       TLOG = 2.0*(T2-T1)/(T2+T1)
1582     C       HLOG = 2.0*(HS2-HS1)/(HS2+HS1)
1583            DDLOG = 1.0
1584           ENDIF
1585     C
1586           DO 55 K=1, 4
1587             VSREZ(K) = 0.
1588             VSM(K) = 0.
1589             VSR(K) = 0.
1590             VSX(K) = 0.
1591             DO 551 L=1, 5
1592               VS1(K,L) = 0.
1593               VS2(K,L) = 0.
1594       551   CONTINUE
1595        55 CONTINUE
1596     C
1597     C---- set triggering constant for local upwinding
1598           HUPWT = 1.0
1599     C
1600     ccc      HDCON = 5.0*HUPWT
1601     ccc      HD_HK1 = 0.0
1602     ccc      HD_HK2 = 0.0
1603     C
1604           HDCON  =  5.0*HUPWT/HK2**2
1605           HD_HK1 =  0.0
1606           HD_HK2 = -HDCON*2.0/HK2
1607     C
1608     C---- use less upwinding in the wake
1609           IF(ITYP.EQ.3) THEN
1610            HDCON  =  HUPWT/HK2**2
1611            HD_HK1 =  0.0
1612            HD_HK2 = -HDCON*2.0/HK2
1613           ENDIF
1614     C
1615     C---- local upwinding is based on local change in  log(Hk-1)
1616     C-    (mainly kicks in at transition)
1617           ARG = ABS((HK2-1.0)/(HK1-1.0))
1618           HL = LOG(ARG)
1619           HL_HK1 = -1.0/(HK1-1.0)
1620           HL_HK2 =  1.0/(HK2-1.0)
1621     C
1622     C---- set local upwinding parameter UPW and linearize it
1623     C
1624     C       UPW = 0.5   Trapezoidal
1625     C       UPW = 1.0   Backward Euler
1626     C
1627           HLSQ = MIN( HL**2 , 15.0 )
1628           EHH = EXP(-HLSQ*HDCON)
1629           UPW = 1.0 - 0.5*EHH
1630           UPW_HL =        EHH * HL  *HDCON
1631           UPW_HD =    0.5*EHH * HLSQ
1632     C
1633           UPW_HK1 = UPW_HL*HL_HK1 + UPW_HD*HD_HK1
1634           UPW_HK2 = UPW_HL*HL_HK2 + UPW_HD*HD_HK2
1635     C
1636           UPW_U1 = UPW_HK1*HK1_U1
1637           UPW_T1 = UPW_HK1*HK1_T1
1638           UPW_D1 = UPW_HK1*HK1_D1
1639           UPW_U2 = UPW_HK2*HK2_U2
1640           UPW_T2 = UPW_HK2*HK2_T2
1641           UPW_D2 = UPW_HK2*HK2_D2
1642           UPW_MS = UPW_HK1*HK1_MS
1643          &       + UPW_HK2*HK2_MS
1644     C
1645     C
1646           IF(ITYP.EQ.0) THEN
1647     C
1648     C***** LE point -->  set zero amplification factor
1649            VS2(1,1) = 1.0
1650            VSR(1)   = 0.
1651            VSREZ(1) = -AMPL2
1652     C
1653           ELSE IF(ITYP.EQ.1) THEN
1654     C
1655     C***** laminar part -->  set amplification equation
1656     C
1657     C----- set average amplification AX over interval X1..X2
1658            CALL AXSET( HK1,    T1,    RT1, AMPL1,  
1659          &             HK2,    T2,    RT2, AMPL2, AMCRIT, IDAMPV,
1660          &      AX, AX_HK1, AX_T1, AX_RT1, AX_A1,
1661          &          AX_HK2, AX_T2, AX_RT2, AX_A2 )
1662     C
1663            REZC = AMPL2 - AMPL1 - AX*(X2-X1)
1664            Z_AX = -(X2-X1)
1665     C
1666            VS1(1,1) = Z_AX* AX_A1  -  1.0
1667            VS1(1,2) = Z_AX*(AX_HK1*HK1_T1 + AX_T1 + AX_RT1*RT1_T1)
1668            VS1(1,3) = Z_AX*(AX_HK1*HK1_D1                        )
1669            VS1(1,4) = Z_AX*(AX_HK1*HK1_U1         + AX_RT1*RT1_U1)
1670            VS1(1,5) =  AX
1671            VS2(1,1) = Z_AX* AX_A2  +  1.0
1672            VS2(1,2) = Z_AX*(AX_HK2*HK2_T2 + AX_T2 + AX_RT2*RT2_T2)
1673            VS2(1,3) = Z_AX*(AX_HK2*HK2_D2                        )         
1674            VS2(1,4) = Z_AX*(AX_HK2*HK2_U2         + AX_RT2*RT2_U2)
1675            VS2(1,5) = -AX
1676            VSM(1)   = Z_AX*(AX_HK1*HK1_MS         + AX_RT1*RT1_MS
1677          &                + AX_HK2*HK2_MS         + AX_RT2*RT2_MS)
1678            VSR(1)   = Z_AX*(                        AX_RT1*RT1_RE
1679          &                                        + AX_RT2*RT2_RE)
1680            VSX(1)   = 0.
1681            VSREZ(1) = -REZC
1682     C
1683           ELSE
1684     C
1685     C***** turbulent part -->  set shear lag equation
1686     C
1687            SA  = (1.0-UPW)*S1  + UPW*S2
1688            CQA = (1.0-UPW)*CQ1 + UPW*CQ2
1689            CFA = (1.0-UPW)*CF1 + UPW*CF2
1690            HKA = (1.0-UPW)*HK1 + UPW*HK2
1691     C
1692            USA = 0.5*(US1 + US2)
1693            RTA = 0.5*(RT1 + RT2)
1694            DEA = 0.5*(DE1 + DE2)
1695            DA  = 0.5*(D1  + D2 )
1696     C
1697     C
1698            IF(ITYP.EQ.3) THEN
1699     C------ increased dissipation length in wake (decrease its reciprocal)
1700             ALD = DLCON
1701            ELSE
1702             ALD = 1.0
1703            ENDIF
1704     C
1705     C----- set and linearize  equilibrium 1/Ue dUe/dx   ...  NEW  12 Oct 94
1706            IF(ITYP.EQ.2) THEN
1707             GCC = GCCON
1708             HKC     = HKA - 1.0 - GCC/RTA
1709             HKC_HKA = 1.0
1710             HKC_RTA =             GCC/RTA**2
1711             IF(HKC .LT. 0.01) THEN
1712              HKC = 0.01
1713              HKC_HKA = 0.0
1714              HKC_RTA = 0.0
1715             ENDIF
1716            ELSE
1717             GCC = 0.0
1718             HKC = HKA - 1.0
1719             HKC_HKA = 1.0
1720             HKC_RTA = 0.0
1721            ENDIF
1722     C
1723            HR     = HKC     / (GACON*ALD*HKA)
1724            HR_HKA = HKC_HKA / (GACON*ALD*HKA) - HR / HKA
1725            HR_RTA = HKC_RTA / (GACON*ALD*HKA)
1726     C
1727            UQ     = (0.5*CFA - HR**2) / (GBCON*DA)
1728            UQ_HKA =   -2.0*HR*HR_HKA  / (GBCON*DA)
1729            UQ_RTA =   -2.0*HR*HR_RTA  / (GBCON*DA)
1730            UQ_CFA =   0.5             / (GBCON*DA)
1731            UQ_DA  = -UQ/DA
1732            UQ_UPW = UQ_CFA*(CF2-CF1) + UQ_HKA*(HK2-HK1)
1733     C
1734            UQ_T1 = (1.0-UPW)*(UQ_CFA*CF1_T1 + UQ_HKA*HK1_T1) + UQ_UPW*UPW_T1
1735            UQ_D1 = (1.0-UPW)*(UQ_CFA*CF1_D1 + UQ_HKA*HK1_D1) + UQ_UPW*UPW_D1
1736            UQ_U1 = (1.0-UPW)*(UQ_CFA*CF1_U1 + UQ_HKA*HK1_U1) + UQ_UPW*UPW_U1
1737            UQ_T2 =      UPW *(UQ_CFA*CF2_T2 + UQ_HKA*HK2_T2) + UQ_UPW*UPW_T2
1738            UQ_D2 =      UPW *(UQ_CFA*CF2_D2 + UQ_HKA*HK2_D2) + UQ_UPW*UPW_D2
1739            UQ_U2 =      UPW *(UQ_CFA*CF2_U2 + UQ_HKA*HK2_U2) + UQ_UPW*UPW_U2
1740            UQ_MS = (1.0-UPW)*(UQ_CFA*CF1_MS + UQ_HKA*HK1_MS) + UQ_UPW*UPW_MS
1741          &       +      UPW *(UQ_CFA*CF2_MS + UQ_HKA*HK2_MS)
1742            UQ_RE = (1.0-UPW)* UQ_CFA*CF1_RE
1743          &       +      UPW * UQ_CFA*CF2_RE
1744     C
1745            UQ_T1 = UQ_T1             + 0.5*UQ_RTA*RT1_T1
1746            UQ_D1 = UQ_D1 + 0.5*UQ_DA
1747            UQ_U1 = UQ_U1             + 0.5*UQ_RTA*RT1_U1
1748            UQ_T2 = UQ_T2             + 0.5*UQ_RTA*RT2_T2
1749            UQ_D2 = UQ_D2 + 0.5*UQ_DA
1750            UQ_U2 = UQ_U2             + 0.5*UQ_RTA*RT2_U2
1751            UQ_MS = UQ_MS             + 0.5*UQ_RTA*RT1_MS
1752          &                           + 0.5*UQ_RTA*RT2_MS
1753            UQ_RE = UQ_RE             + 0.5*UQ_RTA*RT1_RE
1754          &                           + 0.5*UQ_RTA*RT2_RE
1755     C
1756            SCC = SCCON*1.333/(1.0+USA)
1757            SCC_USA = -SCC/(1.0+USA)
1758     C
1759            SCC_US1 = SCC_USA*0.5
1760            SCC_US2 = SCC_USA*0.5
1761     C
1762     C
1763            SLOG = LOG(S2/S1)
1764            DXI = X2 - X1
1765     C
1766            REZC = SCC*(CQA - SA*ALD)*DXI
1767          &      - DEA*2.0*          SLOG
1768          &      + DEA*2.0*(UQ*DXI - ULOG)*DUXCON
1769     C
1770     
1771     c        if(  ! (rt2.gt.1.0e3 .and. rt1.le.1.0e3) .or.
1772     c     &     (rt2.gt.1.0e4 .and. rt1.le.1.0e4) .or.
1773     c     &     (rt2.gt.1.0e5 .and. rt1.le.1.0e5)        ) then
1774     c           gga = (HKA-1.0-GCC/RTA)/HKA / sqrt(0.5*CFA)
1775     c           write(*,4455) rta, hka, gga, cfa, cqa, sa, uq, ulog/dxi
1776     c 4455      format(1x,f7.0, 2f9.4,f10.6,2f8.5,2f10.5)
1777     c        endif
1778     
1779     
1780            Z_CFA = DEA*2.0*UQ_CFA*DXI * DUXCON
1781            Z_HKA = DEA*2.0*UQ_HKA*DXI * DUXCON
1782            Z_DA  = DEA*2.0*UQ_DA *DXI * DUXCON
1783            Z_SL = -DEA*2.0
1784            Z_UL = -DEA*2.0 * DUXCON
1785            Z_DXI = SCC    *(CQA - SA*ALD)     + DEA*2.0*UQ*DUXCON
1786            Z_USA = SCC_USA*(CQA - SA*ALD)*DXI
1787            Z_CQA = SCC*DXI
1788            Z_SA = -SCC*DXI*ALD
1789            Z_DEA = 2.0*((UQ*DXI - ULOG)*DUXCON - SLOG)
1790     C
1791            Z_UPW = Z_CQA*(CQ2-CQ1) + Z_SA *(S2 -S1 )
1792          &       + Z_CFA*(CF2-CF1) + Z_HKA*(HK2-HK1)
1793            Z_DE1 = 0.5*Z_DEA
1794            Z_DE2 = 0.5*Z_DEA
1795            Z_US1 = 0.5*Z_USA
1796            Z_US2 = 0.5*Z_USA
1797            Z_D1  = 0.5*Z_DA
1798            Z_D2  = 0.5*Z_DA
1799            Z_U1  =                 - Z_UL/U1
1800            Z_U2  =                   Z_UL/U2
1801            Z_X1  = -Z_DXI
1802            Z_X2  =  Z_DXI
1803            Z_S1  = (1.0-UPW)*Z_SA  - Z_SL/S1
1804            Z_S2  =      UPW *Z_SA  + Z_SL/S2
1805            Z_CQ1 = (1.0-UPW)*Z_CQA
1806            Z_CQ2 =      UPW *Z_CQA
1807            Z_CF1 = (1.0-UPW)*Z_CFA
1808            Z_CF2 =      UPW *Z_CFA
1809            Z_HK1 = (1.0-UPW)*Z_HKA
1810            Z_HK2 =      UPW *Z_HKA
1811     C
1812            VS1(1,1) = Z_S1
1813            VS1(1,2) =        Z_UPW*UPW_T1 + Z_DE1*DE1_T1 + Z_US1*US1_T1
1814            VS1(1,3) = Z_D1 + Z_UPW*UPW_D1 + Z_DE1*DE1_D1 + Z_US1*US1_D1
1815            VS1(1,4) = Z_U1 + Z_UPW*UPW_U1 + Z_DE1*DE1_U1 + Z_US1*US1_U1
1816            VS1(1,5) = Z_X1
1817            VS2(1,1) = Z_S2
1818            VS2(1,2) =        Z_UPW*UPW_T2 + Z_DE2*DE2_T2 + Z_US2*US2_T2
1819            VS2(1,3) = Z_D2 + Z_UPW*UPW_D2 + Z_DE2*DE2_D2 + Z_US2*US2_D2
1820            VS2(1,4) = Z_U2 + Z_UPW*UPW_U2 + Z_DE2*DE2_U2 + Z_US2*US2_U2
1821            VS2(1,5) = Z_X2
1822            VSM(1)   =        Z_UPW*UPW_MS + Z_DE1*DE1_MS + Z_US1*US1_MS
1823          &                                + Z_DE2*DE2_MS + Z_US2*US2_MS
1824     C
1825            VS1(1,2) = VS1(1,2) + Z_CQ1*CQ1_T1 + Z_CF1*CF1_T1 + Z_HK1*HK1_T1
1826            VS1(1,3) = VS1(1,3) + Z_CQ1*CQ1_D1 + Z_CF1*CF1_D1 + Z_HK1*HK1_D1
1827            VS1(1,4) = VS1(1,4) + Z_CQ1*CQ1_U1 + Z_CF1*CF1_U1 + Z_HK1*HK1_U1
1828     C
1829            VS2(1,2) = VS2(1,2) + Z_CQ2*CQ2_T2 + Z_CF2*CF2_T2 + Z_HK2*HK2_T2
1830            VS2(1,3) = VS2(1,3) + Z_CQ2*CQ2_D2 + Z_CF2*CF2_D2 + Z_HK2*HK2_D2
1831            VS2(1,4) = VS2(1,4) + Z_CQ2*CQ2_U2 + Z_CF2*CF2_U2 + Z_HK2*HK2_U2
1832     C
1833            VSM(1)   = VSM(1)   + Z_CQ1*CQ1_MS + Z_CF1*CF1_MS + Z_HK1*HK1_MS
1834          &                     + Z_CQ2*CQ2_MS + Z_CF2*CF2_MS + Z_HK2*HK2_MS
1835            VSR(1)   =            Z_CQ1*CQ1_RE + Z_CF1*CF1_RE
1836          &                     + Z_CQ2*CQ2_RE + Z_CF2*CF2_RE
1837            VSX(1)   = 0.
1838            VSREZ(1) = -REZC
1839     C
1840           ENDIF
1841     C
1842     C**** Set up momentum equation
1843           HA = 0.5*(H1 + H2)
1844           MA = 0.5*(M1 + M2)
1845           XA = 0.5*(X1 + X2)
1846           TA = 0.5*(T1 + T2)
1847           HWA = 0.5*(DW1/T1 + DW2/T2)
1848     C
1849     C---- set Cf term, using central value CFM for better accuracy in drag
1850           CFX     = 0.50*CFM*XA/TA  +  0.25*(CF1*X1/T1 + CF2*X2/T2)
1851           CFX_XA  = 0.50*CFM   /TA
1852           CFX_TA  = -.50*CFM*XA/TA**2
1853     C
1854           CFX_X1  = 0.25*CF1   /T1     + CFX_XA*0.5
1855           CFX_X2  = 0.25*CF2   /T2     + CFX_XA*0.5
1856           CFX_T1  = -.25*CF1*X1/T1**2  + CFX_TA*0.5
1857           CFX_T2  = -.25*CF2*X2/T2**2  + CFX_TA*0.5
1858           CFX_CF1 = 0.25*    X1/T1
1859           CFX_CF2 = 0.25*    X2/T2
1860           CFX_CFM = 0.50*    XA/TA
1861     C
1862           BTMP = HA + 2.0 - MA + HWA
1863     C
1864           REZT  = TLOG + BTMP*ULOG - XLOG*0.5*CFX
1865           Z_CFX = -XLOG*0.5
1866           Z_HA  =  ULOG
1867           Z_HWA =  ULOG
1868           Z_MA  = -ULOG
1869           Z_XL  =-DDLOG * 0.5*CFX
1870           Z_UL  = DDLOG * BTMP
1871           Z_TL  = DDLOG
1872     C
1873           Z_CFM = Z_CFX*CFX_CFM
1874           Z_CF1 = Z_CFX*CFX_CF1
1875           Z_CF2 = Z_CFX*CFX_CF2
1876     C
1877           Z_T1 = -Z_TL/T1 + Z_CFX*CFX_T1 + Z_HWA*0.5*(-DW1/T1**2)
1878           Z_T2 =  Z_TL/T2 + Z_CFX*CFX_T2 + Z_HWA*0.5*(-DW2/T2**2)
1879           Z_X1 = -Z_XL/X1 + Z_CFX*CFX_X1
1880           Z_X2 =  Z_XL/X2 + Z_CFX*CFX_X2
1881           Z_U1 = -Z_UL/U1
1882           Z_U2 =  Z_UL/U2
1883     C
1884           VS1(2,2) = 0.5*Z_HA*H1_T1 + Z_CFM*CFM_T1 + Z_CF1*CF1_T1 + Z_T1
1885           VS1(2,3) = 0.5*Z_HA*H1_D1 + Z_CFM*CFM_D1 + Z_CF1*CF1_D1
1886           VS1(2,4) = 0.5*Z_MA*M1_U1 + Z_CFM*CFM_U1 + Z_CF1*CF1_U1 + Z_U1
1887           VS1(2,5) =                                                Z_X1
1888           VS2(2,2) = 0.5*Z_HA*H2_T2 + Z_CFM*CFM_T2 + Z_CF2*CF2_T2 + Z_T2
1889           VS2(2,3) = 0.5*Z_HA*H2_D2 + Z_CFM*CFM_D2 + Z_CF2*CF2_D2
1890           VS2(2,4) = 0.5*Z_MA*M2_U2 + Z_CFM*CFM_U2 + Z_CF2*CF2_U2 + Z_U2
1891           VS2(2,5) =                                                Z_X2
1892     C
1893           VSM(2)   = 0.5*Z_MA*M1_MS + Z_CFM*CFM_MS + Z_CF1*CF1_MS
1894          &         + 0.5*Z_MA*M2_MS                + Z_CF2*CF2_MS
1895           VSR(2)   =                  Z_CFM*CFM_RE + Z_CF1*CF1_RE
1896          &                                         + Z_CF2*CF2_RE
1897           VSX(2)   = 0.
1898           VSREZ(2) = -REZT
1899     C
1900     C**** Set up shape parameter equation
1901     C
1902           XOT1 = X1/T1
1903           XOT2 = X2/T2
1904     C
1905           HA  = 0.5*(H1  + H2 )
1906           HSA = 0.5*(HS1 + HS2)
1907           HCA = 0.5*(HC1 + HC2)
1908           HWA = 0.5*(DW1/T1 + DW2/T2)
1909     C
1910           DIX = (1.0-UPW)*DI1*XOT1 + UPW*DI2*XOT2
1911           CFX = (1.0-UPW)*CF1*XOT1 + UPW*CF2*XOT2
1912           DIX_UPW = DI2*XOT2 - DI1*XOT1
1913           CFX_UPW = CF2*XOT2 - CF1*XOT1
1914     C
1915           BTMP = 2.0*HCA/HSA + 1.0 - HA - HWA
1916     C
1917           REZH  = HLOG + BTMP*ULOG + XLOG*(0.5*CFX-DIX)
1918           Z_CFX =  XLOG*0.5
1919           Z_DIX = -XLOG
1920           Z_HCA = 2.0*ULOG/HSA
1921           Z_HA  = -ULOG
1922           Z_HWA = -ULOG
1923           Z_XL  = DDLOG * (0.5*CFX-DIX)
1924           Z_UL  = DDLOG * BTMP
1925           Z_HL  = DDLOG
1926     C
1927           Z_UPW = Z_CFX*CFX_UPW + Z_DIX*DIX_UPW
1928     C
1929           Z_HS1 = -HCA*ULOG/HSA**2 - Z_HL/HS1
1930           Z_HS2 = -HCA*ULOG/HSA**2 + Z_HL/HS2
1931     C
1932           Z_CF1 = (1.0-UPW)*Z_CFX*XOT1
1933           Z_CF2 =      UPW *Z_CFX*XOT2
1934           Z_DI1 = (1.0-UPW)*Z_DIX*XOT1
1935           Z_DI2 =      UPW *Z_DIX*XOT2
1936     C
1937           Z_T1 = (1.0-UPW)*(Z_CFX*CF1 + Z_DIX*DI1)*(-XOT1/T1)
1938           Z_T2 =      UPW *(Z_CFX*CF2 + Z_DIX*DI2)*(-XOT2/T2)
1939           Z_X1 = (1.0-UPW)*(Z_CFX*CF1 + Z_DIX*DI1)/ T1        - Z_XL/X1
1940           Z_X2 =      UPW *(Z_CFX*CF2 + Z_DIX*DI2)/ T2        + Z_XL/X2
1941           Z_U1 =                                              - Z_UL/U1
1942           Z_U2 =                                                Z_UL/U2
1943     C
1944           Z_T1 = Z_T1 + Z_HWA*0.5*(-DW1/T1**2)
1945           Z_T2 = Z_T2 + Z_HWA*0.5*(-DW2/T2**2)
1946     C
1947           VS1(3,1) =                               Z_DI1*DI1_S1
1948           VS1(3,2) = Z_HS1*HS1_T1 + Z_CF1*CF1_T1 + Z_DI1*DI1_T1 + Z_T1
1949           VS1(3,3) = Z_HS1*HS1_D1 + Z_CF1*CF1_D1 + Z_DI1*DI1_D1
1950           VS1(3,4) = Z_HS1*HS1_U1 + Z_CF1*CF1_U1 + Z_DI1*DI1_U1 + Z_U1
1951           VS1(3,5) =                                              Z_X1
1952           VS2(3,1) =                               Z_DI2*DI2_S2
1953           VS2(3,2) = Z_HS2*HS2_T2 + Z_CF2*CF2_T2 + Z_DI2*DI2_T2 + Z_T2
1954           VS2(3,3) = Z_HS2*HS2_D2 + Z_CF2*CF2_D2 + Z_DI2*DI2_D2
1955           VS2(3,4) = Z_HS2*HS2_U2 + Z_CF2*CF2_U2 + Z_DI2*DI2_U2 + Z_U2
1956           VS2(3,5) =                                              Z_X2
1957           VSM(3)   = Z_HS1*HS1_MS + Z_CF1*CF1_MS + Z_DI1*DI1_MS
1958          &         + Z_HS2*HS2_MS + Z_CF2*CF2_MS + Z_DI2*DI2_MS
1959           VSR(3)   = Z_HS1*HS1_RE + Z_CF1*CF1_RE + Z_DI1*DI1_RE
1960          &         + Z_HS2*HS2_RE + Z_CF2*CF2_RE + Z_DI2*DI2_RE
1961     C
1962           VS1(3,2) = VS1(3,2) + 0.5*(Z_HCA*HC1_T1+Z_HA*H1_T1) + Z_UPW*UPW_T1
1963           VS1(3,3) = VS1(3,3) + 0.5*(Z_HCA*HC1_D1+Z_HA*H1_D1) + Z_UPW*UPW_D1
1964           VS1(3,4) = VS1(3,4) + 0.5*(Z_HCA*HC1_U1           ) + Z_UPW*UPW_U1
1965           VS2(3,2) = VS2(3,2) + 0.5*(Z_HCA*HC2_T2+Z_HA*H2_T2) + Z_UPW*UPW_T2
1966           VS2(3,3) = VS2(3,3) + 0.5*(Z_HCA*HC2_D2+Z_HA*H2_D2) + Z_UPW*UPW_D2
1967           VS2(3,4) = VS2(3,4) + 0.5*(Z_HCA*HC2_U2           ) + Z_UPW*UPW_U2
1968     C
1969           VSM(3)   = VSM(3)   + 0.5*(Z_HCA*HC1_MS           ) + Z_UPW*UPW_MS
1970          &                    + 0.5*(Z_HCA*HC2_MS           )
1971     C
1972           VSX(3)   = 0.
1973           VSREZ(3) = -REZH
1974     C
1975           RETURN
1976           END

1980           SUBROUTINE DAMPL( HK, TH, RT, AX, AX_HK, AX_TH, AX_RT )
1981     C==============================================================
1982     C     Amplification rate routine for envelope e^n method.
1983     C     Reference:
1984     C                Drela, M., Giles, M.,
1985     C               "Viscous/Inviscid Analysis of Transonic and
1986     C                Low Reynolds Number Airfoils",
1987     C                AIAA Journal, Oct. 1987.
1988     C
1989     C     NEW VERSION.   March 1991       (latest bug fix  July 93)
1990     C          - m(H) correlation made more accurate up to H=20
1991     C          - for H > 5, non-similar profiles are used 
1992     C            instead of Falkner-Skan profiles.  These 
1993     C            non-similar profiles have smaller reverse 
1994     C            velocities, are more representative of typical 
1995     C            separation bubble profiles.
1996     C--------------------------------------------------------------
1997     C
1998     C     input :   HK     kinematic shape parameter
1999     C               TH     momentum thickness
2000     C               RT     momentum-thickness Reynolds number
2001     C
2002     C     output:   AX     envelope spatial amplification rate
2003     C               AX_(.) sensitivity of AX to parameter (.)
2004     C
2005     C
2006     C     Usage: The log of the envelope amplitude N(x) is
2007     C            calculated by integrating AX (= dN/dx) with
2008     C            respect to the streamwise distance x.
2009     C                      x
2010     C                     /
2011     C              N(x) = | AX(H(x),Th(x),Rth(x)) dx
2012     C                     /
2013     C                      0
2014     C            The integration can be started from the leading
2015     C            edge since AX will be returned as zero when RT
2016     C            is below the critical Rtheta.  Transition occurs
2017     C            when N(x) reaches Ncrit (Ncrit= 9 is "standard").
2018     C==============================================================
2019           IMPLICIT REAL (A-H,M,O-Z)
2020     ccc   DATA DGR / 0.04 /
2021           DATA DGR / 0.08 /
2022     C
2023           HMI = 1.0/(HK - 1.0)
2024           HMI_HK = -HMI**2
2025     C
2026     C---- log10(Critical Rth) - H   correlation for Falkner-Skan profiles
2027           AA    = 2.492*HMI**0.43
2028           AA_HK =   (AA/HMI)*0.43 * HMI_HK
2029     C
2030           BB    = TANH(14.0*HMI - 9.24)
2031           BB_HK = (1.0 - BB*BB) * 14.0 * HMI_HK
2032     C
2033           GRCRIT = AA    + 0.7*(BB + 1.0)
2034           GRC_HK = AA_HK + 0.7* BB_HK
2035     C
2036     C
2037           GR = LOG10(RT)
2038           GR_RT = 1.0 / (2.3025851*RT)
2039     C
2040           IF(GR .LT. GRCRIT-DGR) THEN
2041     C
2042     C----- no amplification for Rtheta < Rcrit
2043            AX    = 0.
2044            AX_HK = 0.
2045            AX_TH = 0.
2046            AX_RT = 0.
2047     C
2048           ELSE
2049     C
2050     C----- Set steep cubic ramp used to turn on AX smoothly as Rtheta 
2051     C-     exceeds Rcrit (previously, this was done discontinuously).
2052     C-     The ramp goes between  -DGR < log10(Rtheta/Rcrit) < DGR
2053     C
2054            RNORM = (GR - (GRCRIT-DGR)) / (2.0*DGR)
2055            RN_HK =     -  GRC_HK       / (2.0*DGR)
2056            RN_RT =  GR_RT              / (2.0*DGR)
2057     C
2058            IF(RNORM .GE. 1.0) THEN
2059             RFAC    = 1.0
2060             RFAC_HK = 0.
2061             RFAC_RT = 0.
2062            ELSE
2063             RFAC    = 3.0*RNORM**2 - 2.0*RNORM**3
2064             RFAC_RN = 6.0*RNORM    - 6.0*RNORM**2
2065     C
2066             RFAC_HK = RFAC_RN*RN_HK
2067             RFAC_RT = RFAC_RN*RN_RT
2068            ENDIF
2069     C
2070     C----- Amplification envelope slope correlation for Falkner-Skan
2071            ARG    = 3.87*HMI    - 2.52
2072            ARG_HK = 3.87*HMI_HK
2073     C
2074            EX    = EXP(-ARG**2)
2075            EX_HK = EX * (-2.0*ARG*ARG_HK)
2076     C
2077            DADR    = 0.028*(HK-1.0) - 0.0345*EX
2078            DADR_HK = 0.028          - 0.0345*EX_HK
2079     C
2080     C----- new m(H) correlation    1 March 91
2081            AF = -0.05 + 2.7*HMI -  5.5*HMI**2 + 3.0*HMI**3
2082            AF_HMI =     2.7     - 11.0*HMI    + 9.0*HMI**2
2083            AF_HK = AF_HMI*HMI_HK
2084     C
2085            AX    = (AF   *DADR/TH                ) * RFAC
2086            AX_HK = (AF_HK*DADR/TH + AF*DADR_HK/TH) * RFAC
2087          &       + (AF   *DADR/TH                ) * RFAC_HK
2088            AX_TH = -AX/TH
2089            AX_RT = (AF   *DADR/TH                ) * RFAC_RT
2090     C
2091           ENDIF
2092     C
2093           RETURN
2094           END ! DAMPL

2098           SUBROUTINE DAMPL2( HK, TH, RT, AX, AX_HK, AX_TH, AX_RT )
2099     C==============================================================
2100     C     Amplification rate routine for modified envelope e^n method.
2101     C     Reference: 
2102     C                Drela, M., Giles, M.,
2103     C               "Viscous/Inviscid Analysis of Transonic and 
2104     C                Low Reynolds Number Airfoils", 
2105     C                AIAA Journal, Oct. 1987.
2106     C
2107     C     NEWER VERSION.   Nov 1996
2108     C          - Amplification rate changes to the Orr-Sommerfeld
2109     C              maximum ai(H,Rt) function for H > 4 .
2110     C          - This implicitly assumes that the frequency range
2111     C              (around w = 0.09 Ue/theta) which experiences this 
2112     C              maximum amplification rate contains the currently
2113     C              most-amplified frequency.
2114     C--------------------------------------------------------------
2115     C
2116     C     input :   HK     kinematic shape parameter
2117     C               TH     momentum thickness
2118     C               RT     momentum-thickness Reynolds number
2119     C
2120     C     output:   AX     envelope spatial amplification rate
2121     C               AX_(.) sensitivity of AX to parameter (.)
2122     C
2123     C
2124     C     Usage: The log of the envelope amplitude N(x) is 
2125     C            calculated by integrating AX (= dN/dx) with 
2126     C            respect to the streamwise distance x.
2127     C                      x
2128     C                     /
2129     C              N(x) = | AX(H(x),Th(x),Rth(x)) dx
2130     C                     /
2131     C                      0
2132     C            The integration can be started from the leading
2133     C            edge since AX will be returned as zero when RT
2134     C            is below the critical Rtheta.  Transition occurs
2135     C            when N(x) reaches Ncrit (Ncrit= 9 is "standard").
2136     C==============================================================
2137           IMPLICIT REAL (A-H,M,O-Z)
2138           DATA DGR / 0.08 /
2139           DATA HK1, HK2 / 3.5, 4.0 /
2140     C
2141           HMI = 1.0/(HK - 1.0)
2142           HMI_HK = -HMI**2
2143     C
2144     C---- log10(Critical Rth) -- H   correlation for Falkner-Skan profiles
2145           AA    = 2.492*HMI**0.43
2146           AA_HK =   (AA/HMI)*0.43 * HMI_HK
2147     C
2148           BB    = TANH(14.0*HMI - 9.24)
2149           BB_HK = (1.0 - BB*BB) * 14.0 * HMI_HK
2150     C
2151           GRC = AA    + 0.7*(BB + 1.0)
2152           GRC_HK = AA_HK + 0.7* BB_HK
2153     C
2154     C
2155           GR = LOG10(RT)
2156           GR_RT = 1.0 / (2.3025851*RT)
2157     C
2158           IF(GR .LT. GRC-DGR) THEN
2159     C
2160     C----- no amplification for Rtheta < Rcrit
2161            AX    = 0.
2162            AX_HK = 0.
2163            AX_TH = 0.
2164            AX_RT = 0.
2165     C
2166           ELSE
2167     C
2168     C----- Set steep cubic ramp used to turn on AX smoothly as Rtheta 
2169     C-     exceeds Rcrit (previously, this was done discontinuously).
2170     C-     The ramp goes between  -DGR < log10(Rtheta/Rcrit) < DGR
2171     C
2172            RNORM = (GR - (GRC-DGR)) / (2.0*DGR)
2173            RN_HK =     -  GRC_HK       / (2.0*DGR)
2174            RN_RT =  GR_RT              / (2.0*DGR)
2175     C
2176            IF(RNORM .GE. 1.0) THEN
2177             RFAC    = 1.0
2178             RFAC_HK = 0.
2179             RFAC_RT = 0.
2180            ELSE
2181             RFAC    = 3.0*RNORM**2 - 2.0*RNORM**3
2182             RFAC_RN = 6.0*RNORM    - 6.0*RNORM**2
2183     C
2184             RFAC_HK = RFAC_RN*RN_HK
2185             RFAC_RT = RFAC_RN*RN_RT
2186            ENDIF
2187     C
2188     C
2189     C----- set envelope amplification rate with respect to Rtheta
2190     C-       DADR = d(N)/d(Rtheta) = f(H)
2191     C
2192            ARG    = 3.87*HMI    - 2.52
2193            ARG_HK = 3.87*HMI_HK
2194     C
2195            EX    = EXP(-ARG**2)
2196            EX_HK = EX * (-2.0*ARG*ARG_HK)
2197     C
2198            DADR    = 0.028*(HK-1.0) - 0.0345*EX
2199            DADR_HK = 0.028          - 0.0345*EX_HK
2200     C
2201     C
2202     C----- set conversion factor from d/d(Rtheta) to d/dx
2203     C-       AF = Theta d(Rtheta)/dx = f(H)
2204     C
2205            BRG = -20.0*HMI
2206            AF = -0.05 + 2.7*HMI -  5.5*HMI**2 + 3.0*HMI**3 + 0.1*EXP(BRG)
2207            AF_HMI =     2.7     - 11.0*HMI    + 9.0*HMI**2 - 2.0*EXP(BRG)
2208            AF_HK = AF_HMI*HMI_HK
2209     C
2210     C
2211     C----- set amplification rate with respect to x, 
2212     C-     with RFAC shutting off amplification when below Rcrit
2213     C
2214            AX    = (AF   *DADR/TH                ) * RFAC
2215            AX_HK = (AF_HK*DADR/TH + AF*DADR_HK/TH) * RFAC
2216          &       + (AF   *DADR/TH                ) * RFAC_HK
2217            AX_TH = -AX/TH
2218            AX_RT = (AF   *DADR/TH                ) * RFAC_RT
2219     C
2220           ENDIF
2221     C
2222           IF(HK .LT. HK1) RETURN
2223     C
2224     C---- non-envelope max-amplification correction for separated profiles
2225     C
2226           HNORM = (HK - HK1) / (HK2 - HK1)
2227           HN_HK =       1.0  / (HK2 - HK1)
2228     C
2229     C---- set blending fraction HFAC = 0..1 over HK1 < HK < HK2
2230           IF(HNORM .GE. 1.0) THEN
2231            HFAC = 1.0
2232            HF_HK = 0.
2233           ELSE
2234            HFAC  =  3.0*HNORM**2 - 2.0*HNORM**3
2235            HF_HK = (6.0*HNORM    - 6.0*HNORM**2)*HN_HK
2236           ENDIF
2237     C
2238     C---- "normal" envelope amplification rate AX1
2239           AX1    = AX
2240           AX1_HK = AX_HK
2241           AX1_TH = AX_TH
2242           AX1_RT = AX_RT
2243     C
2244     C---- set modified amplification rate AX2
2245           GR0 = 0.30 + 0.35 * EXP(-0.15*(HK-5.0))
2246           GR0_HK =   - 0.35 * EXP(-0.15*(HK-5.0)) * 0.15
2247     C
2248           TNR = TANH(1.2*(GR - GR0))
2249           TNR_RT =  (1.0 - TNR**2)*1.2*GR_RT
2250           TNR_HK = -(1.0 - TNR**2)*1.2*GR0_HK
2251     C
2252           AX2    = (0.086*TNR    -     0.25/(HK-1.0)**1.5) / TH
2253           AX2_HK = (0.086*TNR_HK + 1.5*0.25/(HK-1.0)**2.5) / TH
2254           AX2_RT = (0.086*TNR_RT                         ) / TH
2255           AX2_TH = -AX2/TH
2256     C
2257           IF(AX2 .LT. 0.0) THEN
2258            AX2    = 0.0
2259            AX2_HK = 0.
2260            AX2_RT = 0.
2261            AX2_TH = 0.
2262           ENDIF
2263     C
2264     C---- blend the two amplification rates
2265           AX    = HFAC*AX2    + (1.0 - HFAC)*AX1
2266           AX_HK = HFAC*AX2_HK + (1.0 - HFAC)*AX1_HK + HF_HK*(AX2-AX1)
2267           AX_RT = HFAC*AX2_RT + (1.0 - HFAC)*AX1_RT
2268           AX_TH = HFAC*AX2_TH + (1.0 - HFAC)*AX1_TH
2269     C
2270           RETURN
2271           END ! DAMPL2

2291     C---- Laminar dissipation function  ( 2 CD/H* )     (from Falkner-Skan)
2292           IF(HK.LT.4.0) THEN

2295           ELSE

2299            DI_HK = ( -.0016*2.0*HKB*(1.0/DEN - 0.02*HKB**2/DEN**2) ) / RT
2300           ENDIF

2302     C

2307           SUBROUTINE DILW( HK, RT, DI, DI_HK, DI_RT )
2308           REAL MSQ
2309     C
2310           MSQ = 0.
2311           CALL HSL( HK, RT, MSQ, HS, HS_HK, HS_RT, HS_MSQ )
2312     C
2313     C---- Laminar wake dissipation function  ( 2 CD/H* )
2314           RCD    =  1.10 * (1.0 - 1.0/HK)**2  / HK

2320           DI_RT = -DI/RT             - (DI/HS)*HS_RT
2321     C

2385           SUBROUTINE HST( HK, RT, MSQ, HS, HS_HK, HS_RT, HS_MSQ )
2386           IMPLICIT REAL (A-H,M,O-Z)
2387     C
2388     C---- Turbulent HS correlation
2389     C
2390           DATA HSMIN, DHSINF / 1.500, 0.015 /
2391     C
2392     C---- ###  12/4/94
2393     C---- limited Rtheta dependence for Rtheta < 200
2394     C
2395     C
2396           IF(RT.GT.400.0) THEN
2397            HO    = 3.0 + 400.0/RT
2398            HO_RT =     - 400.0/RT**2
2399           ELSE
2400            HO    = 4.0
2401            HO_RT = 0.
2402           ENDIF
2403     C
2404           IF(RT.GT.200.0) THEN
2405            RTZ    = RT
2406            RTZ_RT = 1.
2407           ELSE
2408            RTZ    = 200.0
2409            RTZ_RT = 0.
2410           ENDIF
2411     C
2412           IF(HK.LT.HO) THEN
2413     C----- attached branch
2414     C=======================================================
2415     C----- old correlation
2416     C-     (from Swafford profiles)
2417     c       SRT = SQRT(RT)
2418     c       HEX = (HO-HK)**1.6
2419     c       RTMP = 0.165 - 1.6/SRT
2420     c       HS    = HSMIN + 4.0/RT + RTMP*HEX/HK
2421     c       HS_HK = RTMP*HEX/HK*(-1.6/(HO-HK) - 1.0/HK)
2422     c       HS_RT = -4.0/RT**2 + HEX/HK*0.8/SRT/RT
2423     c     &             + RTMP*HEX/HK*1.6/(HO-HK)*HO_RT
2424     C=======================================================
2425     C----- new correlation  29 Nov 91
2426     C-     (from  arctan(y+) + Schlichting  profiles)
2427            HR    = ( HO - HK)/(HO-1.0)
2428            HR_HK =      - 1.0/(HO-1.0)
2429            HR_RT = (1.0 - HR)/(HO-1.0) * HO_RT
2430            HS    = (2.0-HSMIN-4.0/RTZ)*HR**2  * 1.5/(HK+0.5) + HSMIN
2431          &       + 4.0/RTZ
2432            HS_HK =-(2.0-HSMIN-4.0/RTZ)*HR**2  * 1.5/(HK+0.5)**2
2433          &       + (2.0-HSMIN-4.0/RTZ)*HR*2.0 * 1.5/(HK+0.5) * HR_HK
2434            HS_RT = (2.0-HSMIN-4.0/RTZ)*HR*2.0 * 1.5/(HK+0.5) * HR_RT
2435          &       + (HR**2 * 1.5/(HK+0.5) - 1.0)*4.0/RTZ**2 * RTZ_RT
2436     C
2437           ELSE
2438     C
2439     C----- separated branch
2440            GRT = LOG(RTZ)
2441            HDIF = HK - HO 
2442            RTMP = HK - HO + 4.0/GRT
2443            HTMP    = 0.007*GRT/RTMP**2 + DHSINF/HK
2444            HTMP_HK = -.014*GRT/RTMP**3 - DHSINF/HK**2
2445            HTMP_RT = -.014*GRT/RTMP**3 * (-HO_RT - 4.0/GRT**2/RTZ * RTZ_RT)
2446          &         + 0.007    /RTMP**2 / RTZ * RTZ_RT
2447            HS    = HDIF**2 * HTMP + HSMIN + 4.0/RTZ
2448            HS_HK = HDIF*2.0* HTMP
2449          &       + HDIF**2 * HTMP_HK
2450            HS_RT = HDIF**2 * HTMP_RT      - 4.0/RTZ**2 * RTZ_RT
2451          &       + HDIF*2.0* HTMP * (-HO_RT)
2452     C
2453           ENDIF
2454     C
2455     C---- fudge HS slightly to make sure   HS -> 2   as   HK -> 1
2456     C-    (unnecessary with new correlation)
2457     c      HTF    = 0.485/9.0 * (HK-4.0)**2/HK  +  1.515
2458     c      HTF_HK = 0.485/9.0 * (1.0-16.0/HK**2)
2459     c      ARG = MAX( 10.0*(1.0 - HK) , -15.0 )
2460     c      HXX = EXP(ARG)
2461     c      HXX_HK = -10.0*HXX
2462     cC
2463     c      HS_HK  = (1.0-HXX)*HS_HK  +  HXX*HTF_HK
2464     c     &       + (        -HS     +      HTF    )*HXX_HK
2465     c      HS_RT  = (1.0-HXX)*HS_RT
2466     c      HS     = (1.0-HXX)*HS     +  HXX*HTF
2467     C
2468     C---- Whitfield's minor additional compressibility correction
2469           FM = 1.0 + 0.014*MSQ
2470           HS     = ( HS + 0.028*MSQ ) / FM
2471           HS_HK  = ( HS_HK          ) / FM
2472           HS_RT  = ( HS_RT          ) / FM
2473           HS_MSQ = 0.028/FM  -  0.014*HS/FM
2474     C
2475           RETURN
2476           END

2481           IMPLICIT REAL (A-H,M,O-Z)
