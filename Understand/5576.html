21           SUBROUTINE PRWALL(DSTAR,THETA,UO,RT,MS,CT, BB,
22          &                  DO, DO_DS, DO_TH, DO_UO, DO_RT, DO_MS,
23          &                  UI, UI_DS, UI_TH, UI_UO, UI_RT, UI_MS,
24          &                  HS, HS_DS, HS_TH, HS_UO, HS_RT, HS_MS,
25          &                  CF, CF_DS, CF_TH, CF_UO, CF_RT, CF_MS,
26          &                  CD, CD_DS, CD_TH, CD_UO, CD_RT, CD_MS, CD_CT )
27           IMPLICIT REAL (A-H,M,O-Z)
28     C================================================================
29     C     Returns wall slip velocity and thickness of wall BL profile
30     C
31     C     Input:
32     C        DSTAR  kinematic displacement thickness
33     C        THETA  kinematic momentum thickness
34     C        RT     momentum thickness based on ue and THETA
35     C        MS     Mach^2 based on ue
36     C
37     C        UO     uo/ue outer velocity;  assumed = 1  in this version
38     C
39     C     Output:
40     C        BB     outer profile exponent
41     C        DO     thickness of profile deck
42     C        UI     inner "slip" velocity
43     C        CF     wall skin friction
44     C================================================================
45     C
46           PARAMETER (N=65)
47           DIMENSION ETA(N), UIP(N), UIP_DP(N), G(N), G_BB(N)
48     C
49     C---- pi/2 ,  2/pi
50           DATA HPI, TOPI / 1.570796327 , 0.6366197723 /
51     C
52           DATA T, SQT / 0.28 , 0.5291502622 /
53     C
54     C---- TCON = ( atan(T^1/2) / T^1/2  -  1/(1+T)) / (2T)  +  0.5/(1.0 + T) 
55     C                - atan(1/T^1/2) / 2T^1/2
56           DATA TCON / -0.3864027035 /
57     C       
58     C---- slip velocity coefficient
59           DATA AK / 0.09 /
60     C
61     C---- log-law constants
62           DATA VKAP, VB / 0.40 , 5.0 /
63     C
64           HK = DSTAR/THETA
65     C
66           UO = 1.0
67           BB = 1.0
68     C
69     C---- initialize variables
70           CALL CFT(HK,RT,MS,CF,CF_HK,CF_RT,CF_MS)
71           SGN = SIGN( 1.0 , CF )
72           UT = SGN * SQRT(0.5*ABS(CF))
73     C
74           UI = MIN( UT/AK * HPI , 0.90 )
75           DO = HK*THETA / (1.0 - 0.5*(UO+UI))
76     C
77           EBK = EXP(-VB*VKAP)
78     C
79           DO 1000 ITER=1, 12
80     C
81             SGN = SIGN( 1.0 , UT )
82     C
83     C------ set d+ = DP(UT DO ; RT TH)
84             DP    = SGN * UT*RT*DO/THETA
85             DP_DO = SGN * UT*RT   /THETA
86             DP_UT = SGN *    RT*DO/THETA
87     C
88             DP_TH = -DP/THETA
89             DP_RT = SGN * UT   *DO/THETA
90             DP_MS = 0.0
91     C
92     C------ estimate inner profile edge velocity Ui+ using log-law
93             UPE = LOG(DP)/VKAP + VB
94     C
95     C------ converge exact Ui+ using Spalding formula
96             DO 10 ITUP=1, 5
97               UK = UPE*VKAP
98               ARG = UK - VB*VKAP
99               EXU = EXP(ARG)
100               REZ  = UPE +  EXU - EBK*(1.0 + UK + UK*UK/2.0 + UK*UK*UK/6.0)
101          &         - DP
102               DP_U = 1.0 + (EXU - EBK*(1.0 + UK + UK*UK/2.0))*VKAP
103     C
104               IF(ABS(REZ/DP) .LT. 1.0E-5) GO TO 11
105     C
106               DUPE = -REZ/DP_U
107               UPE = UPE + DUPE
108      10     CONTINUE
109             WRITE(*,*) 'PRWALL: Ue+ convergence failed,  Res =', REZ/DP
110      11     CONTINUE
111     C
112             UPE_DP = 1.0/DP_U
113     C
114     C            2      2        3      3
115     C------ set d y+/du+   and  d y+/du+   at BL edge
116             DP_UU  = (EXU - EBK*(1.0 + UK))*VKAP**2
117             DP_UUU = (EXU - EBK           )*VKAP**3
118     C
119     C------ set  du+/dy+  at BL edge
120             UPD    = 1.0/DP_U
121             UPD_DP = (-1.0/DP_U**3) * DP_UU
122     C
123     C             2      2
124     C------ set  d u+/dy+   at BL edge
125     CCC     UPD_DP = (-1.0/DP_U**3) * DP_UU
126             UPDD    = UPD_DP
127             UPDD_DP = (-1.0/DP_U**4) * DP_UUU
128          &          + ( 3.0/DP_U**5) * DP_UU**2
129     C
130     C------ set coefficients for Spalding profile correction polynomial
131             DC2    = 0.5*DP*DP*UPDD    - DP*UPD
132             DC2_DP =        DP*UPDD    -    UPD
133          &         + 0.5*DP*DP*UPDD_DP - DP*UPD_DP
134     C
135             DC3    = -(  DP*DP*UPDD    - DP*UPD   ) / 3.0
136             DC3_DP = -(2.0 *DP*UPDD    -    UPD   ) / 3.0
137          &           -(  DP*DP*UPDD_DP - DP*UPD_DP) / 3.0
138     C
139     C------ set outer profile amplitude DUO
140             DUO    = UO - UT*(UPE    + DC2    + DC3   )
141             DUO_DP =    - UT*(UPE_DP + DC2_DP + DC3_DP)
142     C
143             DUO_UT =    -    (UPE    + DC2    + DC3   )
144          &         + DUO_DP*DP_UT
145             DUO_DO = DUO_DP*DP_DO
146     C
147             DUO_TH = DUO_DP*DP_TH
148             DUO_RT = DUO_DP*DP_RT
149             DUO_MS = DUO_DP*DP_MS
150     c
151     c        write(*,*) 'dUo', duo, duo_dp, duo_ut, duo_do
152     c        read(*,*) ddo, dut
153     c        if(ddo.ne.0.0 .or. dut.ne.0.0) then
154     c          do = do+ddo
155     c          ut = ut+dut
156     c          write(*,*) 'new', duo + duo_do*ddo + duo_ut*dut
157     c          go to 666
158     c        endif
159     C
160     C------ set wake profile coefficients
161             BB1    =  3.0*(BB    +2.0)*(BB+3.0)/(BB+7.0)
162             BB1_BB =  3.0*(BB*2.0+5.0         )/(BB+7.0) - BB1/(BB+7.0)
163             BB2    = -5.0*(BB    +1.0)*(BB+3.0)/(BB+7.0)
164             BB2_BB = -5.0*(BB*2.0+4.0         )/(BB+7.0) - BB2/(BB+7.0)
165             BB3    =  2.0*(BB    +1.0)*(BB+2.0)/(BB+7.0)
166             BB3_BB =  2.0*(BB*2.0+3.0         )/(BB+7.0) - BB3/(BB+7.0)
167     C
168     C------ fill eta coordinate and inner profile arrays
169     CCC     EXUPE = EXP(UPE*VKAP - VB*VKAP)
170             EXUPE = EXU
171     C
172             DEXU = (EXUPE - EBK)/FLOAT(N-1)
173     C
174             I = 1
175             UIP(I)    = 0.0
176             UIP_DP(I) = 0.0
177             G(I)    = 0.0
178             G_BB(I) = 0.0
179     C
180             DO 20 I=2, N
181     ccc       EXU = EBK + DEXU*FLOAT(I-1)
182               EXU = EBK + (DEXU - 0.75*DEXU*FLOAT(N-I)/FLOAT(N-1))
183          &                *FLOAT(I-1)
184     C
185     CCC       UK = UP*VKAP
186               UK = LOG(EXU) + VB*VKAP
187     C
188               UP = UK/VKAP
189     C
190     C-------- set "inverse" Spalding profile  y+(u+)  and derivatives
191               YP    =  UP +  EXU - EBK*(1.0 + UK + UK*UK/2.0 + UK*UK*UK/6.0)
192               YP_U  = 1.0 + (EXU - EBK*(1.0 + UK + UK*UK/2.0))*VKAP
193               YP_UU =       (EXU - EBK*(1.0 + UK            ))*VKAP**2
194     C
195               ET = YP/DP
196     C
197     C-------- set final inner profile (fudged Spalding)
198               UIP(I)     = UP       +     DC2   *ET**2 +     DC3   *ET**3
199               UIP_DP(I)  =                DC2_DP*ET**2 +     DC3_DP*ET**3
200     C
201     ccc          UIPD(I)    = 1.0/YP_U + 2.0*DC2   *ET    + 3.0*DC3   *ET**2
202     ccc          UIPD_DP(I) = (-1.0/YP_U**3)*YPUU
203     ccc     &                          + 2.0*DC2_DP*ET    + 3.0*DC3_DP*ET**2
204     C
205     C-------- set outer profile
206               ETB = ET**BB
207               ALE = LOG(ET)
208     C
209     ccc          G(I)    =  2.0*ETB -     ETB**2
210     ccc          G_BB(I) = (2.0*ETB - 2.0*ETB**2)*ALE
211     C
212               G(I)    = (BB1   *ET + BB2   *ET**2 + BB3   *ET**3)*ETB
213               G_BB(I) = (BB1_BB*ET + BB2_BB*ET**2 + BB3_BB*ET**3)*ETB
214          &            + G(I)*ALE
215     C
216               ETA(I) = ET
217     C
218      20     CONTINUE
219     C
220     C
221             DSN    = 0.0
222             DSN_DO = 0.0
223             DSN_UT = 0.0
224             DSN_BB = 0.0
225     C
226             DSN_TH = 0.0
227             DSN_RT = 0.0
228             DSN_MS = 0.0
229     C
230     C
231             THN    = 0.0
232             THN_DO = 0.0
233             THN_UT = 0.0
234             THN_BB = 0.0
235     C
236             THN_TH = 0.0
237             THN_RT = 0.0
238             THN_MS = 0.0
239     C
240     c        TSN    = 0.0
241     c        TSN_DO = 0.0
242     c        TSN_UT = 0.0
243     c        TSN_BB = 0.0
244     c
245     c        TSN_TH = 0.0
246     c        TSN_RT = 0.0
247     c        TSN_MS = 0.0
248     C
249     C------ perform integration
250             DO 100 I=1, N-1
251               DETA = ETA(I+1) - ETA(I)
252               GA    = 0.5*(G(I+1)    + G(I)   )
253               GA_BB = 0.5*(G_BB(I+1) + G_BB(I))
254     C
255               UIPA    = 0.5*(UIP(I+1)    + UIP(I)   )
256               UIPA_DP = 0.5*(UIP_DP(I+1) + UIP_DP(I))
257     C
258               U    = UT*UIPA    + DUO   *GA
259               U_DP = UT*UIPA_DP
260     C
261               U_DO =              DUO_DO*GA + U_DP*DP_DO
262               U_UT =    UIPA    + DUO_UT*GA + U_DP*DP_UT
263               U_BB =              DUO   *GA_BB
264     C
265               U_TH =              DUO_TH*GA + U_DP*DP_TH
266               U_RT =              DUO_RT*GA + U_DP*DP_RT
267               U_MS =              DUO_MS*GA + U_DP*DP_MS
268     C
269     C
270               DSN    = DSN    + (1.0 - U   )*DETA
271               DSN_DO = DSN_DO -        U_DO *DETA
272               DSN_UT = DSN_UT -        U_UT *DETA
273               DSN_BB = DSN_BB -        U_BB *DETA
274     C
275               DSN_TH = DSN_TH -        U_TH *DETA
276               DSN_RT = DSN_RT -        U_RT *DETA
277               DSN_MS = DSN_MS -        U_MS *DETA
278     C
279     C
280               THN    = THN    + (U   -   U*U)     *DETA
281               THN_DO = THN_DO + (1.0 - 2.0*U)*U_DO*DETA
282               THN_UT = THN_UT + (1.0 - 2.0*U)*U_UT*DETA
283               THN_BB = THN_BB + (1.0 - 2.0*U)*U_BB*DETA
284     C
285               THN_TH = THN_TH + (1.0 - 2.0*U)*U_TH*DETA
286               THN_RT = THN_RT + (1.0 - 2.0*U)*U_RT*DETA
287               THN_MS = THN_MS + (1.0 - 2.0*U)*U_MS*DETA
288     C
289     c          TSN    = TSN    + (U   -   U*U*U)     *DETA
290     c          TSN_DO = TSN_DO + (1.0 - 3.0*U*U)*U_DO*DETA
291     c          TSN_UT = TSN_UT + (1.0 - 3.0*U*U)*U_UT*DETA
292     c          TSN_BB = TSN_BB + (1.0 - 3.0*U*U)*U_BB*DETA
293     C
294     c          TSN_TH = TSN_TH + (1.0 - 3.0*U*U)*U_TH*DETA
295     c          TSN_RT = TSN_RT + (1.0 - 3.0*U*U)*U_RT*DETA
296     c          TSN_MS = TSN_MS + (1.0 - 3.0*U*U)*U_MS*DETA
297     C
298      100    CONTINUE
299     C
300     C------ set up 2x2 system for DO UT
301             REZ1 = DO*DSN    - THETA*HK
302             A11  = DO*DSN_DO + DSN
303             A12  = DO*DSN_UT
304     cc        A12  = DO*DSN_BB
305     C
306             REZ2 = DO*THN    - THETA
307             A21  = DO*THN_DO + THN
308             A22  = DO*THN_UT
309     cc        A22  = DO*THN_BB
310     C
311     cc      IF(ABS(REZ1/THETA) .LT. 2.0E-5 .AND.
312     cc   &     ABS(REZ2/THETA) .LT. 2.0E-5      ) GO TO 1010
313             IF(ABS(REZ1/THETA) .LT. 1.0E-3 .AND.
314          &     ABS(REZ2/THETA) .LT. 1.0E-3      ) GO TO 1010
315     C
316             DET = A11*A22 - A12*A21
317             B11 =  A22/DET
318             B12 = -A12/DET
319             B21 = -A21/DET
320             B22 =  A11/DET
321     C
322             DDO = -(B11*REZ1 + B12*REZ2)
323             DUT = -(B21*REZ1 + B22*REZ2)
324     cc        DBB = -(B21*REZ1 + B22*REZ2)
325     C
326             DMAX = MAX( ABS(DDO/DO) , ABS(DUT/0.05) )
327     cc        DMAX = MAX( ABS(DDO/DO) , ABS(DBB/BB  ) )
328             RLX = 1.0
329             IF(DMAX.GT.0.5) RLX = 0.5/DMAX
330     C
331             DO = DO + RLX*DDO
332             UT = UT + RLX*DUT
333     cc        BB = BB + RLX*DBB
334     c
335     cc        write(*,*) iter, do, ut, rez1, rez2
336     cc        write(*,*) iter, do, bb, rez1, rez2
337     C
338      1000 CONTINUE
339     C
340           WRITE(*,*) 'PRWALL: Convergence failed. Res =', REZ1, REZ2
341     C
342      1010 CONTINUE
343     C
344     C
345     C
346     CCC   REZ1  = DO*DSN    - THETA*HK
347           Z1_HK =           - THETA
348           Z1_TH = DO*DSN_TH -       HK
349           Z1_RT = DO*DSN_RT
350           Z1_MS = DO*DSN_MS
351     C
352     CCC   REZ2  = DO*THN    - THETA
353           Z2_HK = 0.0
354           Z2_TH = DO*THN_TH - 1.0
355           Z2_RT = DO*THN_RT
356           Z2_MS = DO*THN_MS
357     C
358           DO_HK = -(B11*Z1_HK + B12*Z2_HK)
359           DO_TH = -(B11*Z1_TH + B12*Z2_TH)
360           DO_RT = -(B11*Z1_RT + B12*Z2_RT)
361           DO_MS = -(B11*Z1_MS + B12*Z2_MS)
362     C
363           UT_HK = 0.0
364           UT_TH = 0.0
365           UT_RT = 0.0
366           UT_MS = 0.0
367     C
368           BB_HK = 0.0
369           BB_TH = 0.0
370           BB_RT = 0.0
371           BB_MS = 0.0
372     C
373           UT_HK = -(B21*Z1_HK + B22*Z2_HK)
374           UT_TH = -(B21*Z1_TH + B22*Z2_TH)
375           UT_RT = -(B21*Z1_RT + B22*Z2_RT)
376           UT_MS = -(B21*Z1_MS + B22*Z2_MS)
377     C
378     cc      BB_HK = -(B21*Z1_HK + B22*Z2_HK)
379     cc      BB_TH = -(B21*Z1_TH + B22*Z2_TH)
380     cc      BB_RT = -(B21*Z1_RT + B22*Z2_RT)
381     cc      BB_MS = -(B21*Z1_MS + B22*Z2_MS)
382     C
383     C
384     C---- set and linearize Cf
385           CF    = SGN*2.0*UT**2
386           CF_UT = SGN*4.0*UT
387           CF_DO = 0.0
388     C
389           CF_HK = CF_UT*UT_HK + CF_DO*DO_HK
390           CF_TH = CF_UT*UT_TH + CF_DO*DO_TH
391           CF_RT = CF_UT*UT_RT + CF_DO*DO_RT
392           CF_MS = CF_UT*UT_MS + CF_DO*DO_MS
393     C
394     C
395     C---- set and linearize "slip" velocity UI = UI( DUO(DO UT TH RT MS) )
396           UI    = UO - DUO
397           UI_UT =    - DUO_UT
398           UI_DO =    - DUO_DO
399     C
400           UI_HK = UI_UT*UT_HK + UI_DO*DO_HK
401           UI_TH = UI_UT*UT_TH + UI_DO*DO_TH - DUO_TH
402           UI_RT = UI_UT*UT_RT + UI_DO*DO_RT - DUO_RT
403           UI_MS = UI_UT*UT_MS + UI_DO*DO_MS - DUO_MS
404     C
405           RETURN
406           END ! PRWALL

410           SUBROUTINE UWALL(TH,UO,DO,UI,RT,CF,BB, Y,U,N)
411     C------------------------------------------
412     C     Returns wall BL profile U(Y).
413     C
414     C     Input:
415     C        TH    kinematic momentum thickness
416     C        UO    uo/ue outer velocity  (= 1 for normal BL)
417     C        DO    BL thickness
418     C        UI    inner "slip" velocity
419     C        RT    momentum thickness based on ue and THETA
420     C        CF    wall skin friction
421     C        BB    outer profile exponent
422     C        N     number of profile array points
423     C
424     C     Output:
425     C        Y(i)  normal coordinate array
426     C        U(i)  u/ue velocity profile array
427     C-------------------------------------------
428     C
429           IMPLICIT REAL (A-H,M,O-Z)
430           DIMENSION Y(N), U(N)
431           DATA HPI / 1.570796327 /
432           DATA AK / 0.09 /
433     C
434     C---- log-law constants
435           DATA VKAP, VB / 0.40 , 5.0 /
436     C
437           EBK = EXP(-VB*VKAP)
438     C
439           SGN = SIGN( 1.0 , CF )
440           UT = SGN * SQRT(0.5*ABS(CF))
441     C
442     C
443     C---- set d+ = DP(UT DO ; RT TH)
444           DP    = SGN * UT*RT*DO/TH
445     C
446     C---- estimate inner profile edge velocity Ui+ using log-law
447           UPE = LOG(DP)/VKAP + VB
448     C
449     C---- converge exact Ui+ using Spalding formula
450           DO 10 ITUP=1, 5
451             UK = UPE*VKAP
452             ARG = UK - VB*VKAP
453             EXU = EXP(ARG)
454             REZ  = UPE +  EXU - EBK*(1.0 + UK + UK*UK/2.0 + UK*UK*UK/6.0)
455          &       - DP
456             DP_U = 1.0 + (EXU - EBK*(1.0 + UK + UK*UK/2.0))*VKAP
457     C
458             IF(ABS(REZ/DP) .LT. 1.0E-5) GO TO 11
459     C
460             DUPE = -REZ/DP_U
461             UPE = UPE + DUPE
462      10   CONTINUE
463           WRITE(*,*) 'UWALL: Ue+ convergence failed,  Res =', REZ/DP
464      11   CONTINUE
465     C
466     C          2      2        3      3
467     C---- set d y+/du+   and  d y+/du+   at BL edge
468           DP_UU  = (EXU - EBK*(1.0 + UK))*VKAP**2
469           DP_UUU = (EXU - EBK           )*VKAP**3
470     C
471     C                         2      2
472     C---- set  du+/dy+  and  d u+/dy+   at BL edge
473           UPD  = 1.0/DP_U
474           UPDD = (-1.0/DP_U**3) * DP_UU
475     C
476     C---- set coefficients for Spalding profile correction polynomial
477           DC2  = 0.5*DP*DP*UPDD    - DP*UPD
478           DC3  = -(  DP*DP*UPDD    - DP*UPD   ) / 3.0
479     C
480     C---- set outer profile amplitude DUO
481           DUO    = UO - UT*(UPE + DC2 + DC3)
482     C
483     C
484           BB1    =  3.0*(BB    +2.0)*(BB+3.0)/(BB+7.0)
485           BB2    = -5.0*(BB    +1.0)*(BB+3.0)/(BB+7.0)
486           BB3    =  2.0*(BB    +1.0)*(BB+2.0)/(BB+7.0)
487     C
488     c      NE = (N*9)/10
489           NE = N
490     C
491     C---- fill Y coordinate and U profile arrays
492     CCC   EXUPE = EXP(UPE*VKAP - VB*VKAP)
493           EXUPE = EXU
494     C
495           DEXU = (EXUPE - EBK)/FLOAT(NE-1)
496     C
497           I = 1
498           Y(I) = 0.0
499           U(I) = 0.0
500           DO 20 I=2, NE
501     ccc     EXU = EBK +  DEXU*FLOAT(I-1)
502             EXU = EBK + (DEXU - 0.75*DEXU*FLOAT(NE-I)/FLOAT(NE-1))
503          &              *FLOAT(I-1)
504     C
505     CCC     UK = UP*VKAP
506             UK = LOG(EXU) + VB*VKAP
507     C
508             UP = UK/VKAP
509     C
510     C------ set "inverse" Spalding profile  y+(u+)
511             YP    =  UP +  EXU - EBK*(1.0 + UK + UK*UK/2.0 + UK*UK*UK/6.0)
512             YP_UP = 1.0 + (EXU - EBK*(1.0 + UK + UK*UK/2.0))*VKAP
513     C
514             ET = YP/DP
515     C
516     C------ set final inner profile (fudged Spalding)
517             UIP = UP + DC2*ET**2 + DC3*ET**3
518     C
519     C------ set outer profile
520             SQE = SQRT(ET)
521             ETB = ET**BB
522     C
523     ccc        G = 2.0*ETB - ETB**2
524     C
525             G = (BB1   *ET + BB2   *ET**2 + BB3   *ET**3)*ETB
526     C
527             Y(I) = ET*DO
528             U(I) = UT*UIP  +  DUO*G
529     c
530       20  CONTINUE
531     C
532     c      DETA = 0.1 / FLOAT(N - NE - 1)
533     c      DO 300 I=NE+1, N
534     c        ETA = 1.0 + DETA*FLOAT(I-NE)
535     c        Y(I) = DO*ETA
536     c        U(I) = 1.0
537     c 300  CONTINUE
538     C
539           RETURN
540           END ! UWALL

544           SUBROUTINE FS(INORM,ISPEC,BSPEC,HSPEC,N,ETAE,GEO,ETA,F,U,S,DELTA)
545           IMPLICIT REAL (A-H,M,O-Z)
546           DIMENSION ETA(N), F(N), U(N), S(N)
547     C-----------------------------------------------------
548     C     Routine for solving the Falkner-Skan equation.
549     C
550     C     Input:
551     C     ------
552     C      INORM   1: eta = y / sqrt(vx/Ue)  "standard" Falkner-Skan coordinate
553     C              2: eta = y / sqrt(2vx/(m+1)Ue)  Hartree's coordinate
554     C              3: eta = y / Theta  momentum thickness normalized coordinate
555     C      ISPEC   1: BU  = x/Ue dUe/dx ( = "m")  specified
556     C              2: H12 = Dstar/Theta  specified
557     C      BSPEC   specified pressure gradient parameter  (if ISPEC = 1)
558     C      HSPEC   specified shape parameter of U profile (if ISPEC = 2)
559     C      N       total number of points in profiles
560     C      ETAE    edge value of normal coordinate
561     C      GEO     exponential stretching factor for ETA:
562     C
563     C     Output:
564     C     -------
565     C      BSPEC   calculated pressure gradient parameter  (if ISPEC = 2)
566     C      HSPEC   calculated shape parameter of U profile (if ISPEC = 1)
567     C      ETA     normal BL coordinate
568     C      F,U,S   Falkner Skan profiles
569     C      DELTA   normal coordinate scale  y = eta * Delta
570     C-----------------------------------------------------
571     C
572           PARAMETER (NMAX=257,NRMAX=3)
573           DIMENSION A(3,3,NMAX),B(3,3,NMAX),C(3,3,NMAX),
574          &          R(3,NRMAX,NMAX)
575     C
576     C---- set number of righthand sides.
577           DATA NRHS / 3 /
578     C
579           ITMAX = 20
580     C
581           IF(N.GT.NMAX) STOP 'FS: Array overflow.'
582     C
583           PI = 4.0*ATAN(1.0)
584     C
585     CCC      if(u(n) .ne. 0.0) go to 9991
586     
587     c
588     C---- initialize H or BetaU with empirical curve fits
589           IF(ISPEC.EQ.1) THEN
590            H = 2.6
591            BU = BSPEC
592           ELSE
593            H = HSPEC
594            IF(H .LE. 2.17) THEN
595             WRITE(*,*) 'FS: Specified H too low'
596             H = 2.17
597            ENDIF
598            BU = (0.058*(H-4.0)**2/(H-1.0) - 0.068) / (6.54*H - 14.07) * H**2
599            IF(H .GT. 4.0) BU = MIN( BU , 0.0 )
600           ENDIF
601     C
602     C---- initialize TN = Delta^2 Ue / vx
603           IF(INORM.EQ.3) THEN
604            TN = (6.54*H - 14.07) / H**2
605           ELSE
606            TN = 1.0
607           ENDIF
608     C
609     C---- set eta array
610           DETA = 1.0
611           ETA(1) = 0.0
612           DO 5 I=2, N
613             ETA(I) = ETA(I-1) + DETA
614             DETA = GEO*DETA
615         5 CONTINUE
616     C
617           DO 6 I=1, N
618             ETA(I) = ETA(I) * ETAE/ETA(N)
619         6 CONTINUE
620     C
621     C
622     C---- initial guess for profiles using a sine loop for U for half near wall
623           IF(H .LE. 3.0) THEN
624     C
625            IF(INORM.EQ.3) THEN
626             ETJOIN = 7.3
627            ELSE
628             ETJOIN = 5.0
629            ENDIF
630     C
631            EFAC = 0.5*PI/ETJOIN
632            DO 10 I=1, N
633              U(I) = SIN(EFAC*ETA(I))
634              F(I) = 1.0/EFAC * (1.0 - COS(EFAC*ETA(I)))
635              S(I) = EFAC*COS(EFAC*ETA(I))
636              IF(ETA(I) .GT. ETJOIN) GO TO 11
637       10   CONTINUE
638       11   CONTINUE
639            IJOIN = I
640     C
641     C----- constant U for outer half
642            DO 12 I=IJOIN+1, N
643              U(I) = 1.0
644              F(I) = F(IJOIN) + ETA(I) - ETA(IJOIN)
645              S(I) = 0.
646        12  CONTINUE
647     C
648           ELSE
649     C
650            IF(INORM.EQ.3) THEN
651             ETJOIN1 = 0.0
652             ETJOIN2 = 8.0
653             IF(H .GT. 4.0) THEN
654              ETJOIN1 = H - 4.0
655              ETJOIN2 = ETJOIN1 + 8.0
656             ENDIF
657            ELSE
658             ETJOIN1 = 0.0
659             ETJOIN2 = 8.0
660            ENDIF
661     C
662            DO 13 I=1, N
663              U(I) = 0.0
664              S(I) = 0.0
665              F(I) = 0.0
666              IF(ETA(I) .GE. ETJOIN1) GO TO 14
667      13    CONTINUE
668      14    CONTINUE
669            IJOIN = I
670     C
671            EFAC = 0.5*PI/(ETJOIN2-ETJOIN1)
672            DO 15 I=IJOIN+1, N
673              EBAR = ETA(I) - ETJOIN1
674              U(I) = 0.5 - 0.5*COS(2.0*EFAC*EBAR)
675              F(I) = 0.5*EBAR - 0.25/EFAC * SIN(2.0*EFAC*EBAR)
676              S(I) = EFAC*SIN(2.0*EFAC*EBAR)
677              IF(ETA(I) .GE. ETJOIN2) GO TO 16
678       15   CONTINUE
679       16   CONTINUE
680            IJOIN = I
681     C
682     C----- constant U for outer half
683            DO 17 I=IJOIN+1, N
684              U(I) = 1.0
685              F(I) = F(IJOIN) + ETA(I) - ETA(IJOIN)
686              S(I) = 0.
687        17  CONTINUE
688     C
689           ENDIF
690     c
691      9991 continue
692     C
693     C
694     C---- Newton iteration loop
695           DO 100 ITER=1, ITMAX
696     C
697     C------ zero out A,B,C blocks and righthand sides R
698             DO 20 I=1, N
699               DO 201 II=1,3
700                 DO 2001 III=1,3
701                   A(II,III,I) = 0.
702                   B(II,III,I) = 0.
703                   C(II,III,I) = 0.
704      2001       CONTINUE
705                 R(II,1,I) = 0.
706                 R(II,2,I) = 0.
707                 R(II,3,I) = 0.
708       201     CONTINUE
709        20   CONTINUE
710     C
711     C...................................................
712     C
713             A(1,1,1) = 1.0
714             A(2,2,1) = 1.0
715             A(3,2,N) = 1.0
716             R(1,1,1) = F(1)
717             R(2,1,1) = U(1)
718             R(3,1,N) = U(N) - 1.0
719     C
720             IF(INORM.EQ.2) THEN
721              BETU    = 2.0*BU/(BU+1.0)
722              BETU_BU = (2.0 - BETU/(BU+1.0))/(BU+1.0)
723              BETN    = 1.0
724              BETN_BU = 0.0
725             ELSE
726              BETU    = BU
727              BETU_BU = 1.0
728              BETN    = 0.5*(1.0 + BU)
729              BETN_BU = 0.5
730             ENDIF
731     C
732             DO 30 I=1,N-1
733     C
734               DETA = ETA(I+1) - ETA(I)
735               R(1,1,I+1) = F(I+1) - F(I) - 0.5*DETA*(U(I+1)+U(I))
736               R(2,1,I+1) = U(I+1) - U(I) - 0.5*DETA*(S(I+1)+S(I))
737               R(3,1,I)   = S(I+1) - S(I)
738          &      + TN * (  BETN*DETA*0.5*(F(I+1)*S(I+1) + F(I)*S(I))
739          &              + BETU*DETA*(1.0 - 0.5*(U(I+1)**2 + U(I)**2)) )
740     C
741               A(3,1,I) =  TN * BETN*0.5*DETA*S(I)
742               C(3,1,I) =  TN * BETN*0.5*DETA*S(I+1)
743               A(3,2,I) = -TN * BETU    *DETA*U(I)
744               C(3,2,I) = -TN * BETU    *DETA*U(I+1)
745               A(3,3,I) =  TN * BETN*0.5*DETA*F(I)   - 1.0
746               C(3,3,I) =  TN * BETN*0.5*DETA*F(I+1) + 1.0
747     C
748               B(1,1,I+1) = -1.0
749               A(1,1,I+1) =  1.0
750               B(1,2,I+1) = -0.5*DETA
751               A(1,2,I+1) = -0.5*DETA
752     C
753               B(2,2,I+1) = -1.0
754               A(2,2,I+1) =  1.0
755               B(2,3,I+1) = -0.5*DETA
756               A(2,3,I+1) = -0.5*DETA
757     C
758               R(3,2,I) = TN 
759          &          * ( BETN_BU*DETA*0.5*(F(I+1)*S(I+1) + F(I)*S(I))
760          &            + BETU_BU*DETA*(1.0 - 0.5*(U(I+1)**2 + U(I)**2)))
761               R(3,3,I) = ( BETN*DETA*0.5*(F(I+1)*S(I+1) + F(I)*S(I))
762          &               + BETU*DETA*(1.0 - 0.5*(U(I+1)**2 + U(I)**2)) )
763     C
764       30    CONTINUE
765     C...........................................................
766     C
767     C
768     C---- solve Newton system for the three solution vectors
769           CALL B3SOLV(A,B,C,R,N,NRHS,NRMAX)
770     C
771     C
772     C---- calculate and linearize Dstar, Theta, in computational space
773           DSI = 0.
774           DSI1 = 0.
775           DSI2 = 0.
776           DSI3 = 0.
777     C
778           THI = 0.
779           THI1 = 0.
780           THI2 = 0.
781           THI3 = 0.
782     C
783           DO 40 I=1,N-1
784             US  = U(I) + U(I+1)
785             DETA = ETA(I+1) - ETA(I)
786     C
787             DSI = DSI + (1.0 - 0.5*US)*DETA
788             DSI_US = -0.5*DETA
789     C
790             THI = THI + (1.0 - 0.5*US)*0.5*US*DETA
791             THI_US = (0.5 - 0.5*US)*DETA
792     C
793             DSI1 = DSI1 + DSI_US*(R(2,1,I) + R(2,1,I+1))
794             DSI2 = DSI2 + DSI_US*(R(2,2,I) + R(2,2,I+1))
795             DSI3 = DSI3 + DSI_US*(R(2,3,I) + R(2,3,I+1))
796     C
797             THI1 = THI1 + THI_US*(R(2,1,I) + R(2,1,I+1))
798             THI2 = THI2 + THI_US*(R(2,2,I) + R(2,2,I+1))
799             THI3 = THI3 + THI_US*(R(2,3,I) + R(2,3,I+1))
800       40  CONTINUE
801     C
802     C
803           IF(ISPEC.EQ.1) THEN
804     C
805     C----- set and linearize  Bu = Bspec  residual
806            R1 = BSPEC - BU
807            Q11 = 1.0
808            Q12 = 0.0
809     C
810           ELSE
811     C
812     C----- set and linearize  H = Hspec  residual
813            R1  =  DSI  - HSPEC*THI
814          &       -DSI1 + HSPEC*THI1
815            Q11 = -DSI2 + HSPEC*THI2
816            Q12 = -DSI3 + HSPEC*THI3
817     C
818           ENDIF
819     C
820     C
821           IF(INORM.EQ.3) THEN
822     C
823     C----- set and linearize  normalized Theta = 1  residual
824            R2  =  THI  - 1.0
825          &       -THI1
826            Q21 = -THI2
827            Q22 = -THI3
828     C
829           ELSE
830     C
831     C----- set eta scaling coefficient to unity
832            R2  =  1.0 - TN
833            Q21 = 0.0
834            Q22 = 1.0
835     C
836           ENDIF
837     C
838     C
839           DET =   Q11*Q22 - Q12*Q21
840           DBU = -(R1 *Q22 - Q12*R2 ) / DET
841           DTN = -(Q11*R2  - R1 *Q21) / DET
842     C
843     C
844     C---- calculate changes in F,U,S, and the max and rms change
845           RMAX = 0.
846           RMS = 0.
847           DO 50 I=1,N
848             DF = -R(1,1,I) - DBU*R(1,2,I) - DTN*R(1,3,I)
849             DU = -R(2,1,I) - DBU*R(2,2,I) - DTN*R(2,3,I)
850             DS = -R(3,1,I) - DBU*R(3,2,I) - DTN*R(3,3,I)
851     C
852             RMAX = MAX(RMAX,ABS(DF),ABS(DU),ABS(DS))
853             RMS = DF**2 + DU**2 + DS**2  +  RMS
854        50 CONTINUE
855           RMS = SQRT(RMS/(3.0*FLOAT(N) + 3.0))
856     C
857           RMAX = MAX(RMAX,ABS(DBU/0.5),ABS(DTN/TN))
858     C
859     C---- set underrelaxation factor if necessary by limiting max change to 0.5
860           RLX = 1.0
861           IF(RMAX.GT.0.5) RLX = 0.5/RMAX
862     C
863     C---- update F,U,S
864           DO 60 I=1,N
865             DF = -R(1,1,I) - DBU*R(1,2,I) - DTN*R(1,3,I)
866             DU = -R(2,1,I) - DBU*R(2,2,I) - DTN*R(2,3,I)
867             DS = -R(3,1,I) - DBU*R(3,2,I) - DTN*R(3,3,I)
868     C
869             F(I) = F(I) + RLX*DF
870             U(I) = U(I) + RLX*DU
871             S(I) = S(I) + RLX*DS
872        60 CONTINUE
873     C
874     C---- update BetaU and Theta
875           BU = BU + RLX*DBU
876           TN = TN + RLX*DTN
877     C    
878     C---- check for convergence
879           IF(ITER.GT.3 .AND. RMS.LT.1.E-5) GO TO 105
880     C
881       100 CONTINUE
882           WRITE(*,*) 'FS: Convergence failed'
883     C
884       105 CONTINUE
885     C
886           HSPEC = DSI/THI
887           BSPEC = BU
888     C
889           DELTA = SQRT(TN)
890     C
891           RETURN
892     C
893     C     The
894           END

897           SUBROUTINE B3SOLV(A,B,C,R,N,NRHS,NRMAX)
898           IMPLICIT REAL (A-H,M,O-Z)
899           DIMENSION A(3,3,N), B(3,3,N), C(3,3,N), R(3,NRMAX,N)
900     C     **********************************************************************
901     C      This routine solves a 3x3 block-tridiagonal system with an arbitrary
902     C      number of righthand sides by a standard block elimination scheme.  
903     C      The solutions are returned in the Rj vectors.
904     C
905     C      |A C      ||d|   |R..|
906     C      |B A C    ||d|   |R..|
907     C      |  B . .  ||.| = |R..|
908     C      |    . . C||.|   |R..|
909     C      |      B A||d|   |R..|
910     C                                                  Mark Drela   10 March 86
911     C     **********************************************************************
912     C
913     CCC** Forward sweep: Elimination of lower block diagonal (B's).
914           DO 1 I=1, N
915     C
916             IM = I-1
917     C
918     C------ don't eliminate B1 block because it doesn't exist
919             IF(I.EQ.1) GO TO 12
920     C
921     C------ eliminate Bi block, thus modifying Ai and Ci blocks
922             DO 11 K=1, 3
923               DO 111 L=1, 3
924                 A(K,L,I) = A(K,L,I)
925          & - (B(K,1,I)*C(1,L,IM) + B(K,2,I)*C(2,L,IM) + B(K,3,I)*C(3,L,IM))
926       111     CONTINUE
927               DO 112 L=1, NRHS
928                 R(K,L,I) = R(K,L,I)
929          & - (B(K,1,I)*R(1,L,IM) + B(K,2,I)*R(2,L,IM) + B(K,3,I)*R(3,L,IM))
930       112     CONTINUE
931        11   CONTINUE
932     C
933     C                                                              -1
934     CCC---- multiply Ci block and righthand side Ri vectors by (Ai)
935     C       using Gaussian elimination.
936     C
937        12   DO 13 KPIV=1, 2
938               KP1 = KPIV+1
939     C
940     C-------- find max pivot index KX
941               KX = KPIV
942               DO 131 K=KP1, 3
943                 IF(ABS(A(K,KPIV,I)) .LE. ABS(A(KX,KPIV,I))) THEN
944                  GO TO 131
945                 ELSE
946                  GO TO 1311
947                 ENDIF
948      1311       KX = K
949       131     CONTINUE
950     C
951               IF(A(KX,KPIV,I).EQ.0.0) THEN
952                WRITE(*,*) 'Singular A block, i = ',I
953                STOP
954               ENDIF
955     C
956               PIVOT = 1.0/A(KX,KPIV,I)
957     C
958     C-------- switch pivots
959               A(KX,KPIV,I) = A(KPIV,KPIV,I)
960     C
961     C-------- switch rows & normalize pivot row
962               DO 132 L=KP1, 3
963                 TEMP = A(KX,L,I)*PIVOT
964                 A(KX,L,I) = A(KPIV,L,I)
965                 A(KPIV,L,I) = TEMP
966       132     CONTINUE
967     C
968               DO 133 L=1, 3
969                 TEMP = C(KX,L,I)*PIVOT
970                 C(KX,L,I) = C(KPIV,L,I)
971                 C(KPIV,L,I) = TEMP
972       133     CONTINUE
973     C
974               DO 134 L=1, NRHS
975                 TEMP = R(KX,L,I)*PIVOT
976                 R(KX,L,I) = R(KPIV,L,I)
977                 R(KPIV,L,I) = TEMP
978       134     CONTINUE
979     CB
980     C-------- forward eliminate everything
981               DO 135 K=KP1, 3
982                 DO 1351 L=KP1, 3
983                   A(K,L,I) = A(K,L,I) - A(K,KPIV,I)*A(KPIV,L,I)
984      1351       CONTINUE
985                 C(K,1,I) = C(K,1,I) - A(K,KPIV,I)*C(KPIV,1,I)
986                 C(K,2,I) = C(K,2,I) - A(K,KPIV,I)*C(KPIV,2,I)
987                 C(K,3,I) = C(K,3,I) - A(K,KPIV,I)*C(KPIV,3,I)
988                 DO 1352 L=1, NRHS
989                   R(K,L,I) = R(K,L,I) - A(K,KPIV,I)*R(KPIV,L,I)
990      1352       CONTINUE
991       135     CONTINUE
992     C
993        13   CONTINUE
994     C
995     C------ solve for last row
996             IF(A(3,3,I).EQ.0.0) THEN
997              WRITE(*,*) 'Singular A block, i = ',I
998              STOP
999             ENDIF
1000             PIVOT = 1.0/A(3,3,I)
1001             C(3,1,I) = C(3,1,I)*PIVOT
1002             C(3,2,I) = C(3,2,I)*PIVOT
1003             C(3,3,I) = C(3,3,I)*PIVOT
1004             DO 14 L=1, NRHS
1005               R(3,L,I) = R(3,L,I)*PIVOT
1006        14   CONTINUE
1007     C
1008     C------ back substitute everything
1009             DO 15 KPIV=2, 1, -1
1010               KP1 = KPIV+1
1011               DO 151 K=KP1, 3
1012                 C(KPIV,1,I) = C(KPIV,1,I) - A(KPIV,K,I)*C(K,1,I)
1013                 C(KPIV,2,I) = C(KPIV,2,I) - A(KPIV,K,I)*C(K,2,I)
1014                 C(KPIV,3,I) = C(KPIV,3,I) - A(KPIV,K,I)*C(K,3,I)
1015                 DO 1511 L=1, NRHS
1016                   R(KPIV,L,I) = R(KPIV,L,I) - A(KPIV,K,I)*R(K,L,I)
1017      1511       CONTINUE
1018       151     CONTINUE
1019        15   CONTINUE
1020         1 CONTINUE
1021     C
1022     CCC** Backward sweep: Back substitution using upper block diagonal (Ci's).
1023           DO 2 I=N-1, 1, -1
1024             IP = I+1
1025             DO 21 L=1, NRHS
1026               DO 211 K=1, 3
1027                 R(K,L,I) = R(K,L,I)
1028          & - (R(1,L,IP)*C(K,1,I) + R(2,L,IP)*C(K,2,I) + R(3,L,IP)*C(K,3,I))
1029       211     CONTINUE
1030        21   CONTINUE
1031         2 CONTINUE
1032     C
1033           RETURN
1034           END ! B3SOLV
1035
