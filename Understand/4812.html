1812           SUBROUTINE NAMMOD(NAME,KDEL,KMOD0)
1813           CHARACTER*(*) NAME
1814     C-------------------------------------------
1815     C     Requests new modified NAME with 
1816     C     version number in brackets, e.g.
1817     C            NACA 0012  [5]
1818     C
1819     C     If bracketed index exists in NAME,
1820     C        it is incremented by KDEL.
1821     C     If no bracketed index exists, it 
1822     C        is added with initial value KMOD0,
1823     C        unless KMOD0 is negative in which 
1824     C        case nothing is added.
1825     C-------------------------------------------
1826           CHARACTER*48 NAMDEF
1827     C
1828           CALL STRIP(NAME,NNAME)
1829           KBRACK1 = INDEX(NAME,'[')
1830           KBRACK2 = INDEX(NAME,']')
1831     C
1832           NAMDEF = NAME(1:NNAME)
1833     C
1834           IF(KBRACK1.NE.0 .AND. 
1835          &   KBRACK2.NE.0 .AND. KBRACK2-KBRACK1.GT.1) THEN
1836     C----- brackets exist... get number, (go get user's input on READ error)
1837            READ(NAME(KBRACK1+1:KBRACK2-1),*,ERR=40) KMOD
1838            KMOD = IABS(KMOD)
1839            KMODP = MOD( KMOD+KDEL , 100 )
1840            IF(KBRACK1.GE.2) THEN
1841             NAME = NAME(1:KBRACK1-1)
1842            ELSE
1843             NAME = ' '
1844            ENDIF
1845            CALL STRIP(NAME,NNAME)
1846           ELSEIF(KMOD0.GT.0) THEN
1847            KMODP = MOD( KMOD0 , 100 )
1848           ELSE
1849            KMODP = 0
1850           ENDIF
1851     C
1852           IF    (KMODP.GE.10) THEN
1853            NAMDEF = NAME(1:NNAME) // ' [  ]'
1854            WRITE(NAMDEF(NNAME+3:NNAME+4),1020) KMODP
1855      1020  FORMAT(I2)
1856           ELSEIF(KMODP.GE. 1) THEN
1857            NAMDEF = NAME(1:NNAME) // ' [ ]'
1858            WRITE(NAMDEF(NNAME+3:NNAME+3),1025) KMODP
1859      1025  FORMAT(I1)
1860           ENDIF
1861     C
1862      40   WRITE(*,1040) NAMDEF
1863      1040 FORMAT(/' Enter airfoil name or <return> for default:  ',A)
1864           READ(*,1000) NAME
1865      1000 FORMAT(A)
1866           IF(NAME .EQ. ' ') NAME = NAMDEF
1867     C
1868           RETURN
1869           END ! NAMMOD

2109           SUBROUTINE MHINGE
2110     C----------------------------------------------------
2111     C     Calculates the hinge moment of the flap about
2112     C     (XOF,YOF) by integrating surface pressures.
2113     C----------------------------------------------------
2114           INCLUDE 'XFOIL.INC'
2115     C
2116           IF(.NOT.LFLAP) THEN
2117     C
2118             CALL GETXYF(X,XP,Y,YP,S,N, TOPS,BOTS,XOF,YOF)
2119             LFLAP = .TRUE.
2120     C
2121           ELSE
2122     C
2123     C------ find top and bottom y at hinge x location
2124             TOPS = XOF
2125             BOTS = S(N) - XOF
2126             CALL SINVRT(TOPS,XOF,X,XP,S,N)      
2127             CALL SINVRT(BOTS,XOF,X,XP,S,N)      
2128     C
2129           ENDIF
2130     C
2131           TOPX = SEVAL(TOPS,X,XP,S,N)
2132           TOPY = SEVAL(TOPS,Y,YP,S,N)
2133           BOTX = SEVAL(BOTS,X,XP,S,N)
2134           BOTY = SEVAL(BOTS,Y,YP,S,N)
2135     C
2136     C
2137           HMOM = 0.
2138           HFX  = 0.
2139           HFY  = 0.
2140     C
2141     C---- integrate pressures on top and bottom sides of flap
2142           DO 20 I=2, N
2143             IF(S(I-1).GE.TOPS .AND. S(I).LE.BOTS) GO TO 20
2144     C
2145              DX = X(I) - X(I-1)
2146              DY = Y(I) - Y(I-1)
2147              XMID = 0.5*(X(I)+X(I-1)) - XOF
2148              YMID = 0.5*(Y(I)+Y(I-1)) - YOF
2149              IF(LVISC) THEN
2150               PMID = 0.5*(CPV(I) + CPV(I-1))
2151              ELSE
2152               PMID = 0.5*(CPI(I) + CPI(I-1))
2153              ENDIF
2154              HMOM = HMOM + PMID*(XMID*DX + YMID*DY)
2155              HFX  = HFX  - PMID* DY
2156              HFY  = HFY  + PMID* DX
2157        20 CONTINUE
2158     C
2159     C---- find S(I)..S(I-1) interval containing s=TOPS
2160           DO I=2, N
2161             IF(S(I).GT.TOPS) GO TO 31
2162           ENDDO
2163     C
2164        31 CONTINUE
2165     C---- add on top surface chunk TOPS..S(I-1),  missed in the DO 20 loop.
2166           DX = TOPX - X(I-1)
2167           DY = TOPY - Y(I-1)
2168           XMID = 0.5*(TOPX+X(I-1)) - XOF
2169           YMID = 0.5*(TOPY+Y(I-1)) - YOF
2170           IF(S(I) .NE. S(I-1)) THEN
2171            FRAC = (TOPS-S(I-1))/(S(I)-S(I-1))
2172           ELSE
2173            FRAC = 0.
2174           ENDIF
2175           IF(LVISC) THEN
2176            TOPP = CPV(I)*FRAC + CPV(I-1)*(1.0-FRAC)
2177            PMID = 0.5*(TOPP+CPV(I-1))
2178           ELSE
2179            TOPP = CPI(I)*FRAC + CPI(I-1)*(1.0-FRAC)
2180            PMID = 0.5*(TOPP+CPI(I-1))
2181           ENDIF
2182           HMOM = HMOM + PMID*(XMID*DX + YMID*DY)
2183           HFX  = HFX  - PMID* DY
2184           HFY  = HFY  + PMID* DX
2185     C
2186     C---- add on inside flap surface contribution from hinge to top surface
2187           DX = XOF - TOPX
2188           DY = YOF - TOPY
2189           XMID = 0.5*(TOPX+XOF) - XOF
2190           YMID = 0.5*(TOPY+YOF) - YOF
2191           HMOM = HMOM + PMID*(XMID*DX + YMID*DY)
2192           HFX  = HFX  - PMID* DY
2193           HFY  = HFY  + PMID* DX
2194     C
2195     C---- find S(I)..S(I-1) interval containing s=BOTS
2196           DO I=N, 2, -1
2197             IF(S(I-1).LT.BOTS) GO TO 41
2198           ENDDO
2199     C
2200        41 CONTINUE
2201     C---- add on bottom surface chunk BOTS..S(I),  missed in the DO 20 loop.
2202           DX = X(I) - BOTX
2203           DY = Y(I) - BOTY
2204           XMID = 0.5*(BOTX+X(I)) - XOF
2205           YMID = 0.5*(BOTY+Y(I)) - YOF
2206           IF(S(I) .NE. S(I-1)) THEN
2207            FRAC = (BOTS-S(I-1))/(S(I)-S(I-1))
2208           ELSE
2209            FRAC = 0.
2210           ENDIF
2211           IF(LVISC) THEN
2212            BOTP = CPV(I)*FRAC + CPV(I-1)*(1.0-FRAC)
2213            PMID = 0.5*(BOTP+CPV(I))
2214           ELSE
2215            BOTP = CPI(I)*FRAC + CPI(I-1)*(1.0-FRAC)
2216            PMID = 0.5*(BOTP+CPI(I))
2217           ENDIF
2218           HMOM = HMOM + PMID*(XMID*DX + YMID*DY)
2219           HFX  = HFX  - PMID* DY
2220           HFY  = HFY  + PMID* DX
2221     C
2222     C---- add on inside flap surface contribution from hinge to bottom surface
2223           DX = BOTX - XOF
2224           DY = BOTY - YOF
2225           XMID = 0.5*(BOTX+XOF) - XOF
2226           YMID = 0.5*(BOTY+YOF) - YOF
2227           HMOM = HMOM + PMID*(XMID*DX + YMID*DY)
2228           HFX  = HFX  - PMID* DY
2229           HFY  = HFY  + PMID* DX
2230     C
2231     C---- add on TE base thickness contribution
2232           DX = X(1) - X(N)
2233           DY = Y(1) - Y(N)
2234           XMID = 0.5*(X(1)+X(N)) - XOF
2235           YMID = 0.5*(Y(1)+Y(N)) - YOF
2236           IF(LVISC) THEN
2237            PMID = 0.5*(CPV(1)+CPV(N))
2238           ELSE
2239            PMID = 0.5*(CPI(1)+CPI(N))
2240           ENDIF
2241           HMOM = HMOM + PMID*(XMID*DX + YMID*DY)
2242           HFX  = HFX  - PMID* DY
2243           HFY  = HFY  + PMID* DX
2244     C
2245           RETURN
2246           END ! MHINGE

2429           SUBROUTINE SPECAL
2430     C-----------------------------------
2431     C     Converges to specified alpha.
2432     C-----------------------------------
2433           INCLUDE 'XFOIL.INC'
2434           REAL MINF_CLM, MSQ_CLM
2435     C
2436     C---- calculate surface vorticity distributions for alpha = 0, 90 degrees
2437           IF(.NOT.LGAMU .OR. .NOT.LQAIJ) CALL GGCALC
2438     C
2439           COSA = COS(ALFA)
2440           SINA = SIN(ALFA)
2441     C
2442     C---- superimpose suitably weighted  alpha = 0, 90  distributions
2443           DO 50 I=1, N
2444             GAM(I)   =  COSA*GAMU(I,1) + SINA*GAMU(I,2)
2445             GAM_A(I) = -SINA*GAMU(I,1) + COSA*GAMU(I,2)
2446        50 CONTINUE
2447           PSIO = COSA*GAMU(N+1,1) + SINA*GAMU(N+1,2)
2448     C
2449           CALL TECALC
2450           CALL QISET
2451     C
2452     C---- set initial guess for the Newton variable CLM
2453           CLM = 1.0
2454     C
2455     C---- set corresponding  M(CLM), Re(CLM)
2456           CALL MRCL(CLM,MINF_CLM,REINF_CLM)
2457           CALL COMSET
2458     C
2459     C---- set corresponding CL(M)
2460           CALL CLCALC(N,X,Y,GAM,GAM_A,ALFA,MINF,QINF, XCMREF,YCMREF,
2461          &            CL,CM,CDP, CL_ALF,CL_MSQ)
2462     C
2463     C---- iterate on CLM
2464           DO 100 ITCL=1, 20
2465     C
2466             MSQ_CLM = 2.0*MINF*MINF_CLM
2467             DCLM = (CL - CLM)/(1.0 - CL_MSQ*MSQ_CLM)
2468     C
2469             CLM1 = CLM
2470             RLX = 1.0
2471     C
2472     C------ under-relaxation loop to avoid driving M(CL) above 1
2473             DO 90 IRLX=1, 12
2474     C
2475               CLM = CLM1 + RLX*DCLM
2476     C
2477     C-------- set new freestream Mach M(CLM)
2478               CALL MRCL(CLM,MINF_CLM,REINF_CLM)
2479     C
2480     C-------- if Mach is OK, go do next Newton iteration
2481               IF(MATYP.EQ.1 .OR. MINF.EQ.0.0 .OR. MINF_CLM.NE.0.0) GO TO 91
2482     C
2483               RLX = 0.5*RLX
2484        90   CONTINUE
2485        91   CONTINUE
2486     C
2487     C------ set new CL(M)
2488             CALL COMSET
2489             CALL CLCALC(N,X,Y,GAM,GAM_A,ALFA,MINF,QINF, XCMREF,YCMREF,
2490          &              CL,CM,CDP,CL_ALF,CL_MSQ)
2491     C
2492             IF(ABS(DCLM).LE.1.0E-6) GO TO 110
2493     C
2494       100 CONTINUE
2495           WRITE(*,*) 'SPECAL:  Minf convergence failed'
2496       110 CONTINUE

2498     C---- set final Mach, CL, Cp distributions, and hinge moment
2499           CALL MRCL(CL,MINF_CL,REINF_CL)
2500           CALL COMSET
2501           CALL CLCALC(N,X,Y,GAM,GAM_A,ALFA,MINF,QINF, XCMREF,YCMREF,
2502          &            CL,CM,CDP, CL_ALF,CL_MSQ)
2503           CALL CPCALC(N,QINV,QINF,MINF,CPI)
2504           IF(LVISC) THEN
2505            CALL CPCALC(N+NW,QVIS,QINF,MINF,CPV)
2506            CALL CPCALC(N+NW,QINV,QINF,MINF,CPI)
2507           ELSE
2508            CALL CPCALC(N,QINV,QINF,MINF,CPI)
2509           ENDIF
2510           IF(LFLAP) CALL MHINGE

2516           SUBROUTINE SPECCL
2517     C-----------------------------------------
2518     C     Converges to specified inviscid CL.
2519     C-----------------------------------------
2520           INCLUDE 'XFOIL.INC'
2521     C
2522     C---- calculate surface vorticity distributions for alpha = 0, 90 degrees
2523           IF(.NOT.LGAMU .OR. .NOT.LQAIJ) CALL GGCALC
2524     C
2525     C---- set freestream Mach from specified CL -- Mach will be held fixed
2526           CALL MRCL(CLSPEC,MINF_CL,REINF_CL)
2527           CALL COMSET
2528     C
2529     C---- current alpha is the initial guess for Newton variable ALFA
2530           COSA = COS(ALFA)
2531           SINA = SIN(ALFA)
2532           DO 10 I=1, N
2533             GAM(I)   =  COSA*GAMU(I,1) + SINA*GAMU(I,2)
2534             GAM_A(I) = -SINA*GAMU(I,1) + COSA*GAMU(I,2)
2535        10 CONTINUE
2536           PSIO = COSA*GAMU(N+1,1) + SINA*GAMU(N+1,2)
2537     C
2538     C---- get corresponding CL, CL_alpha, CL_Mach
2539           CALL CLCALC(N,X,Y,GAM,GAM_A,ALFA,MINF,QINF, XCMREF,YCMREF,
2540          &            CL,CM,CDP, CL_ALF,CL_MSQ)
2541     C
2542     C---- Newton loop for alpha to get specified inviscid CL
2543           DO 100 ITAL=1, 20
2544     C
2545             DALFA = (CLSPEC - CL) / CL_ALF
2546             RLX = 1.0
2547     C
2548             ALFA = ALFA + RLX*DALFA
2549     C
2550     C------ set new surface speed distribution
2551             COSA = COS(ALFA)
2552             SINA = SIN(ALFA)
2553             DO 40 I=1, N
2554               GAM(I)   =  COSA*GAMU(I,1) + SINA*GAMU(I,2)
2555               GAM_A(I) = -SINA*GAMU(I,1) + COSA*GAMU(I,2)
2556        40   CONTINUE
2557             PSIO = COSA*GAMU(N+1,1) + SINA*GAMU(N+1,2)
2558     C
2559     C------ set new CL(alpha)
2560             CALL CLCALC(N,X,Y,GAM,GAM_A,ALFA,MINF,QINF, XCMREF,YCMREF,
2561          &              CL,CM,CDP,CL_ALF,CL_MSQ)
2562     C
2563             IF(ABS(DALFA).LE.1.0E-6) GO TO 110
2564       100 CONTINUE
2565           WRITE(*,*) 'SPECCL:  CL convergence failed'
2566       110 CONTINUE
2567     C
2568     C---- set final surface speed and Cp distributions
2569           CALL TECALC
2570           CALL QISET
2571           IF(LVISC) THEN
2572            CALL CPCALC(N+NW,QVIS,QINF,MINF,CPV)
2573            CALL CPCALC(N+NW,QINV,QINF,MINF,CPI)
2574           ELSE
2575            CALL CPCALC(N,QINV,QINF,MINF,CPI)
2576           ENDIF
2577           IF(LFLAP) CALL MHINGE
2578     C
2579           RETURN
2580           END ! SPECCL

2583           SUBROUTINE VISCAL(NITER1)
2584     C----------------------------------------
2585     C     Converges viscous operating point
2586     C----------------------------------------
2587           INCLUDE 'XFOIL.INC'
2588     C
2589     C---- convergence tolerance
2590           DATA EPS1 / 1.0E-4 /
2591     C
2592           NITER = NITER1
2593     C
2594     C---- calculate wake trajectory from current inviscid solution if necessary
2595           IF(.NOT.LWAKE) THEN
2596            CALL XYWAKE
2597           ENDIF
2598     C
2599     C---- set velocities on wake from airfoil vorticity for alpha=0, 90
2600           CALL QWCALC
2601     C
2602     C---- set velocities on airfoil and wake for initial alpha
2603           CALL QISET
2604     C
2605           IF(.NOT.LIPAN) THEN
2606     C
2607            IF(LBLINI) CALL GAMQV
2608     C
2609     C----- locate stagnation point arc length position and panel index
2610            CALL STFIND
2611     C
2612     C----- set  BL position -> panel position  pointers
2613            CALL IBLPAN
2614     C
2615     C----- calculate surface arc length array for current stagnation point location
2616            CALL XICALC
2617     C
2618     C----- set  BL position -> system line  pointers
2619            CALL IBLSYS
2620     C
2621           ENDIF
2622     C
2623     C---- set inviscid BL edge velocity UINV from QINV
2624           CALL UICALC
2625     C
2626           IF(.NOT.LBLINI) THEN
2627     C
2628     C----- set initial Ue from inviscid Ue
2629            DO IBL=1, NBL(1)
2630              UEDG(IBL,1) = UINV(IBL,1)
2631            ENDDO
2632     C
2633            DO IBL=1, NBL(2)
2634              UEDG(IBL,2) = UINV(IBL,2)
2635            ENDDO
2636     C
2637           ENDIF
2638     C
2639           IF(LVCONV) THEN
2640     C----- set correct CL if converged point exists
2641            CALL QVFUE
2642            IF(LVISC) THEN
2643             CALL CPCALC(N+NW,QVIS,QINF,MINF,CPV)
2644             CALL CPCALC(N+NW,QINV,QINF,MINF,CPI)
2645            ELSE
2646             CALL CPCALC(N,QINV,QINF,MINF,CPI)
2647            ENDIF
2648            CALL GAMQV
2649            CALL CLCALC(N,X,Y,GAM,GAM_A,ALFA,MINF,QINF, XCMREF,YCMREF,
2650          &             CL,CM,CDP, CL_ALF,CL_MSQ)
2651            CALL CDCALC
2652           ENDIF
2653     C
2654     C---- set up source influence matrix if it doesn't exist
2655           IF(.NOT.LWDIJ .OR. .NOT.LADIJ) CALL QDCALC
2656     C
2657     C---- Newton iteration for entire BL solution
2658           IF(NITER.EQ.0) CALL ASKI('Enter number of iterations^',NITER)
2659           WRITE(*,*)
2660           WRITE(*,*) 'Solving BL system ...'
2661           DO 1000 ITER=1, NITER
2662     C
2663     C------ fill Newton system for BL variables
2664             CALL SETBL
2665     C
2666     C------ solve Newton system with custom solver
2667             CALL BLSOLV
2668     C
2669     C------ update BL variables
2670             CALL UPDATE
2671     C
2672             IF(LALFA) THEN
2673     C------- set new freestream Mach, Re from new CL
2674              CALL MRCL(CL,MINF_CL,REINF_CL)
2675              CALL COMSET
2676             ELSE
2677     C------- set new inviscid speeds QINV and UINV for new alpha
2678              CALL QISET
2679              CALL UICALC
2680             ENDIF
2681     C
2682     C------ calculate edge velocities QVIS(.) from UEDG(..)
2683             CALL QVFUE
2684     C
2685     C------ set GAM distribution from QVIS
2686             CALL GAMQV
2687     C
2688     C------ relocate stagnation point
2689             CALL STMOVE
2690     C
2691     C------ set updated CL,CD
2692             CALL CLCALC(N,X,Y,GAM,GAM_A,ALFA,MINF,QINF, XCMREF,YCMREF,
2693          &              CL,CM,CDP,CL_ALF,CL_MSQ)
2694             CALL CDCALC
2695     C
2696     C------ display changes and test for convergence
2697             IF(RLX.LT.1.0) 
2698          &   WRITE(*,2000) ITER, RMSBL, RMXBL, VMXBL,IMXBL,ISMXBL,RLX
2699             IF(RLX.EQ.1.0) 
2700          &   WRITE(*,2010) ITER, RMSBL, RMXBL, VMXBL,IMXBL,ISMXBL
2701              CDP = CD - CDF
2702              WRITE(*,2020) ALFA/DTOR, CL, CM, CD, CDF, CDP
2703     C
2704             IF(RMSBL .LT. EPS1) THEN
2705              LVCONV = .TRUE.
2706              AVISC = ALFA
2707              MVISC = MINF
2708              GO TO 90
2709             ENDIF
2710     C
2711      1000 CONTINUE
2712           WRITE(*,*) 'VISCAL:  Convergence failed'
2713     C
2714        90 CONTINUE
2715           CALL CPCALC(N+NW,QINV,QINF,MINF,CPI)
2716           CALL CPCALC(N+NW,QVIS,QINF,MINF,CPV)
2717           IF(LFLAP) CALL MHINGE
2718           RETURN
