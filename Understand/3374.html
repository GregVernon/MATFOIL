21           SUBROUTINE POLREAD(LU,FNPOL,ERROR,
22          &                   NAX,NA,CPOL, 
23          &                   REYN1,MACH1,ACRIT,XTRIP,
24          &                   PTRAT,ETAP,
25          &                   NAME, IRETYP,IMATYP,
26          &                   ISX,NBL,CPOLSD,
27          &                   CODE,VERSION )
28           INCLUDE 'PINDEX.INC'
29           CHARACTER*(*) FNPOL, NAME
30           LOGICAL ERROR
31           CHARACTER*(*) CODE
32           REAL CPOL(NAX,IPTOT), CPOLSD(NAX,ISX,JPTOT)
33           REAL MACH1, XTRIP(ISX)
34     C--------------------------------------------------------
35     C     Reads in polar save file
36     C
37     C  Input:
38     C     LU      logical unit to use for reading
39     C     FNPOL   name of polar file to be read,
40     C               if FNPOL(1:1).eq.' ', unit LU will be read
41     C               if it is already open
42     C     NAX     polar point array dimension
43     C     ISX     airfoil side array dimension
44     C
45     C  Output:
46     C     ERROR   T if a READ error occurred
47     C     NA      number polar points
48     C     CPOL    polar coefficients and parameters
49     C     REYN1   Reynolds number for CL=1
50     C     MACH1   Mach number for CL=1
51     C     ACRIT   Critical amplification ratio
52     C     XTRIP   Trip locations
53     C     PTRAT   Actuator disk total-pressure ratio
54     C     ETAP    Actuator disk thermal efficiency
55     C     NAME    airfoil name string
56     C     IRETYP  flag giving type of Re variation with CL
57     C     IMATYP  flag giving type of Ma variation with CL
58     C     NBL     number of airfoil elements
59     C     CPOLSD  airfoil side-related parameters
60     C     CODE    code used to compute polar
61     C     VERSION code version
62     C--------------------------------------------------------
63           CHARACTER*128 LINE
64           CHARACTER*1 DUMMY
65           REAL RINP(0:IPTOT+2*JPTOT)
66     C
67           INTEGER IPOL(IPTOT), ISPOL(2,JPTOT)
68           INTEGER ITMP(IPTOT+2*JPTOT), ITMP0(IPTOT+2*JPTOT)
69           LOGICAL LOPEN, LHEAD, LDLAB
70           LOGICAL LIRE, LIMA, LINC, LJTP
71           CHARACTER*20 CPNAME
72     C
73     C
74           ERROR = .FALSE.
75           LHEAD = .TRUE.
76     C
77           NA = 0
78           NBL = 1
79     C
80     c      KCH = 0
81     c      KMC = 0
82     C
83           NIPOL = 0
84           DO IP = 1, IPTOT
85             IPOL(IP) = 0
86           ENDDO
87           DO JP = 1, JPTOT
88             ISPOL(1,JP) = 0
89             ISPOL(2,JP) = 0
90           ENDDO
91     C
92     C---- assume Re,Mach will not be given in header
93           IRETYP = 0
94           IMATYP = 0
95     C
96     C---- do we have to open the file?
97           LOPEN = FNPOL .NE. ' '
98     C
99           IF(LOPEN) OPEN(LU,FILE=FNPOL,STATUS='OLD',ERR=90)
100     C
101     C=============================================================
102     C---- start data reading loop
103      500  CONTINUE
104           READ(LU,1000,END=80) LINE
105           IF(LINE.EQ.' ') GO TO 500
106     C
107           IF(LHEAD) THEN
108     C----- parse to get header info
109     C
110     C----- assume this will be the data-label line
111            LDLAB = .TRUE.
112     C
113     C--------------------------------------------
114            K = INDEX(LINE,'Version')
115            IF(K.NE.0) THEN
116     C------ code,version line
117             DO K1=1, 128
118               IF(LINE(K1:K1).NE.' ') GO TO 10
119             ENDDO
120     C
121      10     CONTINUE
122             IF(K.GT.K1) THEN
123              CODE = LINE(K1:K-1)
124              READ(LINE(K+7:128),*,ERR=11) VERSION
125             ENDIF
126      11     CONTINUE
127             LDLAB = .FALSE.
128            ENDIF
129     C
130     C--------------------------------------------
131            KF = INDEX(LINE,'for:')
132            IF(KF.NE.0) THEN
133     C------ airfoil name line
134             NAME = LINE(KF+5:128)
135             LDLAB = .FALSE.
136            ENDIF
137     C
138     C--------------------------------------------
139            KE = INDEX(LINE,'elements')
140            IF(KE.GT.0) THEN
141     C------ element-number line
142             READ(LINE(KE-4:KE-1),*,ERR=60) NBL
143     C------ truncate name line to eliminate elements #
144             NAME = LINE(KF+5:KE-4)
145     C
146             IF(2*NBL .GT. ISX) THEN
147              NBL = ISX/2
148              WRITE(*,*) 
149          &   'POLREAD: Number of elements set to array limit', NBL
150             ENDIF
151             LDLAB = .FALSE.
152            ENDIF
153     C
154     C--------------------------------------------
155            KR = INDEX(LINE,'Reynolds number')
156            KM = INDEX(LINE,'Mach number')
157     C
158            IF(KR.NE.0) THEN
159     C------ Re-type line
160             IF(KM.GT.KR) THEN
161              KEND = KM-1
162             ELSE
163              KEND = 128
164             ENDIF
165             IF    (INDEX(LINE(KR:KEND),'fixed').NE.0) THEN
166              IRETYP = 1
167             ELSEIF(INDEX(LINE(KR:KEND),'1/sqrt(CL)').NE.0) THEN
168              IRETYP = 2
169             ELSEIF(INDEX(LINE(KR:KEND),'1/CL').NE.0) THEN
170              IRETYP = 3
171             ENDIF
172             LDLAB = .FALSE.
173            ENDIF
174     C
175            IF(KM.NE.0) THEN
176     C------ Ma-type line
177             IF(KR.GT.KM) THEN
178              KEND = KR-1
179             ELSE
180              KEND = 128
181             ENDIF
182             IF    (INDEX(LINE(KM:KEND),'fixed').NE.0) THEN
183              IMATYP = 1
184             ELSEIF(INDEX(LINE(KM:KEND),'1/sqrt(CL)').NE.0) THEN
185              IMATYP = 2
186             ELSEIF(INDEX(LINE(KM:KEND),'1/CL').NE.0) THEN
187              IMATYP = 3
188             ENDIF
189             LDLAB = .FALSE.
190            ENDIF
191     C
192     C--------------------------------------------
193     C---- find specified BL trip location
194            K = INDEX(LINE,'xtrf')
195            IF(K.NE.0) THEN
196     C------ new style xtrip line
197             KT = INDEX(LINE,'(top)')
198             KB = INDEX(LINE,'(bottom)')
199             KE = INDEX(LINE,'element ')
200     C--- check for old style trip line 
201             KS = INDEX(LINE,'(suc')
202             KP = INDEX(LINE,'(pre')
203     C
204             IF(KE.NE.0) THEN
205              READ(LINE(KE+7:KE+12),*,ERR=21) N
206             ELSE
207              N = 1
208             ENDIF
209             IF(N.LE.NBL) THEN
210              IS1 = 2*N-1
211              IS2 = 2*N
212              XTRIP(IS1) = 1.0
213              XTRIP(IS2) = 1.0
214              IF(KT.GT.0)  READ(LINE(K+6:KT-1) ,*,ERR=21) XTRIP(IS1)
215              IF(KB.GT.KT) READ(LINE(KT+5:KB-1),*,ERR=21) XTRIP(IS2)
216              IF(KS.GT.0)  READ(LINE(K+6:KS-1) ,*,ERR=21) XTRIP(IS1)
217              IF(KP.GT.KS) READ(LINE(KS+5:KP-1),*,ERR=21) XTRIP(IS2)
218             ENDIF
219      21     CONTINUE
220             LDLAB = .FALSE.
221            ENDIF
222     C
223     C--------------------------------------------
224            K = INDEX(LINE,'Mach =')
225            IF(K.NE.0) THEN
226             READ(LINE(K+6:128),*,ERR=31) MACH1
227      31     CONTINUE
228             LDLAB = .FALSE.
229            ENDIF
230     C
231     C--------------------------------------------
232            K = INDEX(LINE,'Re =')
233            IF(K.NE.0) THEN
234             READ(LINE(K+4:128),*,ERR=32) REYN1
235             REYN1 = REYN1 * 1.0E6
236      32     CONTINUE
237             LDLAB = .FALSE.
238            ENDIF
239     C
240     C--------------------------------------------
241            K = INDEX(LINE,'Ncrit =')
242            IF(K.NE.0) THEN
243             READ(LINE(K+7:128),*,ERR=33) ACRIT
244      33     CONTINUE
245             LDLAB = .FALSE.
246            ENDIF
247     C
248     C--------------------------------------------
249            K = INDEX(LINE,'pi_p =')
250            IF(K.NE.0) THEN
251             READ(LINE(K+6:128),*,ERR=34) PTRAT
252      34     CONTINUE
253             LDLAB = .FALSE.
254            ENDIF
255     C
256     C--------------------------------------------
257            K = INDEX(LINE,'eta_p =')
258            IF(K.NE.0) THEN
259             READ(LINE(K+7:128),*,ERR=35) ETAP
260      35     CONTINUE
261             LDLAB = .FALSE.
262            ENDIF
263     C
264     C--------------------------------------------
265            IF(LDLAB .AND. NIPOL.EQ.0) THEN
266     C------ process line for possible data labels
267             DO IP = 1, IPTOT
268               CALL STRIP(CPOLNAME(IP),NNAME)
269     C
270     C-------- mark this parameter for reading
271               K = INDEX(LINE,CPOLNAME(IP)(1:NNAME))
272               ITMP0(IP) = K
273               ITMP(IP)  = K
274             ENDDO
275     C
276             DO JP = 1, JPTOT
277               CALL STRIP(CPOLSNAME(JP),NNAME)
278     C
279               CPNAME = 'Top ' // CPOLSNAME(JP)
280               K1 = INDEX(LINE,CPNAME(1:NNAME+4))
281               CPNAME = 'Top_' // CPOLSNAME(JP)
282               K2 = INDEX(LINE,CPNAME(1:NNAME+4))
283               ITMP0(IPTOT+JP) = MAX(K1,K2)
284               ITMP (IPTOT+JP) = MAX(K1,K2)
285     C
286               CPNAME = 'Bot ' // CPOLSNAME(JP)
287               K1 = INDEX(LINE,CPNAME(1:NNAME+4))
288               CPNAME = 'Bot_' // CPOLSNAME(JP)
289               K2 = INDEX(LINE,CPNAME(1:NNAME+4))
290               ITMP0(IPTOT+JP+JPTOT) = MAX(K1,K2)
291               ITMP (IPTOT+JP+JPTOT) = MAX(K1,K2)
292             ENDDO
293     C
294     C------ bubble-sort data label positions in line string
295             DO IPASS = 1, IPTOT+2*JPTOT
296               DO IP = 1, IPTOT+2*JPTOT-1
297                 IF(ITMP(IP).GT.ITMP(IP+1)) THEN
298                  ITMPP1 = ITMP(IP+1)
299                  ITMP(IP+1) = ITMP(IP)
300                  ITMP(IP) = ITMPP1
301                 ENDIF
302               ENDDO
303             ENDDO
304     C
305     C------ assign data position to each parameter
306             DO IPT = 1, IPTOT+2*JPTOT
307               IF(ITMP(IPT).GT.0) THEN
308                NIPOL = NIPOL + 1
309                DO IP = 1, IPTOT
310                  IF(ITMP(IPT).EQ.ITMP0(IP)) IPOL(IP) = NIPOL
311                ENDDO
312                DO JP = 1, JPTOT
313                  IF(ITMP(IPT).EQ.ITMP0(IPTOT+JP      )) ISPOL(1,JP) = NIPOL
314                  IF(ITMP(IPT).EQ.ITMP0(IPTOT+JPTOT+JP)) ISPOL(2,JP) = NIPOL
315                ENDDO
316               ENDIF
317             ENDDO
318     C
319            ENDIF
320     C
321     C--------------------------------------------
322            IF(INDEX(LINE,'-----').NE.0) THEN
323             LHEAD = .FALSE.
324            ENDIF
325     C
326     C--------------------------------------------------------------
327           ELSE
328     C----- read polar data lines
329            IA = NA + 1
330     C
331            NINP = IPTOT+2*JPTOT
332            CALL GETFLT(LINE,RINP(1),NINP,ERROR)
333            IF(ERROR) GO TO 90
334     C
335            DO IP = 1, IPTOT
336              CPOL(IA,IP) = RINP(IPOL(IP))
337            ENDDO
338     C
339            DO JP = 1, JPTOT
340              DO N = 1, NBL
341                IS1 = 2*N-1
342                IS2 = 2*N
343                CPOLSD(IA,IS1,JP) = RINP(ISPOL(1,JP)+2*(N-1))
344                CPOLSD(IA,IS2,JP) = RINP(ISPOL(2,JP)+2*(N-1))
345              ENDDO
346            ENDDO
347     C
348            ACL = MAX( CPOL(IA,ICL) , 0.001 )
349     C
350     C
351     C----- try to find Re, Ma, Ncrit, Xtrip  in polar data
352            LIRE = .FALSE.
353            LIMA = .FALSE.
354            LINC = .FALSE.
355            LJTP = .FALSE.
356            DO KP = 1, NIPOL
357              IF(IPOL(KP) .EQ. IRE) LIRE = .TRUE.
358              IF(IPOL(KP) .EQ. IMA) LIMA = .TRUE.
359              IF(IPOL(KP) .EQ. INC) LINC = .TRUE.
360              IF(ISPOL(1,KP) .EQ. JTP) LJTP = .TRUE.
361            ENDDO
362     C
363            IF(.NOT. LIRE) THEN
364     C------ Re was not in polar data... set using header info
365             IF    (IRETYP.EQ.1) THEN
366              CPOL(IA,IRE) = REYN1
367             ELSEIF(IRETYP.EQ.2) THEN
368              CPOL(IA,IRE) = REYN1/SQRT(ACL) 
369             ELSEIF(IRETYP.EQ.3) THEN
370             CPOL(IA,IRE) = REYN1/ACL
371             ENDIF
372            ENDIF
373     C
374            IF(.NOT. LIMA) THEN
375     C------ Mach was not in polar data... set using header info
376             IF    (IMATYP.EQ.1) THEN
377              CPOL(IA,IMA) = MACH1
378             ELSEIF(IMATYP.EQ.2) THEN
379              CPOL(IA,IMA) = MACH1/SQRT(ACL)
380             ELSEIF(IMATYP.EQ.3) THEN
381              CPOL(IA,IMA) = MACH1/ACL
382             ENDIF
383            ENDIF
384     C
385            IF(.NOT. LINC) THEN
386     C------ Ncrit was not in polar data... set using header info
387             CPOL(IA,INC) = ACRIT
388            ENDIF
389     C
390            IF(.NOT. LJTP) THEN
391     C------ set trip data using header info
392             DO IS = 1, 2*NBL
393               CPOLSD(IA,IS,JTP) = XTRIP(IS)
394             ENDDO
395            ENDIF
396     C
397            NA = IA
398           ENDIF
399     C
400      60   CONTINUE
401     C---- go read next line
402           GO TO 500
403     C=============================================================
404     C
405      80   CONTINUE
406     C---- if file was opened here, then close it
407           IF(LOPEN) CLOSE(LU)
408           RETURN
409     C
410      90   CONTINUE
411           IF(LOPEN) CLOSE(LU)
412           ERROR = .TRUE.
413           RETURN
414     C
415     C..........................................
416      1000 FORMAT(A)
417           END ! POLREAD

420           SUBROUTINE POLWRIT(LU,FNPOL,ERROR, LHEAD,
421          &                   NAX, IA1,IA2, CPOL, IPOL,NIPOL,
422          &                   REYN1,MACH1,ACRIT,XTRIP,
423          &                   PTRAT,ETAP,
424          &                   NAME, IRETYP,IMATYP,
425          &                   ISX,NBL,CPOLSD, JPOL,NJPOL,
426          &                   CODE,VERSION, LQUERY )
427           INCLUDE 'PINDEX.INC'
428           CHARACTER*(*) FNPOL, NAME
429           LOGICAL ERROR, LHEAD,LQUERY
430           CHARACTER*(*) CODE
431           REAL CPOL(NAX,IPTOT), CPOLSD(NAX,ISX,JPTOT)
432           REAL MACH1, XTRIP(ISX)
433           INTEGER IPOL(IPTOT), JPOL(JPTOT)
434     C--------------------------------------------------------
435     C     Writes polar save file
436     C
437     C  Input:
438     C     LU       logical unit to use for writing
439     C     FNPOL    name of polar file to be read,
440     C                if FNPOL(1:1).eq.' ', unit LU is assumed 
441     C                to be already open
442     C     NAX      polar point array dimension
443     C     ISX      airfoil side array dimension
444     C     IA1,IA2  only polar points IA1..IA2 are written
445     C     CPOL     polar coefficients and parameters
446     C     IPOL(.)  indices of data quantities to be written
447     C     NIPOL    number  of data quantities to be written
448     C     REYN1    Reynolds number for CL=1
449     C     MACH1    Mach number for CL=1
450     C     ACRIT    Critical amplification ratio
451     C     XTRIP    Trip locations
452     C     PTRAT   Actuator disk total-pressure ratio
453     C     ETAP    Actuator disk thermal efficiency
454     C     NAME     airfoil name string
455     C     IRETYP   flag giving type of Re variation with CL
456     C     IMATYP   flag giving type of Ma variation with CL
457     C     NBL      number of airfoil elements
458     C     CPOLSD   airfoil side-related parameters
459     C     JPOL(.)  indices of side data quantities to be written
460     C     NJPOL    number  of side data quantities to be written
461     C     LHEAD    T if header and column label are to be written
462     C     CODE     code used to compute polar
463     C     VERSION  code version
464     C     LQUERY   if T, asks permission to overwrite existing file
465     C
466     C  Output:
467     C     ERROR   T if a OPER or WRITE error occurred
468     C--------------------------------------------------------
469           CHARACTER*29 LINE1, LINE2
470           CHARACTER*128 LINEL, LINED, LINEF
471           CHARACTER*1 ANS
472           LOGICAL LOPEN
473     C
474           ERROR = .FALSE.
475     C
476     C---- do we have to open the file?
477           LOPEN = FNPOL .NE. ' '
478     C
479           IF(LOPEN) THEN
480            OPEN(LU,FILE=FNPOL,STATUS='OLD',ERR=20)
481     C
482            IF(LQUERY) THEN
483             WRITE(*,*)
484             WRITE(*,*) 'Output file exists.  Overwrite?  Y'
485             READ(*,1000) ANS
486     C
487             IF(INDEX('Nn',ANS).EQ.0) GO TO 22
488     C
489             CLOSE(LU)
490             WRITE(*,*) 'Polar file not saved'
491             RETURN
492            ENDIF
493     C
494      20    OPEN(LU,FILE=FNPOL,STATUS='UNKNOWN',ERR=90)
495      22    REWIND(LU)
496           ENDIF
497     C
498           IF(LHEAD) THEN
499            WRITE(LU,*) ' '
500            WRITE(LU,8000) CODE, VERSION
501            WRITE(LU,*) ' '
502            IF(NBL.EQ.1) THEN
503             WRITE(LU,9001) NAME
504            ELSE
505             WRITE(LU,9002) NAME, NBL
506            ENDIF
507     C
508            IFFBC = 0
509            ISMOM = 0
510     C
511            IF(IFFBC.NE.0 .AND. ISMOM.NE.0) THEN
512             IF(IFFBC.EQ.1)  LINE1 = ' Solid wall far field        '
513             IF(IFFBC.EQ.2)  LINE1 = ' Vortex + doublet far field  '
514             IF(IFFBC.EQ.3)  LINE1 = ' Constant pressure far field '
515             IF(IFFBC.EQ.4)  LINE1 = ' Supersonic wave far field   '
516             IF(IFFBC.GE.5)  LINE1 = '                             '
517             IF(ISMOM.EQ.1)  LINE2 = '   S-momentum conserved      '
518             IF(ISMOM.EQ.2)  LINE2 = '   Entropy conserved         '
519             IF(ISMOM.EQ.3)  LINE2 = '   Entropy conserved near LE '
520             IF(ISMOM.EQ.4)  LINE2 = '   S-mom conserved at shocks '
521             IF(ISMOM.GE.5)  LINE2 = '                             '
522             WRITE(LU,9006) LINE1, LINE2
523      9006   FORMAT(1X,3X,2A29)
524            ENDIF
525     C
526            WRITE(LU,*) ' '
527     C
528            LINE1 = ' '
529            LINE2 = ' '
530            IF(IRETYP.EQ.1) LINE1 = ' Reynolds number fixed       '
531            IF(IRETYP.EQ.2) LINE1 = ' Reynolds number ~ 1/sqrt(CL)'
532            IF(IRETYP.EQ.3) LINE1 = ' Reynolds number ~ 1/CL      '
533            IF(IMATYP.EQ.1) LINE2 = '   Mach number fixed         '
534            IF(IMATYP.EQ.2) LINE2 = '   Mach number ~ 1/sqrt(CL)  '
535            IF(IMATYP.EQ.3) LINE2 = '   Mach number ~ 1/CL        '
536            WRITE(LU,9005) IRETYP, IMATYP, LINE1, LINE2
537     C
538            WRITE(LU,*) ' '
539            DO N = 1, NBL
540              IS1 = 2*N-1
541              IS2 = 2*N
542              IF(NBL.EQ.1) THEN
543               WRITE(LU,9011) XTRIP(IS1), XTRIP(IS2)
544              ELSE
545               WRITE(LU,9012) XTRIP(IS1), XTRIP(IS2), N
546              ENDIF
547            ENDDO
548            WRITE(LU,9015) MACH1, REYN1/1.0E6, ACRIT
549            IF(PTRAT .NE. 0.0) WRITE(LU,9017) PTRAT, ETAP
550            WRITE(LU,*) ' '
551     C
552            LINEL = ' '
553            LINED = ' '
554     C
555            KL = 1
556            KD = 1
557     C
558            DO 30 KP = 1, NIPOL
559              IP = IPOL(KP)
560              IF(IP.EQ.0) GO TO 30
561     C
562              KDOT = INDEX(CPOLFORM(IP),'.')
563              IF(KDOT.EQ.0) KDOT = LEN(CPOLFORM(IP))
564              READ(CPOLFORM(IP)(2:KDOT-1),*,ERR=95) NFORM
565     C
566              CALL STRIP(CPOLNAME(IP),NNAME) 
567              NBLANK = MAX( (NFORM-NNAME+2)/2 , 0 )
568     C
569              LINEL(KL+1+NBLANK:KL+NNAME+NBLANK) = CPOLNAME(IP)(1:NNAME)
570              KL = KL + NFORM
571     C
572              LINED(KD+2:KD+NFORM) = '--------------------------------'
573              KD = KD + NFORM
574      30    CONTINUE
575     C
576            DO 32 KP = 1, NJPOL
577              JP = JPOL(KP)
578              IF(JP.EQ.0) GO TO 32
579     C
580              KDOT = INDEX(CPOLSFORM(JP),'.')
581              IF(KDOT.EQ.0) KDOT = LEN(CPOLSFORM(JP))
582              READ(CPOLSFORM(JP)(2:KDOT-1),*,ERR=95) NFORM
583     C
584              CALL STRIP(CPOLSNAME(JP),NNAME)
585              NBLANK = MAX( (NFORM-NNAME-2)/2 , 0 )
586     C
587              DO N = 1, NBL
588                LINEL(KL+1+NBLANK:KL+4+NNAME+NBLANK) = 
589          &      'Top_' // CPOLSNAME(JP)(1:NNAME)
590                KL = KL + NFORM
591     C
592                LINED(KD+2:KD+NFORM) = '--------------------------------'
593                KD = KD + NFORM
594     C
595                LINEL(KL+1+NBLANK:KL+4+NNAME+NBLANK) = 
596          &      'Bot_' // CPOLSNAME(JP)(1:NNAME)
597                KL = KL + NFORM
598     C
599                LINED(KD+2:KD+NFORM) = '--------------------------------'
600                KD = KD + NFORM
601              ENDDO
602      32    CONTINUE
603     C
604     C 
605     C
606     C       LINEL = 
607     C     & '  alpha     CL        CD       CDp       CM    Top_Xtr Bot_Xtr'
608     CCC     1234567890123456789012345678901234567890123456789012345678901234567890
609     C       K = 62
610     C
611     C
612     C       LINEL =
613     C     & ' ------- -------- --------- --------- -------- ------- -------'
614     CCC       3.453   1.3750   0.00921   0.00512  -0.1450  0.9231  0.5382
615     CCC       3.453   1.3750   0.00921   0.00213  -0.1450  0.9231  0.5382
616     C       K = 62
617     
618            WRITE(LU,1000) LINEL(1:KL)
619            WRITE(LU,1000) LINED(1:KD)
620     C
621           ENDIF
622     C
623     
624           LINEF = '(1X'
625           KF = 3
626           DO KP = 1, NIPOL
627             IP = IPOL(KP)
628             NF = LEN(CPOLFORM(IP))
629     C
630             LINEF(KF+1:KF+NF+1) = ',' // CPOLFORM(IP)
631             KF = KF + NF + 1
632           ENDDO
633           DO KP = 1, NJPOL
634             JP = JPOL(KP)

690           SUBROUTINE POLREF(LU,FNREF,ERROR,
691          &            NFX,NF,XYREF,LABREF )
692           INCLUDE 'PINDEX.INC'
693           CHARACTER*(*) FNREF,LABREF
694           LOGICAL ERROR
695           DIMENSION NF(4)
696           DIMENSION XYREF(NFX,2,4)
697     C--------------------------------------------------------
698     C     Reads in polar reference data file
699     C
700     C  Input:
701     C     LU      logical unit to use for reading
702     C     FNREF   name of polar file to be read,
703     C               if FNREF(1:1).eq.' ', unit LU is assumed 
704     C               to be already open
705     C     NFX     polar point array dimension
706     C
707     C  Output:
708     C     ERROR      T if a READ error occurred
709     C     NF(.)      number of points in each data block
710     C     XYREF(...) reference polar data
711     C     LABREF(.)  reference polar label
712     C--------------------------------------------------------
713           LOGICAL LOPEN
714           CHARACTER*80 LINE
715     C
716           ERROR = .FALSE.
717           LOPEN = FNREF(1:1) .NE. ' '
718           IF(LOPEN) OPEN(LU,FILE=FNREF,STATUS='OLD',ERR=900)
719     C
720     C---- try to read data label
721           READ(LU,1000,END=900) LINE
722      1000 FORMAT(A)
723     C
724     C---- set data label if present
725           IF(LINE(1:1).EQ.'#') THEN
726             LABREF = LINE(2:80)
727           ELSE
728             LABREF = ' '
729             REWIND(LU)
730           ENDIF
731     C
732           DO 100 K=1, 4
733             DO 10 I=1, NFX
734               READ(LU,*,END=11,ERR=900) XYREF(I,1,K), XYREF(I,2,K)
735               IF(XYREF(I,1,K) .EQ. 999.0) GO TO 11
736        10   CONTINUE
737        11   NF(K) = I-1
738       100 CONTINUE
739           IF(LOPEN) CLOSE(LU)
740           RETURN
741     C
742       900 CONTINUE
743           ERROR = .TRUE.
