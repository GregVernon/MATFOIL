21           SUBROUTINE LEFIND(SLE,X,XP,Y,YP,S,N)
22           DIMENSION X(*),XP(*),Y(*),YP(*),S(*)
23     C------------------------------------------------------
24     C     Locates leading edge spline-parameter value SLE
25     C
26     C     The defining condition is
27     C         
28     C      (X-XTE,Y-YTE) . (X',Y') = 0     at  S = SLE
29     C
30     C     i.e. the surface tangent is normal to the chord
31     C     line connecting X(SLE),Y(SLE) and the TE point.
32     C------------------------------------------------------
33     C
34     C---- convergence tolerance
35           DSEPS = (S(N)-S(1)) * 1.0E-5
36     C
37     C---- set trailing edge point coordinates
38           XTE = 0.5*(X(1) + X(N))
39           YTE = 0.5*(Y(1) + Y(N))
40     C
41     C---- get first guess for SLE
42           DO 10 I=3, N-2
43             DXTE = X(I) - XTE
44             DYTE = Y(I) - YTE
45             DX = X(I+1) - X(I)
46             DY = Y(I+1) - Y(I)
47             DOTP = DXTE*DX + DYTE*DY
48             IF(DOTP .LT. 0.0) GO TO 11
49        10 CONTINUE
50     C
51        11 SLE = S(I)
52     C
53     C---- check for sharp LE case
54           IF(S(I) .EQ. S(I-1)) THEN
55     ccc        WRITE(*,*) 'Sharp LE found at ',I,SLE
56             RETURN
57           ENDIF
58     C
59     C---- Newton iteration to get exact SLE value
60           DO 20 ITER=1, 50
61             XLE  = SEVAL(SLE,X,XP,S,N)
62             YLE  = SEVAL(SLE,Y,YP,S,N)
63             DXDS = DEVAL(SLE,X,XP,S,N)
64             DYDS = DEVAL(SLE,Y,YP,S,N)
65             DXDD = D2VAL(SLE,X,XP,S,N)
66             DYDD = D2VAL(SLE,Y,YP,S,N)
67     C
68             XCHORD = XLE - XTE
69             YCHORD = YLE - YTE
70     C
71     C------ drive dot product between chord line and LE tangent to zero
72             RES  = XCHORD*DXDS + YCHORD*DYDS
73             RESS = DXDS  *DXDS + DYDS  *DYDS
74          &       + XCHORD*DXDD + YCHORD*DYDD
75     C
76     C------ Newton delta for SLE 
77             DSLE = -RES/RESS
78     C
79             DSLE = MAX( DSLE , -0.02*ABS(XCHORD+YCHORD) )
80             DSLE = MIN( DSLE ,  0.02*ABS(XCHORD+YCHORD) )
81             SLE = SLE + DSLE
82             IF(ABS(DSLE) .LT. DSEPS) RETURN
83        20 CONTINUE
84           WRITE(*,*) 'LEFIND:  LE point not found.  Continuing...'
85           SLE = S(I)
86           RETURN
87           END

91           SUBROUTINE XLFIND(SLE,X,XP,Y,YP,S,N)
92           DIMENSION X(*),XP(*),Y(*),YP(*),S(*)
93     C------------------------------------------------------
94     C     Locates leftmost (minimum x) point location SLE
95     C
96     C     The defining condition is
97     C         
98     C      X' = 0     at  S = SLE
99     C
100     C     i.e. the surface tangent is vertical
101     C------------------------------------------------------
102     C
103           DSLEN = S(N) - S(1)
104     C
105     C---- convergence tolerance
106           DSEPS = (S(N)-S(1)) * 1.0E-5
107     C
108     C---- get first guess for SLE
109           DO 10 I=3, N-2
110             DX = X(I+1) - X(I)
111             IF(DX .GT. 0.0) GO TO 11
112        10 CONTINUE
113     C
114        11 SLE = S(I)
115     C
116     C---- check for sharp LE case
117           IF(S(I) .EQ. S(I-1)) THEN
118     ccc        WRITE(*,*) 'Sharp LE found at ',I,SLE
119             RETURN
120           ENDIF
121     C
122     C---- Newton iteration to get exact SLE value
123           DO 20 ITER=1, 50
124             DXDS = DEVAL(SLE,X,XP,S,N)
125             DXDD = D2VAL(SLE,X,XP,S,N)
126     C
127     C------ drive DXDS to zero
128             RES  = DXDS
129             RESS = DXDD
130     C
131     C------ Newton delta for SLE 
132             DSLE = -RES/RESS
133     C
134             DSLE = MAX( DSLE , -0.01*ABS(DSLEN) )
135             DSLE = MIN( DSLE ,  0.01*ABS(DSLEN) )
136             SLE = SLE + DSLE
137             IF(ABS(DSLE) .LT. DSEPS) RETURN
138        20 CONTINUE
139           WRITE(*,*) 'XLFIND:  Left point not found.  Continuing...'
140           SLE = S(I)
141           RETURN
142           END ! XLFIND

146           SUBROUTINE NSFIND(SLE,X,XP,Y,YP,S,N)
147           REAL X(*),Y(*),S(*),XP(*),YP(*)
148     C----------------------------------------------------------
149     C     Finds "nose" of airfoil where curvature is a maximum
150     C----------------------------------------------------------
151     C
152           PARAMETER (NMAX=500)
153           DIMENSION A(NMAX), B(NMAX), C(NMAX), CV(NMAX)
154     C
155           IF(N.GT.NMAX) STOP 'NSFIND: Local array overflow. Increase NMAX.'
156     C
157     C---- set up curvature array
158           DO 3 I=1, N
159             CV(I) = CURV(S(I),X,XP,Y,YP,S,N)
160         3 CONTINUE
161     C
162     C---- curvature smoothing length
163           SMOOL = 0.006*(S(N)-S(1))
164     C
165     C---- set up tri-diagonal system for smoothed curvatures
166           SMOOSQ = SMOOL**2
167           A(1) = 1.0
168           C(1) = 0.0
169           DO 4 I=2, N-1
170             DSM = S(I) - S(I-1)
171             DSP = S(I+1) - S(I)
172             DSO = 0.5*(S(I+1) - S(I-1))
173     C
174             IF(DSM.EQ.0.0 .OR. DSP.EQ.0.0) THEN
175     C------- leave curvature at corner point unchanged
176              B(I) = 0.0
177              A(I) = 1.0
178              C(I) = 0.0
179             ELSE
180              B(I) =  SMOOSQ * (         - 1.0/DSM) / DSO
181              A(I) =  SMOOSQ * ( 1.0/DSP + 1.0/DSM) / DSO  +  1.0
182              C(I) =  SMOOSQ * (-1.0/DSP          ) / DSO
183             ENDIF
184         4 CONTINUE
185           B(N) = 0.0
186           A(N) = 1.0
187     C
188           CALL TRISOL(A,B,C,CV,N)
189     C
190     C---- find max curvature index
191           CVMAX = 0.
192           IVMAX = 0
193           DO 71 I=2, N-1
194             IF(ABS(CV(I)) .GT. CVMAX) THEN
195              CVMAX = ABS(CV(I))
196              IVMAX = I
197             ENDIF
198        71 CONTINUE
199     C
200     C---- fit a parabola to the curvature at the three points near maximum
201           I = IVMAX
202     C
203           IP = I+1
204           IM = I-1
205           IF(S(I) .EQ. S(IP)) IP = I+2
206           IF(S(I) .EQ. S(IM)) IM = I-2
207     
208           DSM = S(I) - S(IM)
209           DSP = S(IP) - S(I)
210     C
211           CVSM = (CV(I)-CV(IM))/DSM
212           CVSP = (CV(IP)-CV(I))/DSP
213     C
214     C---- 1st and 2nd derivatives at i=IVMAX
215           CVS = (CVSM*DSP + CVSP*DSM)/(DSP+DSM)
216           CVSS = 2.0*(CVSP-CVSM)/(DSP+DSM)
217     C
218     C---- set location of arc length at maximum of parabola
219           DS = -CVS/CVSS
220           SLE = S(I) + DS
221     C
222           RETURN
223           END

226           SUBROUTINE SOPPS(SOPP, SI, X,XP,Y,YP,S,N, SLE)
227           DIMENSION X(*),XP(*),Y(*),YP(*),S(*)
228     C--------------------------------------------------
229     C     Calculates arc length SOPP of point 
230     C     which is opposite of point SI, on the 
231     C     other side of the airfoil baseline
232     C--------------------------------------------------
233     C
234     C---- reference length for testing convergence
235           SLEN = S(N) - S(1)
236     C
237     C---- set chordline vector
238           XLE = SEVAL(SLE,X,XP,S,N)
239           YLE = SEVAL(SLE,Y,YP,S,N)
240           XTE = 0.5*(X(1)+X(N))
241           YTE = 0.5*(Y(1)+Y(N))
242           CHORD = SQRT((XTE-XLE)**2 + (YTE-YLE)**2)
243           DXC = (XTE-XLE) / CHORD
244           DYC = (YTE-YLE) / CHORD
245     C
246           IF(SI.LT.SLE) THEN
247            IN = 1
248            INOPP = N
249           ELSE
250            IN = N
251            INOPP = 1
252           ENDIF
253           SFRAC = (SI-SLE)/(S(IN)-SLE)
254           SOPP = SLE + SFRAC*(S(INOPP)-SLE)
255     C     
256           IF(ABS(SFRAC) .LE. 1.0E-5) THEN
257            SOPP = SLE
258            RETURN
259           ENDIF
260     C
261     C---- XBAR = x coordinate in chord-line axes
262           XI  = SEVAL(SI , X,XP,S,N)
263           YI  = SEVAL(SI , Y,YP,S,N)
264           XLE = SEVAL(SLE, X,XP,S,N)
265           YLE = SEVAL(SLE, Y,YP,S,N)
266           XBAR = (XI-XLE)*DXC + (YI-YLE)*DYC
267     C
268     C---- converge on exact opposite point with same XBAR value
269           DO 300 ITER=1, 12
270             XOPP  = SEVAL(SOPP,X,XP,S,N)
271             YOPP  = SEVAL(SOPP,Y,YP,S,N)
272             XOPPD = DEVAL(SOPP,X,XP,S,N)
273             YOPPD = DEVAL(SOPP,Y,YP,S,N)
274     C
275             RES  = (XOPP -XLE)*DXC + (YOPP -YLE)*DYC - XBAR
276             RESD =  XOPPD     *DXC +  YOPPD     *DYC
277     C
278             IF(ABS(RES)/SLEN .LT. 1.0E-5) GO TO 305
279             IF(RESD .EQ. 0.0) GO TO 303
280     C
281             DSOPP = -RES/RESD
282             SOPP = SOPP + DSOPP
283     C
284             IF(ABS(DSOPP)/SLEN .LT. 1.0E-5) GO TO 305
285      300  CONTINUE
286      303  WRITE(*,*)
287          &      'SOPPS: Opposite-point location failed. Continuing...'
288           SOPP = SLE + SFRAC*(S(INOPP)-SLE)
289     C
290      305  CONTINUE
291           RETURN
292           END ! SOPPS

296           SUBROUTINE NORM(X,XP,Y,YP,S,N)
297           DIMENSION X(*),XP(*),Y(*),YP(*),S(*)
298     C-----------------------------------------------
299     C     Scales coordinates to get unit chord
300     C-----------------------------------------------
301     C
302           CALL SCALC(X,Y,S,N)
303           CALL SEGSPL(X,XP,S,N)
304           CALL SEGSPL(Y,YP,S,N)
305     C
306           CALL LEFIND(SLE,X,XP,Y,YP,S,N)
307     C
308           XMAX = 0.5*(X(1) + X(N))
309           XMIN = SEVAL(SLE,X,XP,S,N)
310           YMIN = SEVAL(SLE,Y,YP,S,N)
311     C
312           FUDGE = 1.0/(XMAX-XMIN)
313           DO 40 I=1, N
314             X(I) = (X(I)-XMIN)*FUDGE
315             Y(I) = (Y(I)-YMIN)*FUDGE
316             S(I) = S(I)*FUDGE
317        40 CONTINUE
318     C
319           RETURN
320           END

323           SUBROUTINE GEOPAR(X,XP,Y,YP,S,N, T,
324          &             SLE,CHORD,AREA,RADLE,ANGTE,
325          &             EI11A,EI22A,APX1A,APX2A,
326          &             EI11T,EI22T,APX1T,APX2T,
327          &             THICK,CAMBR)
328           DIMENSION X(*), XP(*), Y(*), YP(*), S(*), T(*)
329     C
330           PARAMETER (IBX=600)
331           DIMENSION
332          &     XCAM(2*IBX), YCAM(2*IBX), YCAMP(2*IBX),
333          &     XTHK(2*IBX), YTHK(2*IBX), YTHKP(2*IBX)
334     C------------------------------------------------------
335     C     Sets geometric parameters for airfoil shape
336     C------------------------------------------------------
337           CALL LEFIND(SLE,X,XP,Y,YP,S,N)
338     C
339           XLE = SEVAL(SLE,X,XP,S,N)
340           YLE = SEVAL(SLE,Y,YP,S,N)
341           XTE = 0.5*(X(1)+X(N))
342           YTE = 0.5*(Y(1)+Y(N))
343     C
344           CHSQ = (XTE-XLE)**2 + (YTE-YLE)**2
345           CHORD = SQRT(CHSQ)
346     C
347           CURVLE = CURV(SLE,X,XP,Y,YP,S,N)
348     C
349           RADLE = 0.0
350           IF(ABS(CURVLE) .GT. 0.001*(S(N)-S(1))) RADLE = 1.0 / CURVLE
351     C
352           ANG1 = ATAN2( -YP(1) , -XP(1) )
353           ANG2 = ATANC(  YP(N) ,  XP(N) , ANG1 )
354           ANGTE = ANG2 - ANG1
355     C
356     
357           DO I=1, N
358             T(I) = 1.0
359           ENDDO
360     C
361           CALL AECALC(N,X,Y,T, 1, 
362          &            AREA,XCENA,YCENA,EI11A,EI22A,APX1A,APX2A)
363     C
364           CALL AECALC(N,X,Y,T, 2, 
365          &            SLEN,XCENT,YCENT,EI11T,EI22T,APX1T,APX2T)
366     C
367     C--- Old, approximate thickness,camber routine (on discrete points only)
368           CALL TCCALC(X,XP,Y,YP,S,N, THICK,XTHICK, CAMBR,XCAMBR )
369     C
370     C--- More accurate thickness and camber estimates
371     cc      CALL GETCAM(XCAM,YCAM,NCAM,XTHK,YTHK,NTHK,
372     cc     &            X,XP,Y,YP,S,N )
373     cc      CALL GETMAX(XCAM,YCAM,YCAMP,NCAM,XCAMBR,CAMBR)
374     cc      CALL GETMAX(XTHK,YTHK,YTHKP,NTHK,XTHICK,THICK)
375     cc      THICK = 2.0*THICK
376     C
377           WRITE(*,1000) THICK,XTHICK,CAMBR,XCAMBR
378      1000 FORMAT( ' Max thickness = ',F12.6,'  at x = ',F7.3,
379          &       /' Max camber    = ',F12.6,'  at x = ',F7.3)
380     
381     
382     C
383           RETURN
384           END ! GEOPAR

387           SUBROUTINE AECALC(N,X,Y,T, ITYPE, 
388          &                  AREA,XCEN,YCEN,EI11,EI22,APX1,APX2)
389           DIMENSION X(*),Y(*),T(*)
390     C---------------------------------------------------------------
391     C     Calculates geometric properties of shape X,Y
392     C
393     C     Input:
394     C       N      number of points
395     C       X(.)   shape coordinate point arrays
396     C       Y(.)
397     C       T(.)   skin-thickness array, used only if ITYPE = 2
398     C       ITYPE  = 1 ...   integration is over whole area  dx dy
399     C              = 2 ...   integration is over skin  area   t ds
400     C
401     C     Output:
402     C       XCEN,YCEN  centroid location
403     C       EI11,EI22  principal moments of inertia
404     C       APX1,APX2  principal-axis angles
405     C---------------------------------------------------------------
406           DATA PI / 3.141592653589793238 /
407     C
408           SINT  = 0.0
409           AINT  = 0.0
410           XINT  = 0.0
411           YINT  = 0.0
412           XXINT = 0.0
413           XYINT = 0.0
414           YYINT = 0.0
415     C
416           DO 10 IO = 1, N
417             IF(IO.EQ.N) THEN
418               IP = 1
419             ELSE
420               IP = IO + 1
421             ENDIF
422     C
423             DX =  X(IO) - X(IP)
424             DY =  Y(IO) - Y(IP)
425             XA = (X(IO) + X(IP))*0.50
426             YA = (Y(IO) + Y(IP))*0.50
427             TA = (T(IO) + T(IP))*0.50
428     C
429             DS = SQRT(DX*DX + DY*DY)
430             SINT = SINT + DS
431     
432             IF(ITYPE.EQ.1) THEN
433     C-------- integrate over airfoil cross-section
434               DA = YA*DX
435               AINT  = AINT  +       DA
436               XINT  = XINT  + XA   *DA
437               YINT  = YINT  + YA   *DA/2.0
438               XXINT = XXINT + XA*XA*DA
439               XYINT = XYINT + XA*YA*DA/2.0
440               YYINT = YYINT + YA*YA*DA/3.0
441             ELSE
442     C-------- integrate over skin thickness
443               DA = TA*DS
444               AINT  = AINT  +       DA
445               XINT  = XINT  + XA   *DA
446               YINT  = YINT  + YA   *DA
447               XXINT = XXINT + XA*XA*DA
448               XYINT = XYINT + XA*YA*DA
449               YYINT = YYINT + YA*YA*DA
450             ENDIF
451     C
452      10   CONTINUE
453     C
454           AREA = AINT
455     C
456           IF(AINT .EQ. 0.0) THEN
457             XCEN  = 0.0
458             YCEN  = 0.0
459             EI11  = 0.0
460             EI22  = 0.0
461             APX1 = 0.0
462             APX2 = ATAN2(1.0,0.0)
463             RETURN
464           ENDIF
465     C
466     C
467     C---- calculate centroid location
468           XCEN = XINT/AINT
469           YCEN = YINT/AINT
470     C
471     C---- calculate inertias
472           EIXX = YYINT - YCEN*YCEN*AINT
473           EIXY = XYINT - XCEN*YCEN*AINT
474           EIYY = XXINT - XCEN*XCEN*AINT
475     C
476     C---- set principal-axis inertias, EI11 is closest to "up-down" bending inertia
477           EISQ  = 0.25*(EIXX - EIYY)**2  + EIXY**2
478           SGN = SIGN( 1.0 , EIYY-EIXX )
479           EI11 = 0.5*(EIXX + EIYY) - SGN*SQRT(EISQ)
480           EI22 = 0.5*(EIXX + EIYY) + SGN*SQRT(EISQ)
481     C
482           IF(EI11.EQ.0.0 .OR. EI22.EQ.0.0) THEN
483     C----- vanishing section stiffness
484            APX1 = 0.0
485            APX2 = ATAN2(1.0,0.0)
486     C
487           ELSEIF(EISQ/(EI11*EI22) .LT. (0.001*SINT)**4) THEN
488     C----- rotationally-invariant section (circle, square, etc.)
489            APX1 = 0.0
490            APX2 = ATAN2(1.0,0.0)
491     C
492           ELSE
493     C----- normal airfoil section
494            C1 = EIXY
495            S1 = EIXX-EI11
496     C
497            C2 = EIXY
498            S2 = EIXX-EI22
499     C
500            IF(ABS(S1).GT.ABS(S2)) THEN
501              APX1 = ATAN2(S1,C1)
502              APX2 = APX1 + 0.5*PI
503            ELSE
504              APX2 = ATAN2(S2,C2)
505              APX1 = APX2 - 0.5*PI
506            ENDIF
507     
508            IF(APX1.LT.-0.5*PI) APX1 = APX1 + PI
509            IF(APX1.GT.+0.5*PI) APX1 = APX1 - PI
510            IF(APX2.LT.-0.5*PI) APX2 = APX2 + PI
511            IF(APX2.GT.+0.5*PI) APX2 = APX2 - PI
512     C
513           ENDIF
514     C
515           RETURN
516           END ! AECALC

520           SUBROUTINE TCCALC(X,XP,Y,YP,S,N, 
521          &                  THICK,XTHICK, CAMBR,XCAMBR )
522           DIMENSION X(*),XP(*),Y(*),YP(*),S(*)
523     C---------------------------------------------------------------
524     C     Calculates max thickness and camber at airfoil points
525     C
526     C     Note: this routine does not find the maximum camber or 
527     C           thickness exactly as it only looks at discrete points
528     C
529     C     Input:
530     C       N      number of points
531     C       X(.)   shape coordinate point arrays
532     C       Y(.)
533     C
534     C     Output:
535     C       THICK  max thickness
536     C       CAMBR  max camber
537     C---------------------------------------------------------------
538           CALL LEFIND(SLE,X,XP,Y,YP,S,N)
539           XLE = SEVAL(SLE,X,XP,S,N)
540           YLE = SEVAL(SLE,Y,YP,S,N)
541           XTE = 0.5*(X(1)+X(N))
542           YTE = 0.5*(Y(1)+Y(N))
543           CHORD = SQRT((XTE-XLE)**2 + (YTE-YLE)**2)
544     C
545     C---- set unit chord-line vector
546           DXC = (XTE-XLE) / CHORD
547           DYC = (YTE-YLE) / CHORD
548     C
549           THICK = 0.
550           XTHICK = 0.
551           CAMBR = 0.
552           XCAMBR = 0.
553     C
554     C---- go over each point, finding the y-thickness and camber
555           DO 30 I=1, N
556             XBAR = (X(I)-XLE)*DXC + (Y(I)-YLE)*DYC
557             YBAR = (Y(I)-YLE)*DXC - (X(I)-XLE)*DYC
558     C
559     C------ set point on the opposite side with the same chord x value
560             CALL SOPPS(SOPP, S(I), X,XP,Y,YP,S,N, SLE)
561             XOPP = SEVAL(SOPP,X,XP,S,N)
562             YOPP = SEVAL(SOPP,Y,YP,S,N)
563     C
564             YBAROP = (YOPP-YLE)*DXC - (XOPP-XLE)*DYC
565     C
566             YC = 0.5*(YBAR+YBAROP)
567             YT =  ABS(YBAR-YBAROP)
568     C
569             IF(ABS(YC) .GT. ABS(CAMBR)) THEN
570              CAMBR = YC
571              XCAMBR = XOPP
572             ENDIF
573             IF(ABS(YT) .GT. ABS(THICK)) THEN
574              THICK = YT
575              XTHICK = XOPP
576             ENDIF
577        30 CONTINUE
578     C
579           RETURN
580           END ! TCCALC

584           SUBROUTINE YSYM(X,XP,Y,YP,S,NX,N,ISIDE, XNEW,YNEW)
585     C---------------------------------------------------------
586     C     Makes passed-in airfoil symmetric about chord line.
587     C---------------------------------------------------------
588     
589           DIMENSION X(NX),XP(NX),Y(NX),YP(NX),S(NX)
590           DIMENSION XNEW(NX), YNEW(NX)
591     C
592           SREF = S(N) - S(1)
593     C
594           CALL LEFIND(SLE,X,XP,Y,YP,S,N)
595           XLE = SEVAL(SLE,X,XP,S,N)
596           YLE = SEVAL(SLE,Y,YP,S,N)
597           XTE = 0.5*(X(1)+X(N))
598           YTE = 0.5*(Y(1)+Y(N))
599           CHSQ = (XTE-XLE)**2 + (YTE-YLE)**2
600     C
601     C---- set unit chord-line vector
602           DXC = (XTE-XLE) / SQRT(CHSQ)
603           DYC = (YTE-YLE) / SQRT(CHSQ)
604     C
605     C---- find index of node ILE which is just before leading edge point
606           DO 5 I=2, N
607             DS = S(I) - S(I-1)
608             IF(S(I)-SLE .GE. -0.01*DS) GO TO 6
609      5    CONTINUE
610      6    CONTINUE
611           ILE = I-1
612     C
613           DS = S(ILE+1) - S(ILE)
614           IF(SLE-S(ILE-1) .LT. 0.1*DS) THEN
615     C------ point is just before LE, we will move it ahead to LE
616             ILE1 = ILE - 1
617             ILE2 = ILE + 1
618           ELSE IF(S(ILE+1)-SLE .LT. 0.1*DS) THEN
619     C------ point is just after LE, we will move it back to LE
620             ILE1 = ILE
621             ILE2 = ILE + 2
622           ELSE
623     C------ no point is near LE ... we will add new point
624             ILE1 = ILE
625             ILE2 = ILE + 1
626           ENDIF
627     C
628     C---- set index limits of side which will set symmetric geometry
629           IF(ISIDE.EQ.1) THEN
630            IG1 = 1
631            IG2 = ILE1
632            IGDIR = +1
633           ELSE
634            IG1 = N
635            IG2 = ILE2
636            IGDIR = -1
637           ENDIF
638     C
639     C---- set new number of points, including LE point
640           NNEW = 2*(IABS(IG2-IG1) + 1) + 1
641           IF(NNEW.GT.NX) STOP 'YSYM:  Array overflow on passed arrays.'
642     C
643     C---- set symmetric geometry
644           DO 10 I=IG1, IG2, IGDIR
645     C
646     C------ coordinates in chord-line axes
647             XBAR = (X(I)-XLE)*DXC + (Y(I)-YLE)*DYC
648             YBAR = (Y(I)-YLE)*DXC - (X(I)-XLE)*DYC
649     C
650             I1 = 1    + (I - IG1)*IGDIR
651             I2 = NNEW - (I - IG1)*IGDIR
652     C
653             XNEW(I1) = XLE + XBAR*DXC - YBAR*DYC
654             XNEW(I2) = XLE + XBAR*DXC + YBAR*DYC
655     C
656             YNEW(I1) = YLE + YBAR*DXC + XBAR*DYC
657             YNEW(I2) = YLE - YBAR*DXC + XBAR*DYC
658      10   CONTINUE
659     C
660     C---- set new LE point
661           XNEW(NNEW/2+1) = XLE
662           YNEW(NNEW/2+1) = YLE
663     C
664     C---- set geometry for returning
665           N = NNEW
666           DO 20 IG = 1, N
667             IF(IGDIR.EQ.+1) THEN
668              I = IG
669             ELSE
670              I = N - IG + 1
671             ENDIF
672             X(I) = XNEW(IG)
673             Y(I) = YNEW(IG)
674      20   CONTINUE
675     C
676           CALL SCALC(X,Y,S,N)
677           CALL SEGSPL(X,XP,S,N)
678           CALL SEGSPL(Y,YP,S,N)
679     C
680           RETURN
681           END ! YSYM

685           SUBROUTINE LERSCL(X,XP,Y,YP,S,N, DOC,RFAC, XNEW,YNEW)
686     C---------------------------------------------------------
687     C     Adjusts airfoil to scale LE radius by factor RFAC.
688     C     Blending of new shape is done with decay length DOC.
689     C---------------------------------------------------------
690           DIMENSION X(*),XP(*),Y(*),YP(*),S(*)
691           DIMENSION XNEW(*), YNEW(*)
692     C
693           CALL LEFIND(SLE,X,XP,Y,YP,S,N)
694           XLE = SEVAL(SLE,X,XP,S,N)
695           YLE = SEVAL(SLE,Y,YP,S,N)
696           XTE = 0.5*(X(1)+X(N))
697           YTE = 0.5*(Y(1)+Y(N))
698           CHORD = SQRT((XTE-XLE)**2 + (YTE-YLE)**2)
699     C
700     C---- set unit chord-line vector
701           DXC = (XTE-XLE) / CHORD
702           DYC = (YTE-YLE) / CHORD
703     C
704           SRFAC = SQRT(ABS(RFAC))
705     C
706     C---- go over each point, changing the y-thickness appropriately
707           DO 30 I=1, N
708             XBAR = (X(I)-XLE)*DXC + (Y(I)-YLE)*DYC
709             YBAR = (Y(I)-YLE)*DXC - (X(I)-XLE)*DYC
710     C
711     C------ set point on the opposite side with the same chord x value
712             CALL SOPPS(SOPP, S(I), X,XP,Y,YP,S,N, SLE)
713             XOPP = SEVAL(SOPP,X,XP,S,N)
714             YOPP = SEVAL(SOPP,Y,YP,S,N)
715     C
716             YBAROP = (YOPP-YLE)*DXC - (XOPP-XLE)*DYC
717     C
718     C------ thickness factor tails off exponentially towards trailing edge
719             XOC = XBAR/CHORD
720             ARG = MIN( XOC/DOC , 15.0 )
721             TFAC = 1.0 - (1.0-SRFAC)*EXP(-ARG)
722     C
723     C------ set new chord x,y coordinates by changing thickness locally
724             YBARCT = 0.5*(YBAR+YBAROP) + TFAC*0.5*(YBAR-YBAROP)
725     C
726             XNEW(I) = XLE + XBAR  *DXC - YBARCT*DYC
727             YNEW(I) = YLE + YBARCT*DXC + XBAR  *DYC
728        30 CONTINUE
729     C
730           RETURN
731           END

735           SUBROUTINE SSS(SS,S1,S2,DEL,XBF,YBF,X,XP,Y,YP,S,N,ISIDE)
736           DIMENSION X(*),XP(*),Y(*),YP(*),S(*)
737     C----------------------------------------------------------------
738     C     Returns arc length points S1,S2 at flap surface break
739     C     locations.  S1 is on fixed airfoil part, S2 is on flap.
740     C     The points are defined according to two cases:
741     C
742     C
743     C     If DEL > 0:  Surface will be eliminated in S1 < s < S2
744     C
745     C     Returns the arc length values S1,S2 of the endpoints
746     C     of the airfoil surface segment which "disappears" as a
747     C     result of the flap deflection.  The line segments between
748     C     these enpoints and the flap hinge point (XBF,YBF) have
749     C     an included angle of DEL.  DEL is therefore the flap
750     C     deflection which will join up the points at S1,S2.
751     C     SS is an approximate arc length value near S1 and S2.
752     C     It is used as an initial guess for the Newton loop 
753     C     for S1 and S2.
754     C
755     C
756     C     If DEL = 0:  Surface will be created at s = S1 = S2
757     C
758     C     If DEL=0, then S1,S2 will cooincide, and will be located
759     C     on the airfoil surface where the segment joining the
760     C     point at S1,S2 and the hinge point is perpendicular to
761     C     the airfoil surface.  This will be the point where the
762     C     airfoil surface must be broken to permit a gap to open
763     C     as a result of the flap deflection.
764     C----------------------------------------------------------------
765     C
766     C---- convergence epsilon
767           DATA EPS / 1.0E-5 /
768     C
769           STOT = ABS( S(N) - S(1) )
770     C
771           SIND = SIN(0.5*ABS(DEL))
772     C
773           SSGN = 1.0
774           IF(ISIDE.EQ.1) SSGN = -1.0
775     C
776     C---- initial guesses for S1, S2
777           RSQ = (SEVAL(SS,X,XP,S,N)-XBF)**2 + (SEVAL(SS,Y,YP,S,N)-YBF)**2
778           S1 = SS - (SIND*SQRT(RSQ) + EPS*STOT)*SSGN
779           S2 = SS + (SIND*SQRT(RSQ) + EPS*STOT)*SSGN
780     C
781     C---- Newton iteration loop
782           DO 10 ITER=1, 10
783             X1  = SEVAL(S1,X,XP,S,N)
784             X1P = DEVAL(S1,X,XP,S,N)
785             Y1  = SEVAL(S1,Y,YP,S,N)
786             Y1P = DEVAL(S1,Y,YP,S,N)
787     C
788             X2  = SEVAL(S2,X,XP,S,N)
789             X2P = DEVAL(S2,X,XP,S,N)
790             Y2  = SEVAL(S2,Y,YP,S,N)
791             Y2P = DEVAL(S2,Y,YP,S,N)
792     C
793             R1SQ = (X1-XBF)**2 + (Y1-YBF)**2
794             R2SQ = (X2-XBF)**2 + (Y2-YBF)**2
795             R1 = SQRT(R1SQ)
796             R2 = SQRT(R2SQ)
797     C
798             RRSQ = (X1-X2)**2 + (Y1-Y2)**2
799             RR = SQRT(RRSQ)
800     C
801             IF(R1.LE.EPS*STOT .OR. R2.LE.EPS*STOT) THEN
802              S1 = SS
803              S2 = SS
804              RETURN
805             ENDIF
806     C
807             R1_S1 = (X1P*(X1-XBF) + Y1P*(Y1-YBF))/R1
808             R2_S2 = (X2P*(X2-XBF) + Y2P*(Y2-YBF))/R2
809     C
810             IF(SIND.GT.0.01) THEN
811     C
812              IF(RR.EQ.0.0) RETURN
813     C
814              RR_S1 =  (X1P*(X1-X2) + Y1P*(Y1-Y2))/RR
815              RR_S2 = -(X2P*(X1-X2) + Y2P*(Y1-Y2))/RR
816     C
817     C------- Residual 1: set included angle via dot product
818              RS1 = ((XBF-X1)*(X2-X1) + (YBF-Y1)*(Y2-Y1))/RR - SIND*R1
819              A11 = ((XBF-X1)*( -X1P) + (YBF-Y1)*( -Y1P))/RR
820          &       + ((  -X1P)*(X2-X1) + (  -Y1P)*(Y2-Y1))/RR
821          &       - ((XBF-X1)*(X2-X1) + (YBF-Y1)*(Y2-Y1))*RR_S1/RRSQ
822          &       - SIND*R1_S1
823              A12 = ((XBF-X1)*(X2P  ) + (YBF-Y1)*(Y2P  ))/RR
824          &       - ((XBF-X1)*(X2-X1) + (YBF-Y1)*(Y2-Y1))*RR_S2/RRSQ
825     C
826     C------- Residual 2: set equal length segments
827              RS2 = R1 - R2
828              A21 = R1_S1
829              A22 =    - R2_S2
830     C
831             ELSE
832     C
833     C------- Residual 1: set included angle via small angle approximation
834              RS1 = (R1+R2)*SIND + (S1 - S2)*SSGN
835              A11 =  R1_S1 *SIND + SSGN
836              A12 =  R2_S2 *SIND - SSGN
837     C
838     C------- Residual 2: set vector sum of line segments beteen the 
839     C-       endpoints and flap hinge to be perpendicular to airfoil surface.
840              X1PP = D2VAL(S1,X,XP,S,N)
841              Y1PP = D2VAL(S1,Y,YP,S,N)
842              X2PP = D2VAL(S2,X,XP,S,N)
843              Y2PP = D2VAL(S2,Y,YP,S,N)
844     C
845              XTOT = X1+X2 - 2.0*XBF
846              YTOT = Y1+Y2 - 2.0*YBF
847     C
848              RS2 = XTOT*(X1P+X2P) + YTOT*(Y1P+Y2P)
849              A21 =  X1P*(X1P+X2P) +  Y1P*(Y1P+Y2P) + XTOT*X1PP + YTOT*Y1PP
850              A22 =  X2P*(X1P+X2P) +  Y2P*(Y1P+Y2P) + XTOT*X2PP + YTOT*Y2PP
851     C
852             ENDIF
853     C
854             DET =   A11*A22 - A12*A21
855             DS1 = -(RS1*A22 - A12*RS2) / DET
856             DS2 = -(A11*RS2 - RS1*A21) / DET
857     C
858             DS1 = MIN( DS1 , 0.01*STOT )
859             DS1 = MAX( DS1 , -.01*STOT )
860             DS2 = MIN( DS2 , 0.01*STOT )
861             DS2 = MAX( DS2 , -.01*STOT )
862     C
863             S1 = S1 + DS1
864             S2 = S2 + DS2
865             IF(ABS(DS1)+ABS(DS2) .LT. EPS*STOT ) GO TO 11
866        10 CONTINUE
867           WRITE(*,*) 'SSS: failed to converge subtending angle points'
868           S1 = SS
869           S2 = SS
870     C
871        11 CONTINUE
872     C
873     C---- make sure points are identical if included angle is zero.
874           IF(DEL.EQ.0.0) THEN
875            S1 = 0.5*(S1+S2)
876            S2 = S1
877           ENDIF
878     C
879           RETURN
880           END

1111           SUBROUTINE CANG(X,Y,N,IPRINT, IMAX,AMAX)
1112           DIMENSION X(*), Y(*)
1113     C-------------------------------------------------------------------
1114     C     IPRINT=2:   Displays all panel node corner angles
1115     C     IPRINT=1:   Displays max panel node corner angle
1116     C     IPRINT=0:   No display... just returns values
1117     C-------------------------------------------------------------------
1118     C
1119           AMAX = 0.0
1120           IMAX = 1
1121     C
1122     C---- go over each point, calculating corner angle
1123           IF(IPRINT.EQ.2) WRITE(*,1050)
1124           DO 30 I=2, N-1
1125             DX1 = X(I) - X(I-1)
1126             DY1 = Y(I) - Y(I-1)
1127             DX2 = X(I) - X(I+1)
1128             DY2 = Y(I) - Y(I+1)
1129     C
1130     C------ allow for doubled points

1163           SUBROUTINE INTER(X0,XP0,Y0,YP0,S0,N0,SLE0,
1164          &                 X1,XP1,Y1,YP1,S1,N1,SLE1,
1165          &                 X,Y,N,FRAC)
1166     C     .....................................................................
1167     C
1168     C     Interpolates two source airfoil shapes into an "intermediate" shape.
1169     C
1170     C     Procedure:
1171     C        The interpolated x coordinate at a given normalized spline 
1172     C        parameter value is a weighted average of the two source 
1173     C        x coordinates at the same normalized spline parameter value.
1174     C        Ditto for the y coordinates. The normalized spline parameter 
1175     C        runs from 0 at the leading edge to 1 at the trailing edge on 
1176     C        each surface.
1177     C     .....................................................................
1178     C
1179           REAL X0(N0),Y0(N0),XP0(N0),YP0(N0),S0(N0)
1180           REAL X1(N1),Y1(N1),XP1(N1),YP1(N1),S1(N1)
1181           REAL X(*),Y(*)
1182     C
1183     C---- number of points in interpolated airfoil is the same as in airfoil 0
1184           N = N0
1185     C
1186     C---- interpolation weighting fractions
1187           F0 = 1.0 - FRAC
1188           F1 = FRAC
1189     C
1190     C---- top side spline parameter increments
1191           TOPS0 = S0(1) - SLE0
1192           TOPS1 = S1(1) - SLE1
1193     C
1194     C---- bottom side spline parameter increments
1195           BOTS0 = S0(N0) - SLE0
1196           BOTS1 = S1(N1) - SLE1
1197     C
1198           DO 50 I=1, N
1199     C
1200     C------ normalized spline parameter is taken from airfoil 0 value
1201             IF(S0(I).LT.SLE0) SN = (S0(I) - SLE0) / TOPS0    ! top side
1202             IF(S0(I).GE.SLE0) SN = (S0(I) - SLE0) / BOTS0    ! bottom side
1203     C
1204     C------ set actual spline parameters
1205             ST0 = S0(I)
1206             IF(ST0.LT.SLE0) ST1 = SLE1 + TOPS1 * SN
1207             IF(ST0.GE.SLE0) ST1 = SLE1 + BOTS1 * SN
1208     C
1209     C------ set input coordinates at common spline parameter location
1210             XT0 = SEVAL(ST0,X0,XP0,S0,N0)
1211             YT0 = SEVAL(ST0,Y0,YP0,S0,N0)
1212             XT1 = SEVAL(ST1,X1,XP1,S1,N1)
1213             YT1 = SEVAL(ST1,Y1,YP1,S1,N1)
1214     C
1215     C------ set interpolated x,y coordinates
1216             X(I) = F0*XT0 + F1*XT1
1217             Y(I) = F0*YT0 + F1*YT1
1218     C
1219        50 CONTINUE
1220     C
1221           RETURN
1222           END ! INTER

1236     C        x coordinates at the same normalized spline parameter value.
1237     C        Ditto for the y coordinates. The normalized spline parameter

1451           SUBROUTINE IJSECT(N,X,Y, PEX,
1452          &  AREA, SLEN, 
1453          &  XMIN, XMAX, XEXINT,
1454          &  YMIN, YMAX, YEXINT,
1455          &  XC , YC , 
1456          &  XCT, YCT, 
1457          &  AIXX , AIYY , 
1458          &  AIXXT, AIYYT,
1459          &  AJ   , AJT    )
1460           DIMENSION X(*), Y(*)
1461     C
1462           XMIN = X(1)
1463           XMAX = X(1)
1464           YMIN = Y(1)
1465           YMAX = Y(1)
1466     C
1467           DX = X(1) - X(N)
1468           DY = Y(1) - Y(N)
1469           DS = SQRT(DX*DX + DY*DY)
1470           XAVG = 0.5*(X(1) + X(N))
1471           YAVG = 0.5*(Y(1) + Y(N))
1472     C
1473           X_DY   = DY * XAVG
1474           XX_DY  = DY * XAVG**2
1475           XXX_DY = DY * XAVG**3
1476           X_DS   = DS * XAVG
1477           XX_DS  = DS * XAVG**2
1478     C
1479           Y_DX   = DX * YAVG
1480           YY_DX  = DX * YAVG**2
1481           YYY_DX = DX * YAVG**3
1482           Y_DS   = DS * YAVG
1483           YY_DS  = DS * YAVG**2
1484     C
1485           C_DS   = DS
1486     C
1487           DO 10 I = 2, N
1488             DX = X(I) - X(I-1)
1489             DY = Y(I) - Y(I-1)
1490             DS = SQRT(DX*DX + DY*DY)
1491             XAVG = 0.5*(X(I) + X(I-1))
1492             YAVG = 0.5*(Y(I) + Y(I-1))
1493     C
1494             X_DY   = X_DY   + DY * XAVG
1495             XX_DY  = XX_DY  + DY * XAVG**2
1496             XXX_DY = XXX_DY + DY * XAVG**3
1497             X_DS   = X_DS   + DS * XAVG
1498             XX_DS  = XX_DS  + DS * XAVG**2
1499     C
1500             Y_DX   = Y_DX   + DX * YAVG
1501             YY_DX  = YY_DX  + DX * YAVG**2
1502             YYY_DX = YYY_DX + DX * YAVG**3
1503             Y_DS   = Y_DS   + DS * YAVG
1504             YY_DS  = YY_DS  + DS * YAVG**2
1505     C
1506             C_DS   = C_DS   + DS
1507     C
1508             XMIN = MIN(XMIN,X(I))
1509             XMAX = MAX(XMAX,X(I))
1510             YMIN = MIN(YMIN,Y(I))
1511             YMAX = MAX(YMAX,Y(I))
1512      10   CONTINUE
1513     C
1514           AREA = -Y_DX
1515           SLEN =  C_DS
1516     C
1517           IF(AREA.EQ.0.0) RETURN
1518     C
1519           XC = XX_DY / (2.0*X_DY)
1520           XCT = X_DS / C_DS
1521           AIYY  =  XXX_DY/3.0 - XX_DY*XC      + X_DY*XC**2
1522           AIYYT =   XX_DS     -  X_DS*XCT*2.0 + C_DS*XCT**2
1523     C
1524           YC = YY_DX / (2.0*Y_DX)
1525           YCT = Y_DS / C_DS
1526           AIXX  = -YYY_DX/3.0 + YY_DX*YC      - Y_DX*YC**2
1527           AIXXT =   YY_DS     -  Y_DS*YCT*2.0 + C_DS*YCT**2
1528     C
1529     C
1530           SINT = 0.
1531           XINT = 0.
1532           YINT = 0.
1533     C
1534           DO 20 I=2, N
1535             DX = X(I) - X(I-1)
1536             DY = Y(I) - Y(I-1)
1537             DS = SQRT(DX*DX + DY*DY)
1538             XAVG = 0.5*(X(I) + X(I-1)) - XC
1539             YAVG = 0.5*(Y(I) + Y(I-1)) - YC
1540     C
1541             SINT = SINT + DS
1542     cc        XINT = XINT + DS * ABS(XAVG)**PEX
1543     cc        YINT = YINT + DS * ABS(YAVG)**PEX
1544      20   CONTINUE
1545     C
1546           DO I=1, N-1
1547             IF(X(I+1) .GE. X(I)) GO TO 30
1548           ENDDO
1549           IMID = N/2
1550      30   IMID = I
1551     C
1552           AJ = 0.0
1553           DO I = 2, IMID
1554             XAVG = 0.5*(X(I) + X(I-1))
1555             YAVG = 0.5*(Y(I) + Y(I-1))
1556             DX = X(I-1) - X(I)
1557     C
1558             IF(XAVG.GT.X(N)) THEN
1559              YOPP = Y(N)
1560              GO TO 41
1561             ENDIF
1562             IF(XAVG.LE.X(IMID)) THEN
1563              YOPP = Y(IMID)
1564              GO TO 41
1565             ENDIF
1566     C
1567             DO J = N, IMID, -1
1568               IF(XAVG.GT.X(J-1) .AND. XAVG.LE.X(J)) THEN
1569                 FRAC = (XAVG - X(J-1))
1570          &           / (X(J) - X(J-1))
1571                 YOPP = Y(J-1) + (Y(J)-Y(J-1))*FRAC
1572                 GO TO 41
1573               ENDIF
1574             ENDDO
1575      41     CONTINUE
1576     C
1577             AJ = AJ + ABS(YAVG-YOPP)**3 * DX / 3.0
1578           ENDDO
1579     C
1580           AJT = 4.0*AREA**2/SLEN
1581     C
1582     cc      XEXINT = (XINT/SINT)**(1.0/PEX)
1583     cc      YEXINT = (YINT/SINT)**(1.0/PEX)
1584     C
1585           RETURN
1586           END ! IJSECT

1590           SUBROUTINE AREFINE(X,Y,S,XS,YS,N, ATOL, 
1591          &                   NDIM,NNEW,XNEW,YNEW,X1,X2)
1592     C-------------------------------------------------------------
1593     C     Adds points to a x,y spline contour wherever 
1594     C     the angle between adjacent segments at a node 
1595     C     exceeds a specified threshold.  The points are 
1596     C     added 1/3 of a segment before and after the 
1597     C     offending node.
1598     C
1599     C     The point adding is done only within X1..X2.
1600     C
1601     C     Intended for doubling the number of points
1602     C     of Eppler and Selig airfoils so that they are
1603     C     suitable for clean interpolation using Xfoil's
1604     C     arc-length spline routines.
1605     C------------------------------------------------------
1606           REAL X(*), Y(*), S(*), XS(*), YS(*)
1607           REAL XNEW(NDIM), YNEW(NDIM)
1608           LOGICAL LREF
1609     C
1610           ATOLR = ATOL * 3.14159/180.0
1611     C
1612           K = 1
1613           XNEW(K) = X(1)
1614           YNEW(K) = Y(1)
1615     C
1616           DO 10 I = 2, N-1
1617             IM = I-1
1618             IP = I+1
1619     C
1620             DXM = X(I) - X(I-1)
1621             DYM = Y(I) - Y(I-1)
1622             DXP = X(I+1) - X(I)
1623             DYP = Y(I+1) - Y(I)
1624     C
1625             CRSP = DXM*DYP - DYM*DXP
1626             DOTP = DXM*DXP + DYM*DYP
1627             IF(CRSP.EQ.0.0 .AND. DOTP.EQ.0.0) THEN
1628              ASEG = 0.0
1629             ELSE
1630              ASEG = ATAN2( CRSP , DOTP )
1631             ENDIF
1632     C
1633             LREF = ABS(ASEG) .GT. ATOLR
1634     C
1635             IF(LREF) THEN
1636     C------- add extra point just before this node
1637              SMID = S(I) - 0.3333*(S(I)-S(I-1))
1638              XK = SEVAL(SMID,X,XS,S,N)
1639              YK = SEVAL(SMID,Y,YS,S,N)
1640              IF(XK.GE.X1 .AND. XK.LE.X2) THEN
1641               K = K + 1
1642               IF(K .GT. NDIM) GO TO 90
1643               XNEW(K) = XK
1644               YNEW(K) = YK
1645              ENDIF
1646             ENDIF
1647     C
1648     C------ add the node itself
1649             K = K + 1
1650             IF(K .GT. NDIM) GO TO 90
1651             XNEW(K) = X(I)
1652             YNEW(K) = Y(I)
1653     C
1654             IF(LREF) THEN
1655     C------- add extra point just after this node
1656              SMID = S(I) + 0.3333*(S(I+1)-S(I))
1657              XK = SEVAL(SMID,X,XS,S,N)
1658              YK = SEVAL(SMID,Y,YS,S,N)
1659              IF(XK.GE.X1 .AND. XK.LE.X2) THEN
1660               K = K + 1
1661               IF(K .GT. NDIM) GO TO 90
1662               XNEW(K) = XK
1663               YNEW(K) = YK
1664              ENDIF
1665             ENDIF
1666      10   CONTINUE
1667     C
1668           K = K + 1
1669           IF(K .GT. NDIM) GO TO 90
1670           XNEW(K) = X(N)
1671           YNEW(K) = Y(N)
1672     C
1673           NNEW = K
1674           RETURN
1675     C
1676      90   CONTINUE
1677           WRITE(*,*) 'SDOUBLE:  Arrays will overflow.  No action taken.'
1678           NNEW = 0
1679           RETURN
1680     C
1681           END ! AREFINE

1684           SUBROUTINE SCHECK(X,Y,N, STOL, LCHANGE)
1685     C-------------------------------------------------------------
1686     C     Removes points from an x,y spline contour wherever 
1687     C     the size of a segment between nodes falls below a 
1688     C     a specified threshold of the adjacent segments.  
1689     C     The two node points defining the short segment are
1690     C     replaced with a single node at their midpoint.
1691     C     Note that the number of nodes may be altered by 
1692     C     this routine.
1693     C
1694     C     Intended for eliminating odd "micro" panels 
1695     C     that occur when blending a flap to a foil.
1696     C     If LCHANGE is set on return the airfoil definition 
1697     C     has been changed and resplining should be done.
1698     C
1699     C     The recommended value for STOL is 0.05 (meaning 
1700     C     segments less than 5% of the length of either adjoining 
1701     C     segment are removed).  4/24/01 HHY
1702     C------------------------------------------------------
1703           REAL X(*), Y(*)
1704           LOGICAL LCHANGE
1705     C
1706           LCHANGE = .FALSE.
1707     C--- Check STOL for sanity
1708           IF(STOL.GT.0.3) THEN
1709            WRITE(*,*) 'SCHECK:  Bad value for small panels (STOL > 0.3)'
1710            RETURN
1711           ENDIF
1712     C
1713      10   DO 20 I = 2, N-2
1714             IM1 = I-1
1715             IP1 = I+1
1716             IP2 = I+2
1717     C
1718             DXM1 = X(I) - X(I-1)
1719             DYM1 = Y(I) - Y(I-1)
1720             DSM1 = SQRT(DXM1*DXM1 + DYM1*DYM1)
1721     C
1722             DXP1 = X(I+1) - X(I)
1723             DYP1 = Y(I+1) - Y(I)
1724             DSP1 = SQRT(DXP1*DXP1 + DYP1*DYP1)
1725     C
1726             DXP2 = X(I+2) - X(I+1)
1727             DYP2 = Y(I+2) - Y(I+1)
1728             DSP2 = SQRT(DXP2*DXP2 + DYP2*DYP2)
1729     C
1730     C------- Don't mess with doubled points (slope breaks)
1731             IF(DSP1.EQ.0.0) GO TO 20
1732     C
1733             IF(DSP1.LT.STOL*DSM1 .OR. DSP1.LT.STOL*DSP2) THEN
1734     C------- Replace node I with average of I and I+1
1735              X(I) = 0.5*(X(I)+X(I+1))
1736              Y(I) = 0.5*(Y(I)+Y(I+1))
1737     C------- Remove node I+1
1738              DO L = I+1, N
1739                X(L) = X(L+1)
1740                Y(L) = Y(L+1)
1741              END DO         
1742              N = N - 1
1743              LCHANGE = .TRUE.
1744              WRITE(*,*) 'SCHECK segment removed at ',I
1745              GO TO 10
1746             ENDIF
1747     C
1748      20   CONTINUE
1749     C
1750           RETURN
1751           END ! SCHECK

1755           SUBROUTINE HALF(X,Y,S,N)
1756     C-------------------------------------------------
1757     C     Halves the number of points in airfoil
1758     C-------------------------------------------------
1759           REAL X(*), Y(*), S(*)
1760     C
1761           K = 1
1762           INEXT = 3
1763           DO 20 I=2, N-1
1764     C------ if corner is found, preserve it.
1765             IF(S(I) .EQ. S(I+1)) THEN
1766               K = K+1
1767               X(K) = X(I)
1768               Y(K) = Y(I)
1769               K = K+1
1770               X(K) = X(I+1)
1771               Y(K) = Y(I+1)
1772               INEXT = I+3
1773             ENDIF
1774     C
1775             IF(I.EQ.INEXT) THEN
1776               K = K+1
1777               X(K) = X(I)
1778               Y(K) = Y(I)
1779               INEXT = I+2
1780             ENDIF
1781     C
1782        20 CONTINUE
1783           K = K+1
1784           X(K) = X(N)
1785           Y(K) = Y(N)
1786     C
1787     C---- set new number of points
1788           N = K
1789     C
1790           RETURN
1791           END ! HALF
