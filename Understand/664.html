473           SUBROUTINE GETCLEN(NAME,CLEN,NCLEN)
474           CHARACTER*(*) NAME, CLEN
475     C--------------------------------------------------
476     C     Looks for substring  "(c=01234***)"
477     C     in the NAME string.  If found, then
478     C     the "***" string is returned in CLEN.
479     C     If not found, then CLEN is returned blank.
480     C--------------------------------------------------
481     C
482           CLEN = ' '
483     C
484           K1 = INDEX( NAME , '(c=' )
485           IF(K1.EQ.0) RETURN
486     C
487           NNAME = LEN(NAME)
488           K2 = INDEX( NAME(K1:NNAME) , ')' ) + K1 - 2
489           IF(K2-K1.LT.3) RETURN
490     C
491           DO K = K1+3, K2
492             IF(INDEX( '0123456789.,)' , NAME(K:K) ) .EQ. 0) THEN
493              CLEN = NAME(K:K2)
494              NCLEN = K2-K+1
495              RETURN
496             ENDIF
497           ENDDO
498     C
499           RETURN
500           END

504           SUBROUTINE MINMAX(NAX,NPOL,NA,CPOL,CPOLPLF)
505           INCLUDE 'PINDEX.INC'
506           DIMENSION NA(NPOL)
507           DIMENSION CPOL(NAX,IPTOT,NPOL), CPOLPLF(3,*)
508     C--------------------------------------------
509     C     Determines max and min limits of polar
510     C     quantities among all polars passed in.
511     C--------------------------------------------
512     C
513           IF(NPOL.LT.1) RETURN
514     C
515           DO K = 1, 4
516             CPOLPLF(1,K) = CPOL(1,K,1)
517             CPOLPLF(2,K) = CPOL(1,K,1)
518           END DO
519     C
520           DO IP=1, NPOL
521             DO K=1, 4
522               DO I=1, NA(IP)
523                 CPOLPLF(1,K) = MIN( CPOL(I,K,IP) , CPOLPLF(1,K) )
524                 CPOLPLF(2,K) = MAX( CPOL(I,K,IP) , CPOLPLF(2,K) )
525               END DO
526             END DO
527           END DO
528     C
529           RETURN
530           END ! MINMAX

1190          &     / LOG( CPOL(NA,ICL)/CPOL(1,ICL) )
1191             IF     (ABS(EX) .LT. 0.25) THEN
1192               IMATYP = 1
1193             ELSEIF (ABS(EX) .LT. 0.75) THEN
1194               IMATYP = 2
1195             ELSE
1196               IMATYP = 3
1197             ENDIF
1198           ENDIF
1199     C
1200           IF(CPOL(NA,IRE)*CPOL(1,IRE) .LE. 0.0) THEN
1201             IRETYP = 1
1202           ELSE
1203             EX = LOG( CPOL(NA,IRE)/CPOL(1,IRE) )
1204          &     / LOG( CPOL(NA,ICL)/CPOL(1,ICL) )
1205             IF     (ABS(EX) .LT. 0.25) THEN
1206               IRETYP = 1
1207             ELSEIF (ABS(EX) .LT. 0.75) THEN
1208               IRETYP = 2
1209             ELSE
1210               IRETYP = 3
1211             ENDIF
1212           ENDIF
1213     C
1214           RETURN
1215           END ! GETTYP

1218           SUBROUTINE RESCAL(SCAL)
1219           INCLUDE 'PPLOT.INC'
1220     C--------------------------------------------
1221     C     Rescales forces and moments
1222     C--------------------------------------------
1223     C---- rescale polar forces by SCAL, moments by SCAL**2

1227               CPOL(I,ICD,IP) = CPOL(I,ICD,IP)*SCAL

1320           subroutine AXISADJ2(xmin,xmax,xspan,deltax,ntics)
1321     C...Make scaled axes with engineering increments between tics
1322     C
1323     C   Input:    xmin, xmax - input range for which scaled axis is desired
1324     C
1325     C   Output:   xmin, xmax - adjusted range for scaled axis
1326     C             xspan      - adjusted span of scaled axis
1327     C             deltax     - increment to be used for scaled axis
1328     C             nincr      - number of tics to be used on axis
1329     C                          note that ntics=1+(xspan/deltax)
1330     C
1331           real    xmin,xmax,xspan,deltax,xinc,xinctbl(4)
1332           integer ntics,i
1333           data    xinctbl / 0.1, 0.2, 0.5, 1. /
1334     c
1335           xspan1 = xmax-xmin
1336           if (xspan1.eq.0.) xspan1 = 1.
1337     c
1338           xpon = ifix(log10(xspan1))
1339           xspan = xspan1 / 10.**xpon
1340     c
1341           do i = 1, 4
1342             xinc = xinctbl(i)
1343             ntics = 1 + ifix(xspan/xinc + 0.1)
1344             if (ntics.LE.6) go to 1
1345           end do
1346     c
1347        1  deltax = xinc*10.**xpon
1348           xmin = deltax*  ifloor2(xmin/deltax)
1349           xmax = deltax*iceiling2(xmax/deltax)
1350           xspan = xmax - xmin
1351           ntics = 1 + ifix(xspan/xinc + 0.1)
1352           return
1353           end
1354     
1355           function iceiling2(x)
1356     c--- returns next highest integer value if fraction is non-zero 
1357           integer iceiling2
1358           real x
1359           i = ifix(x)
1360           if(x-i.GT.0.) i = i+1
1361           iceiling2 = i
1362           return
1363           end
1364     
1365           function ifloor2(x)
1366     c--- returns next lowest integer value if fraction is negative, non-zero
1367           integer ifloor2
1368           real x
1369           i = ifix(x)
1370           if(x-i.LT.0.) i = i-1
1371           ifloor2 = i
1372           return
1373           end
