21           SUBROUTINE DPLOT(NPR1,XPR,YPR)
22           INCLUDE 'XFOIL.INC'     
23     C-----------------------------------------------------------
24     C     Plots analytical profiles at specified points.
25     C     If NPR=0, then cursor-selected points are requested.
26     C-----------------------------------------------------------
27           DIMENSION XPR(*), YPR(*)
28     C
29           CHARACTER*1 KCHAR
30           LOGICAL LCRS, TURB
31           LOGICAL LGUI
32     C
33           CALL GETCOLOR(ICOL0)
34     C
35           LCRS = NPR1 .LE. 0
36     C
37           IF(LCRS) THEN
38            KDONE  = 1
39            XDWIN = XPAGE - 2.0*XMARG
40            YDWIN = YPAGE - 2.0*YMARG
41            X1 = XMARG + 0.91*XDWIN
42            X2 = XMARG + 0.99*XDWIN
43            Y1 = YMARG + 0.01*YDWIN
44            Y2 = YMARG + 0.05*YDWIN
45            CALL NEWPEN(5)
46            CALL GUIBOX(KDONE, X1,X2,Y1,Y2, 'GREEN'   , ' Done ')
47     C
48            WRITE(*,*) ' '
49            WRITE(*,*) 'Locate profiles with cursor, type "D" when done...'
50            NPR = 12345
51     C
52           ELSE
53            NPR = NPR1
54     C
55           ENDIF
56     C
57     C---- go over profiles ...
58           DO 50 IPR=1, NPR
59     C
60             IF(LCRS) THEN 
61     C------- get cursor plot coordinates
62              CALL GETCURSORXY(XC,YC,KCHAR)
63              IF(INDEX('Dd',KCHAR).NE.0 .OR. LGUI(KDONE,XC,YC)) THEN
64               RETURN
65              ENDIF
66     C
67     C------- transform to airfoil coordinates
68              XC = XC/FACA - XOFA
69              YC = YC/FACA - YOFA
70     C
71             ELSE
72              XC = XPR(IPR)
73              YC = YPR(IPR)
74     C
75             ENDIF
76     C
77     C------ find nearest airfoil surface point
78             RSQMIN = 1.0E23
79             ISMIN = 0
80             IBLMIN = 0
81             DOFF = 0.00001*(S(N)-S(1))
82             DO IS = 1, 2
83               DO IBL = 2, IBLTE(IS)
84                 I = IPAN(IBL,IS)
85                 XSURF = X(I) + DOFF*YP(I)
86                 YSURF = Y(I) - DOFF*XP(I)
87                 RSQ = (XC-XSURF)**2 + (YC-YSURF)**2
88                 IF(RSQ .LE. RSQMIN) THEN
89                  RSQMIN = RSQ
90                  ISMIN = IS
91                  IBLMIN = IBL
92                 ENDIF
93               ENDDO
94             ENDDO
95     C
96             IS = ISMIN
97             IBL = IBLMIN
98     C
99             I = IPAN(IBL,IS)
100             CRSP = (XC-X(I))*NY(I) - (YC-Y(I))*NX(I)
101             IF(IS.EQ.2) CRSP = -CRSP
102     C
103             IF(CRSP.GT.0.0) THEN
104              IBLP = IBL+1
105              IBLO = IBL
106             ELSE
107              IBLP = IBL
108              IBLO = IBL-1
109             ENDIF
110             ISP = IS
111             ISO = IS
112     C
113             IF(IBLP.GT.IBLTE(IS)) THEN
114              IBLP = IBLTE(IS)
115              IBLO = IBLP-1
116              IBL = IBLTE(IS)
117             ELSEIF(IBLO.LT.2) THEN
118              IBLO = 2
119              IF(ISO.EQ.1) THEN
120               ISO = 2
121              ELSE
122               ISO = 1
123              ENDIF
124             ENDIF
125     C
126             IP = IPAN(IBLP,ISP)
127             IO = IPAN(IBLO,ISO)
128     C
129     C------ set interpolation fraction at profile location
130             DX = X(IP) - X(IO)
131             DY = Y(IP) - Y(IO)
132             VX = XC - X(IO)
133             VY = YC - Y(IO)
134             FRAC = (DX*VX + DY*VY)/(DX*DX+DY*DY)
135             FRAC = MIN( MAX( FRAC , 0.0 ) , 1.0 )
136     C
137     C------ set averaged displacement vector at profile location
138             CA = FRAC*NY(IP) + (1.0-FRAC)*NY(IO)
139             SA = FRAC*NX(IP) + (1.0-FRAC)*NX(IO)
140             CSMOD = SQRT(CA**2 + SA**2)
141             CA = CA/CSMOD
142             SA = SA/CSMOD
143     C
144             X0 = FRAC*X(IP) + (1.0-FRAC)*X(IO)
145             Y0 = FRAC*Y(IP) + (1.0-FRAC)*Y(IO)
146     C
147             DS = FRAC*DSTR(IBLP,ISP) + (1.0-FRAC)*DSTR(IBLO,ISO)
148             TH = FRAC*THET(IBLP,ISP) + (1.0-FRAC)*THET(IBLO,ISO)
149             UE = FRAC*UEDG(IBLP,ISP) + (1.0-FRAC)*UEDG(IBLO,ISO)
150     C
151             XI = FRAC*XSSI(IBLP,ISP) + (1.0-FRAC)*XSSI(IBLO,ISO)
152             TURB = XI .GT. XSSITR(IS)
153     C
154     C------ 1 / (total enthalpy)
155             HSTINV = GAMM1*(MINF/QINF)**2 / (1.0 + 0.5*GAMM1*MINF**2)
156     C
157     C------ Sutherland's const./To   (assumes stagnation conditions are at STP)
158             HVRAT = 0.35
159     C
160     C------ fill Rtheta arrays
161             UEC = UE * (1.0-TKLAM) / (1.0 - TKLAM*(UE/QINF)**2)
162             HERAT = (1.0 - 0.5*HSTINV*UEC **2)
163          &        / (1.0 - 0.5*HSTINV*QINF**2)
164             RHOE = HERAT ** (1.0/GAMM1)
165             AMUE = SQRT(HERAT**3) * (1.0+HVRAT)/(HERAT+HVRAT)
166             RTHETA = REINF * RHOE*UE*TH/AMUE
167     C
168             AMSQ = UEC*UEC*HSTINV / (GAMM1*(1.0 - 0.5*UEC*UEC*HSTINV))
169             CALL HKIN( DS/TH, AMSQ, HK, DUMMY, DUMMY)
170     C
171             WRITE(*,9100) X0,Y0, DS, RTHETA, HK
172      9100   FORMAT(1X,'x y =', 2F8.4,'    Delta* =', G12.4,
173          &         '    Rtheta =', F10.2,'    Hk =', F9.4)
174     C
175             IF(IS.EQ.1) THEN
176              UDIR = 1.0
177             ELSE
178              UDIR = -1.0
179             ENDIF
180     C
181             UEI = UE/QINF
182             UN = 0.0
183             CALL NEWCOLORNAME('green')
184             UPRWTS = UPRWT*0.5*(S(N)-S(1))
185             CALL PRPLOT(X0,Y0,TH,UEI,UN,HK,RTHETA,AMSQ,TURB,
186          &              -XOFA,-YOFA,FACA,UPRWTS,SA,CA,UDIR)
187        50 CONTINUE
188     C
189           CALL NEWCOLOR(ICOL0)
190           CALL PLFLUSH
191     C
192           RETURN
193           END ! DPLOT

197           SUBROUTINE PRPLOT(X0,Y0,TH,UE,UN,HK,RET,MSQ,TURB,
198          &                  XOFA,YOFA,FACA,UWT,SINA,COSA,UDIR)
199     C-----------------------------------------------------------------
200     C     Plots velocity profile taken from flow solution.
201     C
202     C   X0,Y0  coordinates of point through which profile axis passes
203     C   SA,CA  sin,cos of profile axis angle (cw from vertical)
204     C-----------------------------------------------------------------
205           REAL MSQ
206           LOGICAL TURB
207     C
208           PARAMETER (KPRX=129)
209           DIMENSION XX(KPRX), YY(KPRX), FFS(KPRX), SFS(KPRX)
210     c
211           XMOD(XTMP) = FACA * (XTMP - XOFA)
212           YMOD(YTMP) = FACA * (YTMP - YOFA)
213     C
214           NN = KPRX
215           UO = 1.0
216           DK = HK*TH
217           CT = 0.
218     C
219           IF(TURB) THEN
220     C------ set Spalding + power-law turbulent profile
221             CALL PRWALL(DK,TH,UO,RET,MSQ,CT, BB,
222          &        DE, DE_DS, DE_TH, DE_UO, DE_RT, DE_MS,
223          &        US, US_DS, US_TH, US_UO, US_RT, US_MS,
224          &        HS, HS_DS, HS_TH, HS_UO, HS_RT, HS_MS,
225          &        CF, CF_DS, CF_TH, CF_UO, CF_RT, CF_MS,
226          &        CD, CD_DS, CD_TH, CD_UO, CD_RT, CD_MS,
227          &            CD_CT  )
228     c
229             CALL UWALL(TH,UO,DE,US,RET,CF,BB,   YY,XX,NN)
230     C
231     C------ limit profile height
232             DECORR = 1.5 * (3.15 + 1.72/(HK-1.0) + HK) * TH
233             DO 422 K=NN, 1, -1
234               IF(YY(K) .LE. DECORR) GO TO 423
235      422    CONTINUE
236      423    NN = K
237             DE = YY(K)
238     C
239           ELSE
240     C------ set Falkner-Skan profile
241             INORM = 3
242             ISPEC = 2
243             HSPEC = HK
244             ETAE = 1.5*(3.15 + 1.72/(HK-1.0) + HK)
245             GEO = 1.0
246             CALL FS(INORM,ISPEC,BU,HSPEC,NN,ETAE,GEO,YY,FFS,XX,SFS,DEFS)
247             DE = ETAE*TH
248     C
249             DO 425 K=1, NN
250               YY(K) = YY(K)*TH
251      425    CONTINUE
252     C
253           ENDIF
254     C
255           YAX = 1.1*DE
256     C
257           X1 = X0
258           Y1 = Y0
259           X2 = X0 + YAX*SINA
260           Y2 = Y0 + YAX*COSA
261     C
262     C---- plot axis
263           CALL NEWPEN(1)
264           CALL PLOT(XMOD(X1),YMOD(Y1),3)
265           CALL PLOT(XMOD(X2),YMOD(Y2),2)
266     C
267           DO K=1, NN
268             ULOC = UE + UN*(YY(K)-DK)
269             XX(K) = XX(K)*UE * UWT * UDIR
270     CCC     YY(K) = YY(K)
271           ENDDO
272     C
273     C---- rotate and position profile
274           DO K=1, NN
275             XBAR = XX(K)
276             YBAR = YY(K)
277             XROT = XBAR*COSA + YBAR*SINA + X0
278             YROT = YBAR*COSA - XBAR*SINA + Y0
279             XX(K) = XMOD(XROT)
280             YY(K) = YMOD(YROT)
281           ENDDO
282     C
283           CALL NEWPEN(2)
284           CALL XYLINE(NN,XX,YY,0.0,1.0,0.0,1.0,1)
285     C
286           RETURN
287           END ! PRPLOT
